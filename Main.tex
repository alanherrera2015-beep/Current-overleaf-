\section{Questions 1--10: Impulse (Section 9.1)}

\begin{conceptbox}
\textbf{Key Concepts for This Section:}
\begin{itemize}
\item \textbf{Impulse} $\vec{J}$ is defined as the product of a force and the time interval over which it acts: $\vec{J} = \vec{F}_{\text{avg}} \,\Delta t$.
\item More generally, impulse is the integral of force over time: $\vec{J} = \int_{t_i}^{t_f} \vec{F}\,dt$.
\item Impulse is a \textbf{vector} quantity with units of $\text{N}\cdot\text{s}$ (equivalently, $\text{kg}\cdot\text{m/s}$).
\item The area under a force-vs-time graph equals the impulse delivered.
\item A large force over a short time can deliver the same impulse as a small force over a long time.
\end{itemize}
\end{conceptbox}

\section{Questions 11--20: Momentum and the Impulse-Momentum Theorem (9.2)}

\begin{conceptbox}
\textbf{Core Concepts for Section 9.2:}

\textbf{Momentum} is defined as:
\[
\vec{p} = m\vec{v}
\]
Momentum is a vector quantity with units of kg$\cdot$m/s. It points in the same direction as velocity.

\textbf{The Impulse-Momentum Theorem} states:
\[
\vec{J} = \Delta\vec{p} = \vec{p}_f - \vec{p}_i = m\vec{v}_f - m\vec{v}_i
\]

This is a direct consequence of Newton's second law written as $\vec{F}_{\text{net}} = \frac{d\vec{p}}{dt}$. The net impulse on an object equals its change in momentum.

\textbf{Key relationships:}
\begin{itemize}
    \item A heavier object moving at the same speed has more momentum
    \item A faster object of the same mass has more momentum
    \item Momentum can be changed only by applying an external force over time (impulse)
    \item The impulse-momentum theorem connects dynamics (forces) to kinematics (motion)
\end{itemize}
\end{conceptbox}

\section{Questions 21--40: Solving Impulse and Momentum Problems (9.3--9.5)}

\begin{conceptbox}
\textbf{Problem-Solving Strategy for Impulse and Momentum:}
\begin{enumerate}
\item \textbf{Identify the system:} Determine which object(s) you are analyzing.
\item \textbf{Choose a coordinate system:} Define positive and negative directions consistently.
\item \textbf{Identify the initial and final states:} Determine $\vec{v}_i$ and $\vec{v}_f$ (or $\vec{p}_i$ and $\vec{p}_f$).
\item \textbf{Apply the impulse-momentum theorem:} $\vec{J} = \Delta\vec{p} = m\vec{v}_f - m\vec{v}_i$.
\item \textbf{If force and time are involved:} Use $\vec{J} = \vec{F}_{\text{avg}}\,\Delta t$.
\item \textbf{If a force-vs-time graph is given:} The impulse equals the area under the curve.
\item \textbf{Solve and check:} Verify units and check that the direction of impulse makes physical sense.
\end{enumerate}

\textbf{Important:} Per the course approach, avoid using kinematic equations. Instead, use the definitions of velocity and acceleration, conservation of energy, and forces (including impulse-momentum) to solve problems.
\end{conceptbox}

\section{Questions 41--48: Momentum and Collisions in Two Dimensions (9.6)}

\begin{conceptbox}
\textbf{Two-Dimensional Collisions:} When collisions occur in two dimensions, conservation of momentum must be applied \emph{independently} to each component direction:
\[
\sum p_{x,\text{initial}} = \sum p_{x,\text{final}} \qquad \text{and} \qquad \sum p_{y,\text{initial}} = \sum p_{y,\text{final}}
\]
This gives two equations (one for each component), which can be solved simultaneously. For 2D problems, vector addition of momenta is essential---you cannot simply add magnitudes.
\end{conceptbox}

\section{Questions 74--90: Motion with Constant Acceleration (Section 2.5)}

\begin{conceptbox}
\textbf{Approach Without Kinematic Equations:} Per the course instructions, we do \emph{not} use the standard ``kinematic equations'' (e.g., $x = x_0 + v_0 t + \frac{1}{2}at^2$). Instead, we derive motion results from:

\begin{enumerate}
    \item \textbf{Definition of velocity:} $v = \dfrac{dx}{dt}$ (instantaneous) or $v_{\text{avg}} = \dfrac{\Delta x}{\Delta t}$
    \item \textbf{Definition of acceleration:} $a = \dfrac{dv}{dt}$ (instantaneous) or $a_{\text{avg}} = \dfrac{\Delta v}{\Delta t}$
    \item \textbf{Conservation of energy:} $\frac{1}{2}mv_f^2 = \frac{1}{2}mv_i^2 + W_{\text{net}}$, where $W_{\text{net}} = F_{\text{net}}\cdot d$ for constant forces
    \item \textbf{Newton's second law (forces):} $F_{\text{net}} = ma$
\end{enumerate}

For constant acceleration, the velocity changes linearly with time, so the average velocity equals $\dfrac{v_i + v_f}{2}$.
\end{conceptbox}

\subsection*{Detailed Answers and Explanations}

\subsubsection*{Question 1}

\begin{answerbox}
\textbf{d) 48\,N$\cdot$s}
\end{answerbox}

\begin{explanationbox}
Impulse is defined as $J = F_{\text{avg}} \cdot \Delta t$. Here $F = 12\,\text{N}$ and $\Delta t = 4.0\,\text{s}$, so:
\[
J = (12\,\text{N})(4.0\,\text{s}) = 48\,\text{N}\cdot\text{s}
\]
Note that the mass of the object is not needed to compute impulse directly from force and time. The mass would be needed if you wanted to find the change in velocity.
\end{explanationbox}

\subsubsection*{Question 2}

\begin{answerbox}
\textbf{e) Both (b) and the equivalent N$\cdot$s}
\end{answerbox}

\begin{explanationbox}
Impulse $J = F\,\Delta t$ has units of $\text{N}\cdot\text{s}$. Since $1\,\text{N} = 1\,\text{kg}\cdot\text{m/s}^2$, we have:
\[
\text{N}\cdot\text{s} = \text{kg}\cdot\frac{\text{m}}{\text{s}^2}\cdot\text{s} = \text{kg}\cdot\frac{\text{m}}{\text{s}}
\]
So $\text{N}\cdot\text{s}$ and $\text{kg}\cdot\text{m/s}$ are equivalent and both correct units for impulse. Note that $\text{kg}\cdot\text{m/s}^2$ is the unit of force (newtons), and $\text{N}\cdot\text{m}$ is the unit of energy (joules).
\end{explanationbox}

\subsubsection*{Question 8}

\begin{answerbox}
\textbf{d) The product of force and the time interval during which the force acts}
\end{answerbox}

\begin{explanationbox}
By definition, impulse $\vec{J} = \vec{F}_{\text{avg}}\,\Delta t$. Let's identify the other choices:
\begin{itemize}
\item (a) Rate of change of momentum is \emph{force} (Newton's second law:


$\vec{F} = d\vec{p}/dt$), not impulse itself.
\item (b) Product of mass and velocity is \emph{momentum} $\vec{p} = m\vec{v}$.
\item (c) Product of force and displacement is \emph{work} $W = \vec{F}\cdot\vec{d}$.
\item (e) Product of mass and acceleration is \emph{force} $\vec{F} = m\vec{a}$.
\end{itemize}

Impulse is uniquely defined as force multiplied by the time interval over which it acts. It has units of N$\cdot$s, which are equivalent to kg$\cdot$m/s (the same units as momentum), reflecting the impulse-momentum theorem: $\vec{J} = \Delta\vec{p}$.
\end{explanationbox}

\subsubsection*{Question 10}

\begin{answerbox}
\textbf{b) The force decreases by a factor of 5}
\end{answerbox}

\begin{explanationbox}

\]

This is the fundamental engineering principle behind crumple zones, airbags, and padding: by extending the collision time, the peak force on the human body is reduced, even though the total impulse (change in momentum) is the same. This is a direct, life-saving application of the impulse-momentum theorem.
\end{explanationbox}

\begin{answerbox}

Since both objects have the same kinetic energy $K$:

\subsubsection*{Question 29}

\begin{answerbox}
\textbf{c) When the net external force on the system is zero}
\end{answerbox}

\begin{explanationbox}
Conservation of momentum follows directly from Newton's second law applied to a system:
\[
\vec{F}_{\text{ext, net}} = \frac{d\vec{p}_{\text{total}}}{dt}
\]

If $\vec{F}_{\text{ext, net}} = 0$, then $d\vec{p}_{\text{total}}/dt = 0$, which means $\vec{p}_{\text{total}}$ is constant.

Important distinctions:
\begin{itemize}
\item Internal forces (forces between objects within the system) do \emph{not} change total momentum---they come in Newton's third law pairs that cancel.
\item Friction between colliding objects is an internal force if both objects are in the system.
\item Momentum can be conserved even when kinetic energy is not (inelastic collisions).
\item Momentum conservation does not require equal masses, elasticity, or absence of friction---only zero net external force.
\end{itemize}
\end{explanationbox}

\subsubsection*{Question 33}

\begin{answerbox}
\textbf{c) Momentum is conserved but kinetic energy is not}
\end{answerbox}

\begin{explanationbox}
In a perfectly inelastic collision, the objects stick together after colliding. The key principles are:

\begin{itemize}
\item \textbf{Momentum is always conserved} in collisions where the net external force is zero (or negligible during the collision). This is true for elastic, inelastic, and perfectly inelastic collisions.
\item \textbf{Kinetic energy is NOT conserved} in a perfectly inelastic collision. Some kinetic energy is transformed into thermal energy, sound, and permanent deformation.
\item A perfectly inelastic collision results in the \emph{maximum possible loss} of kinetic energy while still conserving momentum.
\end{itemize}

\begin{center}
\begin{tabular}{lccc}
\toprule
\textbf{Collision Type} & \textbf{Momentum} & \textbf{Kinetic Energy} & \textbf{Objects After} \\
\midrule
Elastic & Conserved & Conserved & Separate \\
Inelastic & Conserved & NOT conserved & Separate \\
Perfectly Inelastic & Conserved & NOT conserved (max loss) & Stick together \\
\bottomrule
\end{tabular}
\end{center}
\end{explanationbox}

\subsubsection*{Question 47}

\begin{answerbox}
\textbf{b) The first particle stops and the second moves at speed $v$ in the $+x$ direction}
\end{answerbox}

\begin{explanationbox}
For a head-on (one-dimensional) elastic collision between equal masses where one is initially stationary, the result is that the moving particle stops completely and the stationary particle moves off with the original speed. This can be proven using both conservation of momentum and conservation of kinetic energy.

Conservation of momentum: $mv = mv_{1f} + mv_{2f} \implies v = v_{1f} + v_{2f}$

Conservation of kinetic energy: $\frac{1}{2}mv^2 = \frac{1}{2}mv_{1f}^2 + \frac{1}{2}mv_{2f}^2 \implies v^2 = v_{1f}^2 + v_{2f}^2$

From the first equation: $v_{2f} = v - v_{1f}$. Substituting into the second:
\[
v^2 = v_{1f}^2 + (v - v_{1f})^2 = v_{1f}^2 + v^2 - 2v\,v_{1f} + v_{1f}^2
\]
\[
0 = 2v_{1f}^2 - 2v\,v_{1f} = 2v_{1f}(v_{1f} - v)
\]

So either $v_{1f} = 0$ or $v_{1f} = v$. The solution $v_{1f} = v$ gives $v_{2f} = 0$ (no collision occurred). Therefore $v_{1f} = 0$ and $v_{2f} = v$.

This is the principle behind Newton's cradle: the first ball stops and the last ball swings out with the same speed.
\end{explanationbox}

\subsubsection*{Question 56}

\begin{answerbox}
\textbf{c) $60^\circ$}
\end{answerbox}

\begin{explanationbox}
For an elastic collision between equal masses where one is initially at rest, there is a remarkable result: \textbf{the two balls always scatter at $90^\circ$ relative to each other}.

This can be proven using conservation of momentum (vector equation) and conservation of kinetic energy:

\textbf{Momentum (vector):} $m\vec{v}_i = m\vec{v}_{1f} + m\vec{v}_{2f}$, so $\vec{v}_i = \vec{v}_{1f} + \vec{v}_{2f}$

\textbf{Kinetic energy:} $\frac{1}{2}mv_i^2 = \frac{1}{2}mv_{1f}^2 + \frac{1}{2}mv_{2f}^2$, so $v_i^2 = v_{1f}^2 + v_{2f}^2$

The momentum equation says $\vec{v}_i = \vec{v}_{1f} + \vec{v}_{2f}$, which means these three vectors form a triangle. The energy equation says $v_i^2 = v_{1f}^2 + v_{2f}^2$, which is the Pythagorean theorem! Therefore the triangle is a right triangle, and $\vec{v}_{1f} \perp \vec{v}_{2f}$.

Since ball 1 scatters at $30^\circ$ above, ball 2 must scatter at $90^\circ - 30^\circ = 60^\circ$ below.
\end{explanationbox}

\subsubsection*{Question 57}

\begin{answerbox}
\textbf{d) $597\,\text{m/s}$}
\end{answerbox}

\begin{explanationbox}
This classic problem requires two steps, each using a different conservation law:

\textbf{Step 1: After collision $\to$ top of swing (Conservation of Energy)}

The
block+bullet system swings upward. No external horizontal forces act, so we use conservation of energy:
\[
\frac{1}{2}(m_{\text{bullet}} + m_{\text{block}})v_{\text{after}}^2 = (m_{\text{bullet}} + m_{\text{block}})gh
\]
\[
v_{\text{after}} = \sqrt{2gh} = \sqrt{2(9.8)(0.45)} = \sqrt{8.82} = 2.97\,\text{m/s}
\]

\textbf{Step 2: Before collision $\to$ after collision (Conservation of Momentum)}

During the collision (very short time), momentum is conserved:
\[
m_{\text{bullet}}\,v_{\text{bullet}} = (m_{\text{bullet}} + m_{\text{block}})\,v_{\text{after}}
\]
\[
(0.010)\,v_{\text{bullet}} = (0.010 + 2.0)(2.97) = (2.010)(2.97) = 5.97
\]
\[
v_{\text{bullet}} = \frac{5.97}{0.010} = 597\,\text{m/s}
\]

\textbf{Why we use different conservation laws for each step:}
\begin{itemize}
\item \textbf{During the collision:} Energy is NOT conserved (inelastic collision---bullet embeds in block). But momentum IS conserved because the collision happens so fast that external forces (gravity, string tension) deliver negligible impulse.
\item \textbf{After the collision:} Momentum is NOT useful (the string exerts a force). But energy IS conserved because only the conservative gravitational force does work.
\end{itemize}

This two-step reasoning---using the right conservation law at the right time---is a hallmark of expert physics problem-solving.
\end{explanationbox}

\subsubsection*{Question 63}

\begin{answerbox}
\textbf{d) $90^\circ$}
\end{answerbox}

\begin{explanationbox}
As proven in Question 56, for an elastic collision between equal masses where one is initially at rest, the two final velocity vectors are always perpendicular. This result follows from combining the vector momentum equation with the scalar energy equation:

\[
\vec{v}_i = \vec{v}_{1f} + \vec{v}_{2f} \quad \text{(momentum)}
\]
\[
v_i^2 = v_{1f}^2 + v_{2f}^2 \quad \text{(energy)}
\]

Squaring the momentum equation: $v_i^2 = v_{1f}^2 + v_{2f}^2 + 2\vec{v}_{1f}\cdot\vec{v}_{2f}$

Comparing with the energy equation: $2\vec{v}_{1f}\cdot\vec{v}_{2f} = 0$

Therefore $\vec{v}_{1f} \perp \vec{v}_{2f}$, meaning the angle between them is exactly $90^\circ$.

This result is independent of the impact parameter (how ``head-on'' or ``glancing'' the collision is). The impact parameter only determines how the $90^\circ$ is divided between the two scattering angles.
\end{explanationbox}

v_y = v_{0y} + a_y \cdot t = 4.0 + (-2.0)(3.0) = 4.0 - 6.0 = -2.0\,\text{m/s}
\]

\textbf{Speed:}
\[
v = \sqrt{v_x^2 + v_y^2} = \sqrt{(3.0)^2 + (-2.0)^2} = \sqrt{9.0 + 4.0} = \sqrt{13} \approx 3.6\,\text{m/s}
\]

The key principle: in 2D motion with constant acceleration, the $x$ and $y$ components of motion are \textbf{independent}. We treat each direction separately using the definition of acceleration, then combine using the Pythagorean theorem.
\end{explanationbox}

\subsubsection*{Question 92}

\begin{answerbox}
\textbf{d) $9.4\,\text{m/s}$}
\end{answerbox}

\begin{explanationbox}
\textbf{Step 1:} Find position at each time:
\[
\vec{r}(1) = (2(1)^2\hat{i} + 3(1)\hat{j}) = (2\hat{i} + 3\hat{j})\,\text{m}
\]
\[
\vec{r}(3) = (2(9)\hat{i} + 3(3)\hat{j}) = (18\hat{i} + 9\hat{j})\,\text{m}
\]

\textbf{Step 2:} Displacement:
\[
\Delta\vec{r} = (18 - 2)\hat{i} + (9 - 3)\hat{j} = (16\hat{i} + 6\hat{j})\,\text{m}

\]

This gives $8.5\,\text{m/s}$, which is answer \textbf{(c)}.

\end{explanationbox}

\subsubsection*{Question 93}

\begin{answerbox}
\textbf{d) Radially inward toward the center}
\end{answerbox}

\begin{explanationbox}
Even though the speed is constant, the velocity is changing because its \textbf{direction} is changing. The acceleration that causes this change in direction is called \textbf{centripetal acceleration} and always points toward the center of the circle.

\textbf{Why inward?} Consider the definition of acceleration: $\vec{a} = \frac{\Delta\vec{v}}{\Delta t}$. If you draw the velocity vectors at two nearby points on the circle and find $\Delta\vec{v} = \vec{v}_f - \vec{v}_i$, the difference vector points toward the center.

\textbf{Common misconception (choice e):} Many students think ``constant speed = no acceleration.'' This is wrong! Acceleration is the rate of change of \emph{velocity} (a vector), not speed (a scalar). A change in direction alone constitutes acceleration.

The magnitude of centripetal acceleration is:
\[
a_c = \frac{v^2}{R}
\]
\end{explanationbox}

\subsubsection*{Question 97}

\]

This acceleration is directed toward the center of the curve. On a flat road, the force providing this centripetal acceleration is static friction between the tires and road:

\subsubsection*{Question 100}

\begin{answerbox}
\textbf{d) The satellite's acceleration is directed radially inward toward Earth's center}
\end{answerbox}

\begin{explanationbox}
A satellite in circular orbit moves at constant speed but constantly changes direction. By the definition of acceleration ($\vec{a} = \frac{\Delta\vec{v}}{\Delta t}$), any change in the velocity vector---even a change in direction alone---constitutes acceleration.

For uniform circular motion, this acceleration is centripetal (center-seeking):
\[
a_c = \frac{v^2}{R}
\]

The force providing this centripetal acceleration is gravity, which always points toward Earth's center. By Newton's second law:
\[
\frac{GMm}{R^2} = \frac{mv^2}{R}
\]

This gives the orbital speed: $v = \sqrt{\frac{GM}{R}}$

\textbf{Common misconception:} There is no ``centrifugal force'' pushing the satellite outward. The satellite is in free fall, constantly falling toward Earth but moving fast enough horizontally that it keeps missing the surface.
\end{explanationbox}

\textbf{Step 1: Find speed at bottom of ramp using energy conservation.}
\[
mgh = \frac{1}{2}mv_1^2 \implies v_1 = \sqrt{2gh} = \sqrt{2(10)(5.0)} = \sqrt{100} = 10\,\text{m/s}
\]

\[
m_1 v_1 + m_2 v_2 = (m_1 + m_2)v_f

\[
(2.0)(10) + (3.0)(0) = (2.0 + 3.0)v_f
\]
\[
20 = 5.0\,v_f \implies v_f = 4.0\,\text{m/s}
\]

\[
KE_i = \frac{1}{2}(2.0)(10)^2 = 100\,\text{J}
\]
\[
KE_f = \frac{1}{2}(5.0)(4.0)^2 = 40\,\text{J}
\]
\[
\text{Energy lost} = 100 - 40 = 60\,\text{J} \quad (60\%\text{ lost})
\]
\end{explanationbox}

\subsubsection*{Question 123}

\begin{answerbox}
\textbf{b) $v_A = 0\,\text{m/s}$, $v_B = 6.0\,\text{m/s}$}
\end{answerbox}

\begin{explanationbox}
For an elastic collision between equal masses where one is initially at rest, there is a beautiful result: \textbf{the moving object stops completely, and the stationary object moves off with the original velocity.}

\textbf{Proof using both conservation laws:}

Conservation of momentum:
\[
m v_0 + m(0) = m v_{Af} + m v_{Bf}
\]
\[
v_0 = v_{Af} + v_{Bf} \quad \Rightarrow \quad 6.0 = v_{Af} + v_{Bf} \quad \text{...(1)}
\]

Conservation of kinetic energy (elastic collision):
\[
\frac{1}{2}mv_0^2 = \frac{1}{2}mv_{Af}^2 + \frac{1}{2}mv_{Bf}^2
\]
\[
v_0^2 = v_{Af}^2 + v_{Bf}^2 \quad \Rightarrow \quad 36 = v_{Af}^2 + v_{Bf}^2 \quad \text{...(2)}
\]

From (1): $v_{Bf} = 6.0 - v_{Af}$. Substituting into (2):
\[
36 = v_{Af}^2 + (6.0 - v_{Af})^2 = v_{Af}^2 + 36 - 12v_{Af} + v_{Af}^2
\]
\[
0 = 2v_{Af}^2 - 12v_{Af} = 2v_{Af}(v_{Af} - 6.0)
\]

Solutions: $v_{Af} = 0$ or $v_{Af} = 6.0$

The solution $v_{Af} = 6.0$ corresponds to no collision (objects pass through each other), so the physical solution is $v_{Af} = 0$, giving $v_{Bf} = 6.0\,\text{m/s}$.

\textbf{Real-world example:} This is exactly what happens in Newton's cradle---the momentum and energy transfer completely from one ball to the next identical ball.
\end{explanationbox}

\subsubsection*{Question 144}

\begin{answerbox}
\textbf{c) $60^\circ$ south of east}
\end{answerbox}

\begin{explanationbox}
\textbf{Key result for elastic collisions between equal masses in 2D:} When a moving object elastically collides with a stationary object of equal mass, the two objects always move off at $90^\circ$ to each other.

If the first ball moves at $30^\circ$ north of east, then the second ball must move at:
\[
90^\circ - 30^\circ = 60^\circ \text{ south of east}
\]

\textbf{Why $90^\circ$?} This can be proven using both conservation of momentum (a vector equation) and conservation of kinetic energy (a scalar equation). For equal masses with one initially at rest:

Momentum conservation: $\vec{v}_0 = \vec{v}_1 + \vec{v}_2$

Energy conservation: $v_0^2 = v_1^2 + v_2^2$

The energy equation is the Pythagorean theorem applied to the momentum triangle! This means $\vec{v}_1 \perp \vec{v}_2$---the velocity vectors after collision are perpendicular.

This result applies \emph{only} to elastic collisions between equal masses. It is beautifully demonstrated in billiards (pool), where the cue ball and object ball separate at approximately $90^\circ$ after a non-head-on collision.
\end{explanationbox}

\begin{answerbox}

\end{answerbox}

\begin{explanationbox}
For an elastic collision between equal masses where one is initially at rest, the result is always a complete transfer of motion: the moving ball stops and the stationary ball moves with the original velocity.

\textbf{Proof using conservation of momentum and energy:}

Let $v_1$ and $v_2$ be the final velocities. Since $m_1 = m_2 = m$:

\textbf{Conservation of momentum:}
\[
mv_0 = mv_1 + mv_2 \implies v_0 = v_1 + v_2
\]
\[
5.0 = v_1 + v_2 \quad \text{...(1)}
\]

\textbf{Conservation of kinetic energy:}
\[
\frac{1}{2}mv_0^2 = \frac{1}{2}mv_1^2 + \frac{1}{2}mv_2^2 \implies v_0^2 = v_1^2 + v_2^2
\]
\[
25 = v_1^2 + v_2^2 \quad \text{...(2)}
\]

From equation (1): $v_2 = 5.0 - v_1$. Substituting into (2):
\[
25 = v_1^2 + (5.0 - v_1)^2 = v_1^2 + 25 - 10v_1 + v_1^2
\]
\[
0 = 2v_1^2 - 10v_1 = 2v_1(v_1 - 5.0)
\]

So $v_1 = 0$ or $v_1 = 5.0\,\text{m/s}$.

The solution $v_1 = 5.0$ corresponds to no collision occurring (the balls pass through each other), which is physically meaningless. Therefore $v_1 = 0$ and $v_2 = 5.0\,\text{m/s}$.

\textbf{Real-world example:} This is the principle behind Newton's cradle---when one ball swings in, it stops and the ball on the opposite end swings out with the same speed.
\end{explanationbox}

\begin{answerbox}
\textbf{d) $2.0\,\text{m/s}$ to the left}
\end{answerbox}

\begin{explanationbox}
For a one-dimensional elastic collision where the target is initially at rest, we can derive the final velocities using both conservation of momentum and conservation of kinetic energy simultaneously.

\textbf{Conservation of momentum:}
\[
m_1 v_{1i} = m_1 v_{1f} + m_2 v_{2f}
\]
\[
(2.0)(6.0) = (2.0)v_{1f} + (4.0)v_{2f}
\]
\[
12 = 2v_{1f} + 4v_{2f} \quad \text{...(1)}
\]

\textbf{Conservation of kinetic energy:}
\[
\frac{1}{2}m_1 v_{1i}^2 = \frac{1}{2}m_1 v_{1f}^2 + \frac{1}{2}m_2 v_{2f}^2
\]
\[
(2.0)(36) = (2.0)v_{1f}^2 + (4.0)v_{2f}^2
\]
\[
72 = 2v_{1f}^2 + 4v_{2f}^2 \quad \text{...(2)}
\]

\textbf{Solving the system:} From equation (1): $v_{1f} = 6 - 2v_{2f}$. Substituting into equation (2):
\[
72 = 2(6 - 2v_{2f})^2 + 4v_{2f}^2
\]
\[
72 = 2(36 - 24v_{2f} + 4v_{2f}^2) + 4v_{2f}^2
\]
\[
72 = 72 - 48v_{2f} + 8v_{2f}^2 + 4v_{2f}^2
\]
\[
0 = -48v_{2f} + 12v_{2f}^2
\]
\[
0 = v_{2f}(-48 + 12v_{2f})
\]

So $v_{2f} = 0$ (trivial, no collision) or $v_{2f} = 4.0\,\text{m/s}$.

Substituting back: $v_{1f} = 6 - 2(4.0) = -2.0\,\text{m/s}$.

The negative sign means the $2.0\,\text{kg}$ ball bounces back, moving at $2.0\,\text{m/s}$ to the \textbf{left}.

\textbf{Using the derived formulas for elastic collisions:}
\[
v_{1f} = \frac{m_1 - m_2}{m_1 + m_2}v_{1i} = \frac{2.0 - 4.0}{2.0 + 4.0}(6.0) = \frac{-2.0}{6.0}(6.0) = -2.0\,\text{m/s}
\]
\[
v_{2f} = \frac{2m_1}{m_1 + m_2}v_{1i} = \frac{2(2.0)}{6.0}(6.0) = \frac{4.0}{6.0}(6.0) = 4.0\,\text{m/s}
\]

\textbf{Verification---is momentum conserved?}
\[
p_i = (2.0)(6.0) = 12\,\text{kg}\cdot\text{m/s}
\]
\[
p_f = (2.0)(-2.0) + (4.0)(4.0) = -4.0 + 16 = 12\,\text{kg}\cdot\text{m/s} \checkmark
\]

\textbf{Verification---is kinetic energy conserved?}
\[
KE_i = \frac{1}{2}(2.0)(6.0)^2 = 36\,\text{J}
\]
\[
KE_f = \frac{1}{2}(2.0)(2.0)^2 + \frac{1}{2}(4.0)(4.0)^2 = 4.0 + 32 = 36\,\text{J} \checkmark
\]

Both conservation laws are satisfied, confirming this is a valid elastic collision.

\textbf{Physical insight:} When a lighter object collides elastically with a heavier stationary object, the lighter object bounces back. When a heavier object hits a lighter one, both move forward. When equal masses collide elastically (with one at rest), the first stops and the second moves with the first's original velocity.
\end{explanationbox}

\section*{Summary of Key Concepts and Methods}

\begin{conceptbox}
\textbf{The Three Pillars of This Course}

Every problem in this course can be solved using combinations of three fundamental tools:

\begin{enumerate}[label=\textbf{\arabic*.}]
    \item \textbf{Definitions of Velocity and Acceleration:}
    \[
    \vec{v} = \frac{\Delta \vec{r}}{\Delta t}, \qquad \vec{a} = \frac{\Delta \vec{v}}{\Delta t}
    \]
    These replace the traditional ``kinematic equations.'' For constant acceleration, the average velocity is $\vec{v}_{\text{avg}} = \frac{\vec{v}_i + \vec{v}_f}{2}$.
    
    \item \textbf{Conservation of Energy:}
    \[
    KE_i + PE_i + W_{\text{non-conservative}} = KE_f + PE_f
    \]
    Use this whenever you need to relate speeds at different positions. It gives you speed (scalar) but not direction.
    
    \item \textbf{Conservation of Momentum:}
    \[
    \vec{p}_{\text{total,initial}} = \vec{p}_{\text{total,final}} \quad \text{(when } \vec{F}_{\text{ext,net}} = 0\text{)}
    \]
    Use this for collisions and explosions. It is a vector equation---conserve each component separately.
\end{enumerate}
\end{conceptbox}

\begin{conceptbox}
\textbf{When to Use Which Tool}

\begin{center}
\begin{tabular}{>{\raggedright\arraybackslash}p{5cm} >{\raggedright\arraybackslash}p{8cm}}
\toprule
\textbf{Situation} & \textbf{Primary Tool} \\
\midrule
Object speeding up/slowing down with known forces & Definitions of $v$ and $a$ (Newton's 2nd law gives $a$) \\
\midrule
Relating speeds at two positions (no collision) & Conservation of Energy \\
\midrule
Collision or explosion & Conservation of Momentum \\
\midrule
Projectile motion (finding range, time, height) & Definitions of $v$ and $a$ (with $a_x = 0$, $a_y = -g$) combined with energy \\
\midrule
Circular motion & Newton's 2nd law ($F_{\text{net}} = mv^2/R$) \\
\midrule
Relative motion & Vector addition of velocities \\
\midrule
Multi-stage problems (e.g., slide then collide then slide) & Different tools for each stage \\
\bottomrule
\end{tabular}
\end{center}
\end{conceptbox}

\begin{importantbox}
\textbf{Critical Reminders:}
\begin{itemize}
    \item \textbf{Momentum is a vector}---always conserve components separately in 2D problems.
    \item \textbf{Energy is a scalar}---direction does not matter; only speed matters.
    \item \textbf{Kinetic energy is NOT conserved in inelastic collisions}, but momentum IS.
    \item \textbf{In elastic collisions}, BOTH momentum and kinetic energy are conserved.
    \item \textbf{Impulse} $= \Delta \vec{p}$, and this equals $\vec{F}_{\text{avg}} \Delta t$.
    \item For projectile motion: horizontal and vertical motions are \textbf{independent}. The only acceleration is $g$ downward.
    \item For circular motion: the net force points toward the center. ``
Centripetal force'' is not a new type of force---it is the label for whatever force(s) provide the center-directed acceleration.
    \item For relative motion: $\vec{v}_{A/C} = \vec{v}_{A/B} + \vec{v}_{B/C}$. Pay careful attention to subscript order.
    \item \textbf{Do NOT use memorized kinematic equations.} Instead, use the definitions of velocity and acceleration, conservation of energy, and Newton's laws.
\end{itemize}
\end{importantbox}

\begin{conceptbox}
\textbf{Problem-Solving Strategy Summary}

For \emph{every} problem in this course, follow these steps:
\begin{enumerate}
    \item \textbf{Draw a picture.} Label all known and unknown quantities.
    \item \textbf{Identify the system.} What object(s) are you analyzing?
    \item \textbf{Identify the stage(s).} Is this a single-stage or multi-stage problem?
    \item \textbf{Choose the right tool for each stage:}
    \begin{itemize}
        \item If you need to relate positions and speeds $\rightarrow$ \textbf{Conservation of Energy}
        \item If there is a collision or explosion $\rightarrow$ \textbf{Conservation of Momentum}
        \item If you need time or direction $\rightarrow$ \textbf{Definitions of $v$ and $a$} (with Newton's 2nd law for $a$)
        \item If circular motion $\rightarrow$ \textbf{Newton's 2nd law} with $a_c = v^2/R$
    \end{itemize}
    \item \textbf{Set up equations.} Write the conservation law or definition for the chosen tool.
    \item \textbf{Solve algebraically first}, then plug in numbers.
    \item \textbf{Check your answer.} Are the units correct? Is the magnitude reasonable? Does the direction make sense?
\end{enumerate}
\end{conceptbox}
