\section*{Problem 1: Shock Absorber Springs}
%==============================================

\begin{problembox}[Problem 1 -- Chapters 8 \& 10: Springs \& Energy]
A car of mass $m = 1200$ kg drives over a speed bump and the front axle (which carries
60\% of the car's weight) compresses the two front shock-absorber springs equally.
Each spring has spring constant $k = 25{,}000$ N/m.

\begin{enumerate}[(a)]
  \item What is the weight supported by the two front springs combined?
  \item If both springs compress by the same amount $x$, find the compression $x$.
  \item How much elastic potential energy is stored in both front springs together?
\end{enumerate}
\end{problembox}

\section*{Solution 1: Shock Absorber Springs}
%==============================================

\begin{solutionbox}[Solution 1(a) -- Weight on front springs]

\textbf{Plain-English idea:} First we need to know how much weight presses down on the
front springs. The problem says the front axle carries 60\% of the car's total weight.

\smallskip
\textbf{Step 1 -- Find the total weight of the car.}

Weight is the force of gravity pulling the car down. We multiply the car's mass by
$g = 9.8$ m/s$^2$. (Every kilogram of mass creates about 9.8 N of weight.)
\begin{equation}
W_{\text{total}} = mg = (1200\;\text{kg})(9.8\;\text{m/s}^2) = 11{,}760\;\text{N}
\end{equation}

\textbf{Step 2 -- Find the front-axle load (60\% of total weight).}
\begin{equation}
W_{\text{front}} = 0.60 \times 11{,}760 = \boxed{7{,}056\;\text{N}}
\end{equation}
\textit{Think of it like splitting a pizza: the front gets 6 slices out of 10.}

\end{solutionbox}

\begin{solutionbox}[Solution 1(b) -- Spring compression]

\textbf{Plain-English idea:} The two front springs share the load equally. We use
Hooke's Law: a stiffer spring (bigger $k$) compresses less for the same force.

\smallskip
\textbf{Step 1 -- Force on each spring (equal sharing).}
\begin{equation}
F_{\text{each}} = \frac{7{,}056}{2} = 3{,}528\;\text{N}
\end{equation}

\textbf{Step 2 -- Hooke's Law: $F = kx$, so $x = F/k$.}

Rearranging Hooke's Law tells us how far the spring squashes down:
\begin{equation}
x = \frac{F_{\text{each}}}{k} = \frac{3{,}528}{25{,}000} = \boxed{0.141\;\text{m} = 14.1\;\text{cm}}
\end{equation}
\textit{About the width of your hand -- that's a realistic shock-absorber compression!}

\end{solutionbox}

\begin{solutionbox}[Solution 1(c) -- Stored elastic potential energy]

\textbf{Plain-English idea:} A squeezed spring stores energy like a coiled-up toy.
The formula is $U = \frac{1}{2}kx^2$ -- a stiffer spring or bigger compression stores
more energy.

\smallskip
\textbf{Step 1 -- Energy stored in one spring.}
\begin{equation}
U_{\text{one}} = \frac{1}{2}kx^2 = \frac{1}{2}(25{,}000)(0.141)^2
= \frac{1}{2}(25{,}000)(0.01988) \approx 249\;\text{J}
\end{equation}

\textbf{Step 2 -- Energy in both springs.}
\begin{equation}
U_{\text{total}} = 2 \times 249 = \boxed{498\;\text{J}}
\end{equation}
\textit{About the energy in one bite of food -- small, but enough to bounce the car
back up after the bump!}

\end{solutionbox}

\section*{Solution 2: Force on Shoulder Joint}
%==============================================

\begin{solutionbox}[Solution 2(a) -- Free Body Diagram description]

\textbf{Plain-English idea:} Before any math, we draw a picture of every force on the
forearm. Think of the forearm as a seesaw (lever) balanced at the shoulder joint.

\smallskip
Forces acting on the forearm:
\begin{itemize}
  \item \textbf{Muscle force} $F_M$ -- upward (at $15^\circ$ from vertical),
        12 cm from the shoulder joint. This is the force we want to find.
  \item \textbf{Forearm weight} $W_{\text{arm}} = m_{\text{arm}}g = (1.6)(9.8) = 15.7$ N
        -- downward, at the midpoint (17.5 cm from joint).
  \item \textbf{Book weight} $W_{\text{book}} = mg = (3.0)(9.8) = 29.4$ N
        -- downward, at the far end (35 cm from joint).
  \item \textbf{Joint reaction} $\vec{R}$ -- at the shoulder joint (direction unknown).
\end{itemize}

\end{solutionbox}

\begin{solutionbox}[Solution 2(b) \& (c) -- Torque equation and muscle force]

\textbf{Plain-English idea:} We take torques (twisting forces) about the shoulder joint.
By choosing the joint as the pivot, the unknown joint force $\vec{R}$ has zero
``lever arm'' (distance from pivot), so it creates zero torque and vanishes from the
equation. This is the key trick that makes the problem solvable with one equation!

\smallskip
\textbf{Torque sign rule:} Counterclockwise (CCW) $= +$, Clockwise (CW) $= -$.

\smallskip
\textbf{Torques about the shoulder joint:}
\begin{itemize}
  \item Muscle (CCW, $+$): The perpendicular part of the muscle force is
        $F_M\cos15^\circ$ (the component that actually creates rotation).
        \[ \tau_M = +F_M\cos 15^\circ \times (0.12\;\text{m}) \]
  \item Forearm weight (CW, $-$):
        \[ \tau_{\text{arm}} = -15.7\;\text{N} \times 0.175\;\text{m} \]
  \item Book weight (CW, $-$):
        \[ \tau_{\text{book}} = -29.4\;\text{N} \times 0.35\;\text{m} \]
\end{itemize}

\textbf{Set $\sum \tau = 0$ (no rotation $\Rightarrow$ torques cancel):}
\begin{align}
F_M\cos 15^\circ \times 0.12 &= 15.7 \times 0.175 + 29.4 \times 0.35 \\
F_M(0.9659)(0.12) &= 2.748 + 10.29 \\
0.11591\,F_M &= 13.04
\end{align}
\begin{equation}
F_M = \frac{13.04}{0.11591} = \boxed{112.5\;\text{N}}
\end{equation}
\textit{That's about 25 lbs of muscle force just to hold a 3 kg book -- your shoulder
works very hard!}

\end{solutionbox}

\begin{solutionbox}[Solution 2(d) -- Reaction force at the shoulder joint]

\textbf{Plain-English idea:} Now use force balance ($\sum F = 0$) to find the joint
reaction. The joint must push back with whatever force is needed to keep the forearm
from accelerating.

\smallskip
Decompose the muscle force into components:
\begin{itemize}
  \item Vertical (upward): $F_{M,y} = F_M\cos15^\circ = 112.5 \times 0.9659 = 108.7$ N
  \item Horizontal (toward body): $F_{M,x} = F_M\sin15^\circ = 112.5 \times 0.2588 = 29.1$ N
        (the muscle pulls slightly inward as well as upward)
\end{itemize}

\textbf{Vertical force balance ($\sum F_y = 0$):}
\begin{align}
R_y + F_{M,y} - W_{\text{arm}} - W_{\text{book}} &= 0 \\
R_y &= 15.7 + 29.4 - 108.7 = -63.6\;\text{N}
\end{align}
\textit{Negative means $R_y$ points downward -- the joint pulls down on the arm to
prevent it flying upward from the strong muscle pull.}

\textbf{Horizontal force balance ($\sum F_x = 0$):}
\begin{equation}
R_x = -F_{M,x} = -29.1\;\text{N}
\end{equation}

\textbf{Magnitude of joint force:}
\begin{equation}
|\vec{R}| = \sqrt{(29.1)^2 + (63.6)^2} = \sqrt{847 + 4045} = \sqrt{4892}
= \boxed{69.9\;\text{N}}
\end{equation}

\end{solutionbox}

\section*{Solution 3: Superman Rescue}
%==============================================

\begin{solutionbox}[Solution 3(a) -- Superman's speed before the catch]

\textbf{Plain-English idea:} As Superman falls, gravity converts his stored height
(potential energy) into speed (kinetic energy) -- like a roller coaster, the higher
you start, the faster you go. Since no friction acts, total energy is conserved.

\smallskip
\textbf{Setup:} Superman starts from rest ($v = 0$) at $H = 150$ m and falls
70 m (to the height of the person: $150 - 80 = 70$ m above ground), so
$\Delta h_S = 70$ m.

\textbf{Energy conservation (starts from rest, so $KE_i = 0$):}
\begin{equation}
Mg\,\Delta h_S = \frac{1}{2}Mv_S^2
\end{equation}
\textit{(All potential energy lost becomes kinetic energy gained.)}

\textbf{Cancel $M$ and solve for $v_S$:}
\begin{equation}
v_S = \sqrt{2g\,\Delta h_S} = \sqrt{2(9.8)(70)} = \sqrt{1372}
= \boxed{37.0\;\text{m/s (downward)}}
\end{equation}

\end{solutionbox}

\begin{solutionbox}[Solution 3(b) -- Combined speed after the catch]

\textbf{Plain-English idea:} The moment Superman grabs the person is like a football
tackle -- it's a collision where they stick together. In any collision, total
\emph{momentum} (mass $\times$ velocity) is conserved, even if energy is not.

\smallskip
\textbf{Momentum before the catch} (taking downward as positive):
\begin{itemize}
  \item Superman: $p_S = Mv_S = (100)(37.0) = 3700$ kg$\cdot$m/s
  \item Person: $p_p = mv_0 = (65)(39.6) = 2574$ kg$\cdot$m/s
\end{itemize}

\textbf{Conservation of momentum:}
\begin{equation}
(M + m)\,v_f = M\,v_S + m\,v_0
\end{equation}
\begin{align}
v_f &= \frac{3700 + 2574}{100 + 65} = \frac{6274}{165}
= \boxed{38.0\;\text{m/s (downward)}}
\end{align}
\textit{The combined speed is between Superman's (37.0) and the person's (39.6) speeds --
just like two shopping carts colliding and moving together.}

\end{solutionbox}

\begin{solutionbox}[Solution 3(c) -- Work done by Superman's muscles]

\textbf{Plain-English idea:} After the catch, Superman acts as a brake: his muscles
push upward to slow them down over the remaining 70 m to ground. We use the
work-energy theorem:
\begin{center}
\textit{``Total work done = change in kinetic energy.''}
\end{center}

\textbf{Step 1 -- Initial kinetic energy (just after catch).}
\begin{equation}
KE_i = \frac{1}{2}(M+m)v_f^2 = \frac{1}{2}(165)(38.0)^2
= \frac{1}{2}(165)(1444) = 119{,}130\;\text{J}
\end{equation}

\textbf{Step 2 -- Work by each force over the 70 m descent.}
\begin{itemize}
  \item Gravity (downward, same direction as motion $\to$ positive work):
  \[W_g = (M+m)g\,h_1 = (165)(9.8)(70) = 113{,}190\;\text{J}\]
  \item Superman's muscles (upward, opposite to downward motion $\to$ negative work):
  \[W_{\text{muscles}} = \; ?\]
\end{itemize}

\textbf{Step 3 -- Work-energy theorem ($KE_f = 0$, they stop).}
\begin{align}
W_{\text{net}} &= KE_f - KE_i = 0 - 119{,}130 = -119{,}130\;\text{J}\\
W_g + W_{\text{muscles}} &= -119{,}130\\
113{,}190 + W_{\text{muscles}} &= -119{,}130
\end{align}
\begin{equation}
W_{\text{muscles}} = -119{,}130 - 113{,}190 = \boxed{-232{,}320\;\text{J}}
\end{equation}

\textit{The negative sign means Superman's muscles absorb (remove) energy, acting like
a powerful brake. He must absorb over 232 kJ -- about the energy stored in 60
AA batteries -- to bring them safely to the ground!}

\end{solutionbox}

\vspace{1.5em}

%==============================================
\section*{Solution A: Two-Spring Diving Board}
%==============================================

\begin{solutionbox}[Solution A -- Two-Spring System]

\textbf{Plain-English idea:} The board is perfectly still (not rotating, not moving).
That means two things are true at the same time: (1) all upward forces equal all
downward forces, and (2) all clockwise twists equal all counterclockwise twists. We
use these two facts to write two equations and find two unknowns.

\smallskip
\textbf{Known forces:}
\begin{itemize}
  \item Spring 1 pushes up with $F_1 = 8000\,\delta_1$ at $x = 0$
  \item Spring 2 pushes up with $F_2 = 12{,}000\,\delta_2$ at $x = 2.0$ m
  \item Board weight: $W_B = (25)(9.8) = 245$ N down at $x = 1.5$ m (center)
  \item Diver weight: $W_D = (70)(9.8) = 686$ N down at $x = 3.0$ m
\end{itemize}

\textbf{Equation 1 -- Force balance ($\sum F_y = 0$):}
\begin{equation}
8000\,\delta_1 + 12{,}000\,\delta_2 = 245 + 686 = 931\;\text{N}
\label{eq:forceA}
\end{equation}

\textbf{Equation 2 -- Torque balance about $x = 0$ (CCW = +).}

\textit{Smart trick: choosing the pivot at Spring 1's location makes $F_1$ drop out
(zero moment arm)!}
\begin{align}
12{,}000\,\delta_2 \times 2.0 &= 245 \times 1.5 + 686 \times 3.0 \\
24{,}000\,\delta_2 &= 367.5 + 2058 = 2425.5
\end{align}
\begin{equation}
\delta_2 = \frac{2425.5}{24{,}000} = \boxed{0.101\;\text{m} = 10.1\;\text{cm}}
\end{equation}

\textbf{Substitute into Equation~\eqref{eq:forceA}:}
\begin{align}
8000\,\delta_1 + 12{,}000(0.101) &= 931 \\
8000\,\delta_1 &= 931 - 1212 = -281
\end{align}
\begin{equation}
\delta_1 = \frac{-281}{8000} = \boxed{-0.035\;\text{m}}
\end{equation}

\textbf{Physical meaning:} The negative sign means Spring 1 is \emph{stretched}
(in tension) by 3.5 cm -- it is pulling the board down on the left side. This makes
sense because the heavy diver on the right end tips the board so much that the left
end rises.

\end{solutionbox}

\vspace{1.5em}

%==============================================
\section*{Solution B: Variable-Force Braking}
%==============================================

\begin{solutionbox}[Solution B(a) -- Stopping distance with variable force]

\textbf{Plain-English idea:} The braking force gets stronger the further the car
travels. To find the total work done by a force that changes, we use an integral --
think of it as adding up many tiny pushes over many tiny distances.

\smallskip
\textbf{Step 1 -- Initial kinetic energy (energy the car must lose to stop).}
\begin{equation}
KE_i = \frac{1}{2}mv_0^2 = \frac{1}{2}(1500)(25)^2 = 468{,}750\;\text{J}
\end{equation}
\textit{(The car has nearly half a million joules of energy -- that's a lot to remove!)}

\textbf{Step 2 -- Work done by braking force over stopping distance $d$.}

The force $F(x) = 5000 + 200x$ increases with distance. Adding up all the tiny
contributions:
\begin{equation}
W_{\text{brake}} = \int_0^{d}(5000 + 200x)\,dx
= \left[5000x + 100x^2\right]_0^{d}
= 5000d + 100d^2
\end{equation}

\textbf{Step 3 -- Work-energy theorem ($KE_f = 0$).}
\begin{align}
5000d + 100d^2 &= 468{,}750 \\
d^2 + 50d - 4687.5 &= 0
\end{align}

\textbf{Step 4 -- Quadratic formula.}

For $d^2 + 50d - 4687.5 = 0$, use $d = \dfrac{-b + \sqrt{b^2 - 4ac}}{2a}$
(positive root only -- distance is always positive):
\begin{equation}
d = \frac{-50 + \sqrt{2500 + 18750}}{2} = \frac{-50 + \sqrt{21250}}{2}
= \frac{-50 + 145.8}{2} = \boxed{47.9\;\text{m}}
\end{equation}

\end{solutionbox}

\begin{solutionbox}[Solution B(b) -- Constant-force comparison]

\textbf{Plain-English idea:} If the braking force stayed constant at $F_0 = 5000$ N
(no increase with distance), the math is much simpler -- just divide the car's energy
by the constant force:
\begin{equation}
d_{\text{constant}} = \frac{KE_i}{F_0} = \frac{468{,}750}{5000}
= \boxed{93.75\;\text{m}}
\end{equation}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcc}
\toprule
\textbf{Braking Type} & \textbf{Formula} & \textbf{Stopping Distance}\\
\midrule
Variable force ($5000 + 200x$) & integral & 47.9 m\\
Constant force ($5000$ N) & $KE / F_0$ & 93.75 m\\
\bottomrule
\end{tabular}
\end{center}

\textbf{Conclusion:} The increasing force cuts the stopping distance roughly in half.
This is why anti-lock brakes and modern braking systems are so effective -- the harder
you push, the more braking force you get.

\end{solutionbox}

\section*{Common Exam Mistakes to Avoid}
%==============================================

\begin{warningbox}[Top Mistakes on PHYSICS 1101 Exams]
\begin{enumerate}
  \item \textbf{Torque sign errors:} Always define CCW as $+$ and CW as $-$, and stick
        to it throughout the whole problem.
  \item \textbf{Forgetting distributed weights:} A beam or board has its own weight
        acting at its \emph{geometric center} -- not at the endpoints!
  \item \textbf{Wrong pivot choice:} Put the pivot where an unknown force acts;
        that force's moment arm becomes zero so it disappears from the equation.
  \item \textbf{Mixing up moment of inertia formulas:} Disk ($\frac{1}{2}MR^2$),
        ring ($MR^2$), rod about center ($\frac{1}{12}ML^2$), rod about end
        ($\frac{1}{3}ML^2$).
  \item \textbf{Momentum vs.\ energy:} Use \emph{momentum conservation} for
        collisions (even inelastic ones). Use \emph{energy conservation} when there
        is no collision.
  \item \textbf{Integration limits:} For stopping problems integrate from 0 to $d$.
        Always check that the limits match the physical situation.
  \item \textbf{Unit errors:} Convert cm to m, grams to kg, before plugging into
        any formula. SI units only: kg, m, s, N, J.
  \item \textbf{Quadratic sign trap:} Always check both roots and discard the
        negative one (distances and speeds must be positive).
\end{enumerate}
\end{warningbox}
