\section{Introduction to Torque (Section 7.3)}

\begin{diagbox}{1. Introduction to Torque}

\textbf{Definition:} \emph{Torque} (symbol $\tau$) is the rotational analog of force.  It measures how effectively a force causes an object to rotate about a pivot point.

\vspace{0.3cm}
\textbf{General formula:}
\[
  \tau = r\,F\sin\theta
\]
where $r$ is the distance from the pivot to the point of force application, $F$ is the force magnitude, and $\theta$ is the angle between $\vec{r}$ and $\vec{F}$.

\vspace{0.3cm}
\textbf{Units:} newton-meters (N$\cdot$m).

\vspace{0.3cm}
\textbf{Sign convention:}
\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Counterclockwise (CCW)} torques are \textbf{positive} ($+$).
  \item \textbf{Clockwise (CW)} torques are \textbf{negative} ($-$).
\end{itemize}

\end{diagbox}

\subsection{Types of Torque Calculations}

\begin{diagbox}{1.1 Types of Torque Calculations}

\textbf{Case 1 --- Perpendicular force} ($\theta = 90^\circ$):
\[
  \tau = rF
\]
\emph{Example:} A 20\,N force perpendicular to a 0.3\,m wrench gives $\tau = (0.3)(20) = 6$\,N$\cdot$m.

\vspace{0.4cm}
\textbf{Case 2 --- Force at an angle $\theta$:}
\[
  \tau = rF\sin\theta
\]
\emph{Example:} A 20\,N force at $30^\circ$ on a 0.3\,m wrench gives $\tau = (0.3)(20)\sin 30^\circ = 3$\,N$\cdot$m.

\vspace{0.4cm}
\textbf{Case 3 --- Using the moment arm:}
\[
  \tau = d_\perp F
\]
where $d_\perp = r\sin\theta$ is the perpendicular distance from the pivot to the line of action.

\vspace{0.4cm}
\textbf{Case 4 --- Net torque:}
\[
  \tau_{\rm net} = \sum_i \tau_i
\]
Each torque is signed before summing.

\end{diagbox}

\subsection{Key Properties of Torque}

\begin{diagbox}{1.2 Key Properties of Torque}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Torque depends on the pivot point.}  The same force produces different torques for different pivot choices.
  \item \textbf{Maximum torque} is achieved when the force is applied perpendicular to the lever arm ($\theta = 90^\circ$).
  \item \textbf{Zero torque} results when $\theta = 0^\circ$ or $180^\circ$.
  \item \textbf{Longer lever arm $\Rightarrow$ larger torque} for the same force.
  \item \textbf{Newton's second law for rotation:}
        \[
          \tau_{\rm net} = I\alpha
        \]
        where $I$ is the moment of inertia and $\alpha$ is the angular acceleration.
\end{enumerate}

\end{diagbox}

\section{Gravitational Torque and the Center of Gravity (Section 7.4)}

\begin{diagbox}{2. Gravitational Torque and the Center of Gravity}

The entire gravitational force on an extended object can be treated as acting at a single point: the \textbf{center of gravity} (CG).

\vspace{0.3cm}
\textbf{Gravitational torque} about a pivot at horizontal position $x_0$:
\[
  \tau_{\rm grav} = (x_{\rm CG} - x_0)\,M g
\]
where $M$ is the total mass and $x_{\rm CG}$ is the CG's horizontal coordinate.

\end{diagbox}

\subsection{Finding the Center of Gravity}

\begin{diagbox}{2.1 Finding the Center of Gravity}

For a system of particles with masses $m_1, m_2, \ldots$ at positions $x_1, x_2, \ldots$:
\[
  x_{\rm CG} = \frac{\sum_i m_i\,x_i}{\sum_i m_i}
  \qquad
  y_{\rm CG} = \frac{\sum_i m_i\,y_i}{\sum_i m_i}
\]

\textbf{Worked example:}
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
\textbf{Mass} & \textbf{Value (kg)} & \textbf{Position (m)} \\
\hline
$m_1$ & 2.0 & 0.0 \\
$m_2$ & 3.0 & 1.0 \\
$m_3$ & 1.0 & 2.0 \\
\hline
\end{tabular}
\end{center}
\[
  x_{\rm CG} = \frac{(2.0)(0.0)+(3.0)(1.0)+(1.0)(2.0)}{6.0} = \frac{5}{6} \approx 0.83\text{ m}
\]

\textbf{Tip:} For a uniform symmetric object (rod, disk), the CG is at the geometric center.

\end{diagbox}

\subsection{Key Properties of Center of Gravity}

\begin{diagbox}{2.2 Key Properties of Center of Gravity}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Unique point:} Every object has exactly one center of gravity (in a uniform field).
  \item \textbf{Need not lie within the object:} For a hollow ring or boomerang, the CG is in empty space.
  \item \textbf{Stability criterion:} An object supported at a single point is in equilibrium when the support is directly below (or above) the CG.
  \item \textbf{CG vs.\ CM:} In a uniform gravitational field, the center of gravity and center of mass coincide.
  \item \textbf{Composite objects:} The CG of a composite object can be found by treating each part as a point mass at its own CG.
\end{enumerate}

\end{diagbox}

\section{Torque and Static Equilibrium (Section 8.1)}

\begin{diagbox}{3. Torque and Static Equilibrium}

An object is in \textbf{static equilibrium} when both translational and rotational conditions hold:

\vspace{0.3cm}
\textbf{Condition 1 --- Translational equilibrium:}
\[
  \sum F_x = 0 \quad\text{and}\quad \sum F_y = 0
\]

\textbf{Condition 2 --- Rotational equilibrium:}
\[
  \sum \tau = 0 \quad\text{(about any chosen pivot)}
\]

\vspace{0.4cm}
\textbf{Problem-solving strategy:}
\begin{enumerate}[leftmargin=1.5em]
  \item Draw a free-body diagram with all forces and their points of application.
  \item Choose a pivot strategically (where an unknown force acts, to eliminate it).
  \item Write $\sum\tau = 0$ about the pivot.
  \item Write $\sum F_x = 0$ and $\sum F_y = 0$.
  \item Solve the system of equations.
\end{enumerate}

\vspace{0.4cm}
\textbf{Worked example:} A uniform 4.0-m, 10-kg beam rests on a pivot at its left end.  A 30-kg block sits 3.0\,m from the left end.  Find upward force $F$ at the right end.

Taking the pivot at the left end:
\[
  F(4.0) - (10)(9.8)(2.0) - (30)(9.8)(3.0) = 0
  \;\Rightarrow\;
  F = \frac{196+882}{4} \approx 270\text{ N}
\]

\end{diagbox}

\section{Stability and Balance (Section 8.2)}

\begin{diagbox}{4. Stability and Balance}

\textbf{Stable equilibrium:}  A small displacement raises the CG and produces a restoring torque.

\textbf{Unstable equilibrium:}  A small displacement lowers the CG and causes further departure.

\textbf{Neutral equilibrium:}  A displacement does not change the CG height; no restoring or destabilizing torque.

\vspace{0.4cm}
\textbf{Tipping condition:}  An object tips when its CG moves outside the \emph{base of support}.

\end{diagbox}

\subsection{Factors Affecting Stability}

\begin{diagbox}{4.1 Factors Affecting Stability}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Width of the base of support:} A wider base makes tipping less likely.
  \item \textbf{Height of the center of gravity:} A lower CG increases stability.
  \item \textbf{Position of the CG over the base:} Maximum stability when CG is above the center of the base.
  \item \textbf{Dynamic stability:} Moving objects can maintain balance through gyroscopic effects.
\end{enumerate}

\vspace{0.4cm}
\textbf{Example:} A 1.2-m-tall crate (CG at 0.6\,m, base width 0.4\,m) tips when tilted at angle $\theta$:
\[
  \tan\theta_{\rm tip} = \frac{0.2\text{ m}}{0.6\text{ m}} \approx 0.333
  \;\Rightarrow\;
  \theta_{\rm tip} \approx 18.4^\circ
\]

\end{diagbox}

\section{Springs and Hooke's Law (Section 8.3)}

\begin{diagbox}{5. Springs and Hooke's Law}

\textbf{Hooke's Law:}
\[
  F_{\rm sp} = -k\,x
\]
\begin{itemize}[leftmargin=1.5em]
  \item $k$ = spring constant (N/m)
  \item $x$ = displacement from natural (equilibrium) length
  \item Negative sign: restoring force opposes displacement
\end{itemize}

\textbf{Example:} Spring with $k = 200$\,N/m stretched 0.05\,m: $|F_{\rm sp}| = (200)(0.05) = 10$\,N.

\begin{warnbox}
Hooke's Law holds only within the \textbf{elastic limit}.  Beyond this limit the spring is permanently deformed.
\end{warnbox}

\end{diagbox}

\subsection{Graphical Interpretation of Hooke's Law}

\begin{diagbox}{5.1 Graphical Interpretation of Hooke's Law}

On a graph of applied force $F$ vs.\ displacement $x$:
\begin{itemize}[leftmargin=1.5em]
  \item The relationship is \textbf{linear} for an ideal spring.
  \item The \textbf{slope} equals the spring constant $k$.
  \item The \textbf{area under the curve} equals the elastic potential energy stored: $U_{\rm sp} = \tfrac{1}{2}kx^2$.
  \item Beyond the elastic limit, the graph curves.
\end{itemize}

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|l|l|}
\hline
\textbf{Graph feature} & \textbf{Physical meaning} \\
\hline
Slope $= k$ & Spring stiffness (N/m) \\
Area under curve to $x$ & Work done / elastic PE = $\tfrac{1}{2}kx^2$ \\
Linear region & Valid range of Hooke's Law \\
Deviation from linearity & Beyond elastic limit \\
\hline
\end{tabular}
\end{center}

\end{diagbox}

\section{Forces and Torques in the Body (Section 8.5) --- \textsc{Bio}}

\begin{diagbox}{6. Forces and Torques in the Body (BIO)}

The human musculoskeletal system functions as a collection of \textbf{levers}: rigid bones rotate about pivot joints.

\vspace{0.3cm}
\textbf{Lever elements:}
\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Effort} $F_E$: force applied by muscle.
  \item \textbf{Load} $F_L$: force being overcome.
  \item \textbf{Fulcrum}: the joint (pivot point).
\end{itemize}

\textbf{Mechanical advantage:} $\displaystyle\text{MA} = \frac{d_E}{d_L} = \frac{F_L}{F_E}$

\end{diagbox}

\subsection{Types of Levers in the Body}

\begin{diagbox}{6.1 Types of Levers in the Body}

\begin{center}
\renewcommand{\arraystretch}{1.8}
\begin{tabular}{|c|p{3cm}|p{3.5cm}|p{4cm}|}
\hline
\textbf{Class} & \textbf{Arrangement} & \textbf{Body Example} & \textbf{Key Feature} \\
\hline
\textbf{1st Class} & Fulcrum between effort and load & Head nodding: neck muscles (effort), skull weight (load), atlas vertebra (fulcrum) & Can have MA $>1$ or $<1$; changes force direction \\
\hline
\textbf{2nd Class} & Load between fulcrum and effort & Rising on tiptoe: ball of foot (fulcrum), body weight (load), calf/Achilles tendon (effort) & MA $>1$; effort arm $>$ load arm \\
\hline
\textbf{3rd Class} & Effort between fulcrum and load & Forearm curl: elbow (fulcrum), bicep $\approx$5\,cm from joint (effort), hand/weight $\approx$35\,cm (load) & MA $<1$; trades force for speed and range \\
\hline
\end{tabular}
\end{center}

\vspace{0.4cm}
\textbf{Why muscles exert large forces:}  Most body levers are 3rd-class (MA $\approx 0.14$), so the muscle must exert about 7$\times$ the load force.

\end{diagbox}

\section{The Basic Energy Model (Section 10.1)}

\begin{diagbox}{7. The Basic Energy Model}

\textbf{Energy} is the capacity of a system to do work or cause change.

\vspace{0.3cm}
\textbf{Mechanical energy forms:}
\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Kinetic energy} $K = \tfrac{1}{2}mv^2$
  \item \textbf{Gravitational PE} $U_g = mgy$
  \item \textbf{Elastic PE} $U_{\rm sp} = \tfrac{1}{2}kx^2$
\end{itemize}

\textbf{Non-mechanical:} Thermal energy $E_{\rm th}$ (microscopic KE/PE of atoms).

\textbf{Master energy equation (preview):}
\[
  K_f + U_f + \Delta E_{\rm th} = K_i + U_i + W_{\rm ext}
\]

\end{diagbox}

\subsection{System and Environment}

\begin{diagbox}{7.1 System and Environment}

\begin{itemize}[leftmargin=1.5em]
  \item \textbf{System:} the objects under study.
  \item \textbf{Environment:} everything outside the system.
  \item Energy transfers via \emph{work} (mechanical) or \emph{heat} (thermal).
\end{itemize}

\textbf{Isolated system:} no energy enters or leaves; total energy is conserved.

\textbf{Choosing the system wisely} simplifies problems: including interacting objects avoids internal forces in the energy equation.

\end{diagbox}

\section{Work (Section 10.2)}

\begin{diagbox}{8. Work}

\textbf{Work} is energy transferred by a force over a displacement:
\[
  W = F\,d\cos\theta
\]
where $\theta$ is the angle between $\vec{F}$ and $\vec{d}$.

\textbf{Sign:}
\begin{itemize}[leftmargin=1.5em]
  \item $W > 0$: force component in direction of motion (energy added).
  \item $W < 0$: force component opposite to motion (energy removed).
  \item $W = 0$: force perpendicular to motion.
\end{itemize}

\textbf{Units:} joules (J); $1\,\text{J} = 1\,\text{N}\cdot\text{m}$.

\textbf{Example:} 50-N force at $37^\circ$ over 4.0\,m: $W = (50)(4.0)\cos37^\circ \approx 160$\,J.

\end{diagbox}

\subsection{Work Done by Gravity on an Incline}

\begin{diagbox}{8.1 Work Done by Gravity on an Incline}

\[
  W_{\rm grav} = mg\,d\sin\alpha = mg\,\Delta y
\]
where $\Delta y = d\sin\alpha$ is the vertical drop (positive downward).

\textbf{Key insight:} Work by gravity depends only on vertical displacement, not path.  Gravity is a \emph{conservative} force.

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|l|c|}
\hline
\textbf{Motion} & \textbf{Work by gravity} \\
\hline
Down the incline & $+mgd\sin\alpha > 0$ \\
Up the incline   & $-mgd\sin\alpha < 0$ \\
Horizontal       & $0$ \\
\hline
\end{tabular}
\end{center}

\end{diagbox}

\section{Kinetic Energy (Section 10.3)}

\begin{diagbox}{9. Kinetic Energy}

\textbf{Kinetic energy:}
\[
  K = \tfrac{1}{2}mv^2
\]
\textbf{Units:} joules (J).  Always $K \geq 0$.

\end{diagbox}

\subsection{Understanding Kinetic Energy}

\begin{diagbox}{9.1 Understanding Kinetic Energy}

\begin{itemize}[leftmargin=1.5em]
  \item KE is a \emph{scalar}, always non-negative.
  \item KE depends on the reference frame.
  \item \textbf{Doubling the mass} doubles the KE.
  \item \textbf{Doubling the speed} \emph{quadruples} the KE.
\end{itemize}

\textbf{Example:} 2-kg ball at 3\,m/s vs.\ 6\,m/s:
\[
  K_1 = \tfrac{1}{2}(2)(9) = 9\text{ J}; \quad K_2 = \tfrac{1}{2}(2)(36) = 36\text{ J} = 4K_1
\]

\end{diagbox}

\subsection{The \texorpdfstring{$v^2$}{v\texttwosuperior} Dependence: Why It Matters}

\begin{diagbox}{9.2 The $v^2$ Dependence: Why It Matters}

\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Braking distance} $\propto v^2$: doubling speed quadruples stopping distance.
  \item \textbf{Wind power} scales as $v^3$.
  \item \textbf{Crash energy} at 60\,mph is four times that at 30\,mph.
\end{itemize}

\begin{warnbox}
\textbf{Common mistake:} Writing $K = \tfrac{1}{2}mv$ (forgetting to square $v$).  Check units: $\text{kg}\cdot(\text{m/s})^2 = \text{J}$.
\end{warnbox}

\end{diagbox}

\subsection{The Work--Energy Theorem}

\begin{diagbox}{9.3 The Work--Energy Theorem}

\[
  W_{\rm net} = \Delta K = \tfrac{1}{2}mv_f^2 - \tfrac{1}{2}mv_i^2
\]

\textbf{Derivation (1D, constant force):}  From kinematics $v_f^2 = v_i^2 + 2a\Delta x$, multiply by $\tfrac{1}{2}m$:
\[
  \tfrac{1}{2}mv_f^2 - \tfrac{1}{2}mv_i^2 = ma\,\Delta x = F_{\rm net}\,\Delta x = W_{\rm net}
\]

\textbf{Note:} Only the \emph{net} work from all forces equals $\Delta K$.

\end{diagbox}

\subsection{Kinetic Energy and the Work--Energy Theorem on Inclines}

\begin{diagbox}{9.4 Kinetic Energy and the Work--Energy Theorem on Inclines}

\textbf{Setup:} Mass $m$ slides distance $d$ down frictionless incline at angle $\alpha$.

Work by gravity: $W_g = mgd\sin\alpha$; normal force does zero work.

\[
  W_{\rm net} = mgd\sin\alpha = \tfrac{1}{2}mv_f^2 - \tfrac{1}{2}mv_i^2
\]

Starting from rest ($v_i = 0$): $v_f = \sqrt{2gd\sin\alpha}$.

\textbf{Example:} 5-kg block from rest down 4.0-m, $30^\circ$ frictionless ramp:
\[
  v_f = \sqrt{2(9.8)(4.0)(0.5)} = \sqrt{39.2} \approx 6.3\text{ m/s}
\]

\end{diagbox}

\subsection{Multiple Objects and Kinetic Energy}

\begin{diagbox}{9.5 Multiple Objects and Kinetic Energy}

Total KE of a system:
\[
  K_{\rm total} = \sum_i \tfrac{1}{2}m_i v_i^2
\]

For a connected system: $W_{\rm net,\,ext} = \Delta K_{\rm total}$ (internal forces contribute zero net work).

\textbf{Example (Atwood machine):} $m_1 = 3$\,kg, $m_2 = 5$\,kg, released from rest; each moves 1.0\,m:
\[
  W_{\rm net} = (m_2 - m_1)g\,d = (2)(9.8)(1.0) = 19.6\text{ J}
\]
\[
  \tfrac{1}{2}(8)v^2 = 19.6\text{ J} \;\Rightarrow\; v = \sqrt{4.9} \approx 2.2\text{ m/s}
\]

\end{diagbox}

\subsection{Common Mistakes with Kinetic Energy}

\begin{diagbox}{9.6 Common Mistakes with Kinetic Energy}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Forgetting to square speed:} $K \neq \tfrac{1}{2}mv$; use $K = \tfrac{1}{2}mv^2$.
  \item \textbf{Treating KE as a vector:} KE is a scalar, always $\geq 0$.
  \item \textbf{Applying theorem to one force:} Must sum work from \emph{all} forces.
  \item \textbf{Ignoring friction:} Friction does negative work and reduces $\Delta K$.
  \item \textbf{Confusing KE and momentum:} $K = \tfrac{1}{2}mv^2 \neq p = mv$.
  \item \textbf{Wrong units:} Speed must be in m/s, mass in kg, for KE in joules.
\end{enumerate}

\end{diagbox}

\subsection{Kinetic Energy in Context: Real-World Numbers}

\begin{diagbox}{9.7 Kinetic Energy in Context: Real-World Numbers}

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|l|c|c|c|}
\hline
\textbf{Object} & \textbf{Mass (kg)} & \textbf{Speed (m/s)} & \textbf{KE (J)} \\
\hline
Baseball (pitch) & 0.145 & 40 & 116 \\
Person walking & 70 & 1.4 & 69 \\
Car at 60\,mph & 1200 & 26.8 & $4.3\times10^5$ \\
Bullet & 0.010 & 900 & 4050 \\
Jumbo jet & $3.6\times10^5$ & 250 & $1.1\times10^{10}$ \\
\hline
\end{tabular}
\end{center}

The car's KE at highway speed is $\sim$100$\times$ a baseball pitch, despite being much slower, because mass matters too.

\end{diagbox}

\subsection{Practice Problems: Kinetic Energy (Section 10.3)}

\begin{planbox}

\begin{enumerate}[leftmargin=1.5em, label=\textbf{P\arabic*.}]

  \item A 2.0-kg ball is thrown horizontally at 15\,m/s.  What is its kinetic energy?

  \item A 1500-kg car has $K = 2.7\times10^5$\,J.  What is its speed?

  \item A 0.5-kg hockey puck slides at 8.0\,m/s and is stopped by a 2.0-N friction force.  Find the stopping distance using the Work--Energy Theorem.

  \item Two objects have the same kinetic energy.  Object A has mass $m$ and speed $v$.  Object B has mass $4m$.  What is the speed of object B?

  \item A 10-kg block starts from rest and slides 5.0\,m down a frictionless $25^\circ$ incline.  Find its final speed.

  \item A net force of 40\,N acts on an 8.0-kg object over 3.0\,m.  If the initial speed is 2.0\,m/s, find the final speed.

  \item A bullet of mass 0.012\,kg embeds in a 1.0-kg block (initially at rest).  The block then slides 0.80\,m with $\mu_k = 0.25$.  Find the speed of the bullet--block system just after impact.

  \item Compare the KE of (a) a 1500-kg car at 20\,m/s and (b) a 3000-kg truck at 10\,m/s.  Which is larger?

\end{enumerate}

\end{planbox}

\subsection{Solutions: Kinetic Energy (Section 10.3)}

\begin{keybox}

\begin{enumerate}[leftmargin=1.5em, label=\textbf{P\arabic*.}]

  \item \textbf{$K = 225$\,J.} \quad $K = \tfrac{1}{2}(2.0)(15)^2 = 225$\,J.

  \item \textbf{$v \approx 19.0$\,m/s.} \quad $v = \sqrt{2K/m} = \sqrt{2(2.7\times10^5)/1500} = \sqrt{360} \approx 19.0$\,m/s.

  \item \textbf{$d = 8.0$\,m.} \quad $K_i = f_k\,d \Rightarrow \tfrac{1}{2}(0.5)(8)^2 = (2.0)d \Rightarrow d = 16/2 = 8.0$\,m.

  \item \textbf{$v_B = v/2$.} \quad $\tfrac{1}{2}mv^2 = \tfrac{1}{2}(4m)v_B^2 \Rightarrow v_B^2 = v^2/4 \Rightarrow v_B = v/2$.

  \item \textbf{$v_f \approx 6.4$\,m/s.} \quad $W_g = (10)(9.8)(5.0)\sin25^\circ \approx 207$\,J; $v_f = \sqrt{2(207)/10} \approx 6.4$\,m/s.

  \item \textbf{$v_f \approx 5.8$\,m/s.} \quad $\tfrac{1}{2}mv_f^2 = \tfrac{1}{2}(8)(4) + (40)(3) = 16+120 = 136$\,J; $v_f = \sqrt{2(136)/8} = \sqrt{34} \approx 5.8$\,m/s.

  \item \textbf{$v \approx 1.97$\,m/s.} \quad Friction work: $W_f = -(0.25)(1.0)(9.8)(0.80) = -1.96$\,J. Setting $\tfrac{1}{2}(1.012)v^2 = 1.96$\,J: $v \approx 1.97$\,m/s.

  \item \textbf{Car KE is larger.} \quad Car: $K = \tfrac{1}{2}(1500)(400) = 3.0\times10^5$\,J. Truck: $K = \tfrac{1}{2}(3000)(100) = 1.5\times10^5$\,J.  The car's KE is twice the truck's.

\end{enumerate}

\end{keybox}

\section{Potential Energy (Section 10.4)}

\begin{diagbox}{10. Potential Energy}

\textbf{Potential energy} (PE or $U$) is energy stored in a system due to position or configuration within a \emph{conservative} force field.

\end{diagbox}

\subsection{Gravitational Potential Energy}

\begin{diagbox}{10.1 Gravitational Potential Energy}

Near Earth's surface:
\[
  U_g = mgy
\]
\[
  \Delta U_g = mg\,\Delta y = mg(y_f - y_i)
\]
\begin{itemize}[leftmargin=1.5em]
  \item $\Delta U_g > 0$: object moved upward.
  \item $\Delta U_g < 0$: object moved downward.
\end{itemize}

\end{diagbox}

\subsubsection{Choosing the Reference Level}

\begin{diagbox}{10.1.1 Choosing the Reference Level}

The reference level ($y = 0$) is arbitrary:
\begin{itemize}[leftmargin=1.5em]
  \item Common choices: ground level, lowest point of motion, initial position.
  \item \textbf{Only $\Delta U_g$ matters}; the reference level cancels in the final answer.
\end{itemize}

\textbf{Example:} 3-kg ball 2.0\,m above floor. Reference at floor: $U_g = (3)(9.8)(2.0) = 58.8$\,J. Reference at ball: $U_g = 0$.  Both give $\Delta U_g = -58.8$\,J when ball reaches floor.

\end{diagbox}

\subsubsection{Gravitational PE on Inclines}

\begin{diagbox}{10.1.2 Gravitational PE on Inclines}

Object at distance $s$ along a ramp of angle $\alpha$:
\[
  U_g = mgs\sin\alpha = mgh \quad\text{where } h = s\sin\alpha
\]
PE depends only on vertical height $h$, not on path length along ramp.

\textbf{Example:} 5-kg crate 3.0\,m up a $20^\circ$ ramp:
\[
  h = 3.0\sin20^\circ = 1.03\text{ m}, \quad U_g = (5)(9.8)(1.03) \approx 50.3\text{ J}
\]

\end{diagbox}

\subsection{Elastic (Spring) Potential Energy}

\begin{diagbox}{10.2 Elastic (Spring) Potential Energy}

\[
  U_{\rm sp} = \tfrac{1}{2}kx^2
\]
\begin{itemize}[leftmargin=1.5em]
  \item Always $U_{\rm sp} \geq 0$ (compressed or stretched).
  \item Maximum at maximum compression/extension; zero at natural length.
\end{itemize}

\textbf{Example:} $k = 500$\,N/m, compressed 0.10\,m: $U_{\rm sp} = \tfrac{1}{2}(500)(0.01) = 2.5$\,J.

\end{diagbox}

\subsection{Relationship Between Work and Potential Energy}

\begin{diagbox}{10.3 Relationship Between Work and Potential Energy}

For any \textbf{conservative force}:
\[
  W_c = -\Delta U
\]
Work done by a conservative force equals the \emph{decrease} in PE.

\textbf{Gravity:} $W_{\rm grav} = -\Delta U_g = mg(y_i - y_f)$

\textbf{Spring:} $W_{\rm sp} = -\Delta U_{\rm sp}$

\begin{warnbox}
Only conservative forces have associated PE.  Friction is non-conservative --- it permanently converts mechanical energy to thermal energy.
\end{warnbox}

\end{diagbox}

\section{Thermal Energy (Section 10.5)}

\begin{diagbox}{11. Thermal Energy}

\textbf{Thermal energy} $E_{\rm th}$: microscopic KE and PE of atoms/molecules (manifests as temperature).

\textbf{From friction:}
\[
  \Delta E_{\rm th} = f_k\,d \quad\text{where } f_k = \mu_k\,n
\]

This energy is permanently converted from mechanical energy and is not recoverable.

\end{diagbox}

\subsection{Why Thermal Energy Is Different}

\begin{diagbox}{11.1 Why Thermal Energy Is Different}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Irreversible:} Friction converts ordered mechanical energy to disordered thermal energy.
  \item \textbf{Not a PE:} $E_{\rm th}$ cannot be stored and recovered like $U_g$ or $U_{\rm sp}$.
  \item \textbf{Always positive:} $\Delta E_{\rm th} = f_k d \geq 0$; friction never adds mechanical energy.
  \item \textbf{Second Law:} Irreversibility reflects increasing entropy.
\end{enumerate}

\begin{warnbox}
With friction, \emph{mechanical energy is not conserved}:
\[
  K_f + U_f = K_i + U_i - \Delta E_{\rm th}
\]
Total energy (mechanical + thermal) is still conserved.
\end{warnbox}

\end{diagbox}

\subsection{Thermal Energy and Stopping Distance}

\begin{diagbox}{11.2 Thermal Energy and Stopping Distance}

A car of mass $m$ braking from $v_0$ to rest:
\[
  \Delta E_{\rm th} = K_i = \tfrac{1}{2}mv_0^2
\]
Also $\Delta E_{\rm th} = \mu_k\,mg\,d$, so:
\[
  d = \frac{v_0^2}{2\mu_k g}
\]

\textbf{Stopping distance scales as $v_0^2$:} doubling speed quadruples stopping distance.

\textbf{Example:} $\mu_k = 0.70$, $v_0 = 25$\,m/s:
\[
  d = \frac{625}{2(0.70)(9.8)} \approx 45.6\text{ m}
\]

\end{diagbox}

\section{Conservation of Energy (Section 10.6)}

\begin{diagbox}{12. Conservation of Energy}

\textbf{Law of Conservation of Energy:} The total energy of an isolated system is constant.

\[
  \Delta K + \Delta U_g + \Delta U_{\rm sp} + \Delta E_{\rm th} = W_{\rm ext}
\]

\end{diagbox}

\subsection{The Master Energy Equation}

\begin{diagbox}{12.1 The Master Energy Equation}

\[
  \boxed{K_f + U_f + \Delta E_{\rm th} = K_i + U_i + W_{\rm ext}}
\]

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|l|l|}
\hline
\textbf{Term} & \textbf{Meaning} \\
\hline
$K_f$ & Final kinetic energy \\
$U_f$ & Final potential energy ($U_g + U_{\rm sp}$) \\
$\Delta E_{\rm th}$ & Thermal energy from friction ($\geq 0$) \\
$K_i$ & Initial kinetic energy \\
$U_i$ & Initial potential energy \\
$W_{\rm ext}$ & Work by external agent \\
\hline
\end{tabular}
\end{center}

\textbf{Special cases:}
\begin{itemize}[leftmargin=1.5em]
  \item No friction, no external work: $K_f + U_f = K_i + U_i$.
  \item With friction, no external work: $K_f + U_f + \Delta E_{\rm th} = K_i + U_i$.
  \item No friction, with external work: $K_f + U_f = K_i + U_i + W_{\rm ext}$.
\end{itemize}

\end{diagbox}

\subsection{Problem-Solving Strategy}

\begin{diagbox}{12.2 Problem-Solving Strategy}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Define the system.} Include all objects whose energy changes.
  \item \textbf{Identify initial and final states.} Draw diagrams for each.
  \item \textbf{Choose a reference level} for gravitational PE.
  \item \textbf{Identify energy forms} present at each state: $K$, $U_g$, $U_{\rm sp}$.
  \item \textbf{Check for friction.} Compute $\Delta E_{\rm th} = f_k d$ if present.
  \item \textbf{Check for external work.}  Add $W_{\rm ext}$ if an external agent acts.
  \item \textbf{Apply the master energy equation} and solve.
  \item \textbf{Check units and reasonableness.}
\end{enumerate}

\end{diagbox}

\subsection{Worked Examples: No Friction}

\begin{diagbox}{12.3 Worked Examples: No Friction}

\textbf{Example A --- Ball dropped from rest:}

A 2-kg ball dropped from $h = 5$\,m.  Find speed just before impact.
\[
  mgh = \tfrac{1}{2}mv_f^2 \;\Rightarrow\; v_f = \sqrt{2gh} = \sqrt{2(9.8)(5)} \approx 9.9\text{ m/s}
\]

\vspace{0.4cm}
\textbf{Example B --- Spring launching a block:}

Spring ($k = 800$\,N/m) compressed 0.15\,m releases a 0.5-kg block.  Find the block's speed.
\[
  \tfrac{1}{2}kx^2 = \tfrac{1}{2}mv^2 \;\Rightarrow\; v = x\sqrt{k/m} = 0.15\sqrt{1600} = 6.0\text{ m/s}
\]

\vspace{0.4cm}
\textbf{Example C --- Ball on frictionless ramp:}

3-kg ball from rest at top of 4.0-m-high frictionless ramp.
\[
  v = \sqrt{2gh} = \sqrt{2(9.8)(4.0)} \approx 8.9\text{ m/s}
\]

\end{diagbox}

\subsection{Worked Examples: With Friction}

\begin{diagbox}{12.4 Worked Examples: With Friction}

\textbf{Example D --- Block on rough ramp:}

4-kg block slides 6.0\,m down a $30^\circ$ ramp ($\mu_k = 0.20$) from rest.

$h = 6\sin30^\circ = 3.0$\,m; $n = mg\cos30^\circ = 33.9$\,N; $f_k = (0.20)(33.9) = 6.78$\,N.

$\Delta E_{\rm th} = (6.78)(6.0) = 40.7$\,J.

\[
  \tfrac{1}{2}(4)v^2 = mgh - \Delta E_{\rm th} = 117.6 - 40.7 = 76.9\text{ J}
  \;\Rightarrow\; v = \sqrt{38.45} \approx 6.2\text{ m/s}
\]

\vspace{0.4cm}
\textbf{Example E --- Object stopped by friction:}

1.5-kg box at 5.0\,m/s; $\mu_k = 0.35$.
\[
  \tfrac{1}{2}(1.5)(25) = (0.35)(1.5)(9.8)d \;\Rightarrow\; d \approx 3.6\text{ m}
\]

\end{diagbox}

\subsection{Worked Examples: With External Work}

\begin{diagbox}{12.5 Worked Examples: With External Work}

\textbf{Example F --- Motor lifts a crate:}

Motor does 2400\,J lifting a 50-kg crate 4.0\,m from rest to rest.
\[
  W_{\rm ext} = \Delta U_g + \Delta E_{\rm th} \;\Rightarrow\; 2400 = (50)(9.8)(4.0) + \Delta E_{\rm th}
  \;\Rightarrow\; \Delta E_{\rm th} = 440\text{ J}
\]
Efficiency: $1960/2400 \approx 81.7\%$.

\vspace{0.4cm}
\textbf{Example G --- Cyclist over two hills:}

80-kg cyclist (bike included) coasts over a 5.0-m descent then up a 3.0-m hill.  Friction: $\Delta E_{\rm th} = 120$\,J.  Initial speed: 8.0\,m/s.
\[
  K_f = K_i + Mg\Delta h - \Delta E_{\rm th}
      = \tfrac{1}{2}(80)(64) + (80)(9.8)(2) - 120
      = 2560 + 1568 - 120 = 4008\text{ J}
\]
\[
  v_f = \sqrt{2(4008)/80} \approx 10.0\text{ m/s}
\]

\end{diagbox}

\subsection{Energy Bar Charts}

\begin{diagbox}{12.6 Energy Bar Charts}

\textbf{Energy bar charts} (LOL diagrams) visualize the master energy equation.  Each bar represents an energy form; heights must balance.

\textbf{Drawing procedure:}
\begin{enumerate}[leftmargin=1.5em]
  \item Draw initial and final bar sets separated by a process region.
  \item Label bars: $K$, $U_g$, $U_{\rm sp}$, $E_{\rm th}$.
  \item Include a $W_{\rm ext}$ bar in the process region if needed.
  \item Total initial energy $+$ $W_{\rm ext}$ $=$ total final energy.
\end{enumerate}

\textbf{Example (frictionless ramp):} Initial: tall $U_g$, zero $K$. Final: zero $U_g$, tall $K$ (equal height). \\
\textbf{Example (with friction):} Initial: tall $U_g$. Final: shorter $K$ + positive $E_{\rm th}$. $U_{g,i} = K_f + \Delta E_{\rm th}$.

\end{diagbox}

\subsection{When to Use Energy Methods vs.\ Newton's Laws}

\begin{diagbox}{12.7 When to Use Energy Methods vs.\ Newton's Laws}

\begin{center}
\renewcommand{\arraystretch}{1.8}
\begin{tabular}{|p{4.8cm}|p{4.5cm}|p{4.5cm}|}
\hline
\textbf{Situation} & \textbf{Energy Methods} & \textbf{Newton's Laws} \\
\hline
Known positions; want speed & \textbf{Preferred} -- one equation & Requires kinematics; harder \\
\hline
Non-constant force (spring, curved path) & \textbf{Preferred} -- PE handles it & Requires calculus \\
\hline

\hline

\hline
\end{tabular}
\end{center}

\begin{warnbox}
\textbf{Rule of thumb:} Use energy methods when the problem involves \emph{positions and speeds}.  Use Newton's Laws when it involves \emph{forces, accelerations, or time}.
\end{warnbox}

\end{enumerate}
\end{keybox}

\begin{planbox}

\subsection{Topic 7 --- Work}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{7\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        A 15\,kg box is dragged 6.0\,m along a horizontal floor by a rope that makes
        $28^\circ$ above the horizontal (tension $T=70$\,N; $\mu_k=0.20$).
        Find (a) the work done by $T$, (b) the work done by kinetic friction, and
        (c) the net work done on the box.

  \item \textbf{[5 pts --- Difficult]}

  \item (a) Constant speed on frictionless incline: $\sum F_\parallel=0$.

        \]

\begin{planbox}

  \item \textbf{[5 pts --- Difficult]}

\end{keybox}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{9\Alph*.}]

        A spring ($k=250$\,N/m) compressed 0.16\,m launches a 0.30\,kg ball vertically

        Find the maximum height above the launch point.  (Ignore air resistance.)

  \item \textbf{[5 pts --- Difficult]}
        A 2.0\,kg block on a frictionless horizontal surface is attached to a horizontal
        spring ($k=600$\,N/m) at its natural length and given a speed of 4.0\,m/s at $x=0$.
        \begin{enumerate}[label=(\alph*)]
          \item Find the maximum spring compression $x_{\rm max}$.
          \item Find the block's speed when the spring is compressed by 0.12\,m.
          \item What fraction of the initial KE is stored as elastic PE at $x=0.12$\,m?
        \end{enumerate}

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{9\Alph*.}]

  \item $\tfrac{1}{2}kx^2=mgh$
        \[
          h=\frac{kx^2}{2mg}=\frac{(250)(0.16)^2}{2(0.30)(9.8)}
           =\frac{6.40}{5.88}\approx\mathbf{1.09\text{ m}}
        \]

  \item (a) All KE converts to elastic PE at maximum compression:
        \[
          \tfrac{1}{2}mv^2=\tfrac{1}{2}kx_{\rm max}^2
          \;\Rightarrow\;x_{\rm max}=v\sqrt{\frac{m}{k}}
          =4.0\sqrt{\frac{2.0}{600}}\approx\mathbf{0.231\text{ m}}
        \]

        (b) Energy conservation at $x=0.12$\,m:
        \[
          \tfrac{1}{2}mv_1^2=\tfrac{1}{2}mv_0^2-\tfrac{1}{2}kx^2
          =\tfrac{1}{2}(2.0)(16)-\tfrac{1}{2}(600)(0.12)^2
          =16-4.32=11.68\text{ J}
        \]
        \[
          v_1=\sqrt{\frac{2(11.68)}{2.0}}=\sqrt{11.68}\approx\mathbf{3.42\text{ m/s}}
        \]

        (c) $\dfrac{U_{\rm sp}}{K_i}=\dfrac{4.32}{16}=0.270$ $\Rightarrow$
            $\mathbf{27\%}$ of the initial KE is stored as elastic PE.

\end{enumerate}
\end{keybox}

\subsection{Topic 10 --- Conservation of Energy}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{10\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        A 5.0\,kg ball is released from rest at $h_i=8.0$\,m on a frictionless curved
        ramp.  Using conservation of energy, find its speed at $h_f=2.5$\,m.

  \item \textbf{[5 pts --- Difficult]}
        A 2.0\,kg block starts from rest and slides 5.0\,m down a $40^\circ$ incline
        ($\mu_k=0.15$).  It then immediately contacts a spring ($k=1200$\,N/m) on a
        horizontal surface ($\mu_k=0.20$).  Find the maximum spring compression $x_c$.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{10\Alph*.}]

  \item $K_f+U_f=K_i+U_i\;\Rightarrow\;\tfrac{1}{2}mv^2=mg(h_i-h_f)$
        \[
          v=\sqrt{2g(h_i-h_f)}=\sqrt{2(9.8)(8.0-2.5)}=\sqrt{107.8}
           \approx\mathbf{10.4\text{ m/s}}
        \]

  \item \textbf{Step 1 --- KE at bottom of incline.}\\
        $\Delta h=5.0\sin40^\circ=3.21$\,m;\quad
        $N_1=mg\cos40^\circ=(2.0)(9.8)(0.766)=15.0$\,N.
        \[
          \Delta E_{{\rm th},1}=\mu_k N_1(5.0)=(0.15)(15.0)(5.0)=11.3\text{ J}
        \]
        \[
          K_1=mg\,\Delta h-\Delta E_{{\rm th},1}=(2.0)(9.8)(3.21)-11.3
              =62.9-11.3=51.6\text{ J}
        \]

        \textbf{Step 2 --- spring compression (horizontal friction + spring).}\\
        At maximum compression $v=0$;\quad
        $\Delta E_{{\rm th},2}=\mu_k mg\,x_c=(0.20)(2.0)(9.8)x_c=3.92\,x_c$.
        \[
          K_1=\tfrac{1}{2}kx_c^2+\Delta E_{{\rm th},2}
          \;\Rightarrow\;51.6=600\,x_c^2+3.92\,x_c
        \]
        \[
          600\,x_c^2+3.92\,x_c-51.6=0
        \]
        Quadratic formula (positive root):
        \[
          x_c=\frac{-3.92+\sqrt{3.92^2+4(600)(51.6)}}{1200}
             =\frac{-3.92+\sqrt{123\,855}}{1200}
             =\frac{-3.92+351.9}{1200}
             \approx\mathbf{0.290\text{ m}}
        \]

\end{enumerate}
\end{keybox}
