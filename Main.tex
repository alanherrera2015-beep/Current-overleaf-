\section*{Problem 1: Shock Absorber Springs (Advanced)}
%==============================================

\begin{problembox}[Problem 1 -- Chapters 8 \& 10: Asymmetric Spring Suspension]
A 1600 kg SUV has its center of mass $a = 0.80$ m behind the front axle and $b = 1.20$ m
in front of the rear axle (wheelbase $L = 2.00$ m). The center of mass is also
$d = 0.10$ m to the \emph{right} of the vehicle centerline. The front wheel track
(distance between the two front wheels) is $W = 1.60$ m. The front-left spring has
constant $k_L = 24{,}000$ N/m; the front-right spring has $k_R = 30{,}000$ N/m.

\begin{enumerate}[label=(\alph*)]
  \item Find the total vertical force $R_f$ on the front axle.
        (\textit{Hint: take torques about the rear axle.})
  \item Taking torques about the \emph{left} front wheel (left-right axis), find
        the individual spring forces $F_R$ (right) and $F_L$ (left).
  \item Find the compression of each spring ($x_L$ and $x_R$). Are they equal?
  \item Find the total elastic potential energy stored in the two front springs.
  \item When the SUV drives over a speed bump, the dynamic load on the front axle
        momentarily \emph{doubles}. Find the new right-spring compression $x_R'$.
        What percentage increase is this compared to the static value?
\end{enumerate}
\end{problembox}

\section*{Problem 2: Force on Elbow Joint (Advanced)}

\begin{problembox}[Problem 2 -- Chapter 8: Tilted Forearm Equilibrium]
A nurse holds a patient chart of mass $m_c = 2.5$ kg at the end of their forearm
($m_{\text{arm}} = 1.5$ kg, length $L = 0.35$ m). The forearm is tilted at angle

$\beta = 80^\circ$ from the forearm axis (nearly perpendicular to the forearm).
The forearm's center of mass is at its midpoint.

\begin{enumerate}[label=(\alph*)]
  \item Draw a careful free body diagram. Label the elbow joint, the forearm angle,
        the muscle attachment point, and all forces.
  \item Taking torques about the elbow joint, write and solve the torque equation
        to find the bicep muscle force $F_B$.
        (\textit{Hint: for a vertical weight $W$ at distance $r$ along a forearm tilted
        $\phi$ below horizontal, the torque is $\tau = Wr\cos\phi$.})
  \item Find the $x$- and $y$-components of the reaction force $\vec{R}$ at the
        elbow joint, and then find its magnitude.
  \item How would the bicep force change if the forearm were horizontal instead
        ($\phi = 0^\circ$)? Show the calculation and explain why the force is larger
        or smaller.
\end{enumerate}
\end{problembox}

\section*{Solution 1: Shock Absorber Springs (Advanced)}
%==============================================

\begin{solutionbox}[Solution 1(a) -- Total front-axle load]

\textbf{Plain-English idea:} A car's weight is not split 50/50 front and back -- it
depends on where the engine (center of mass) sits. We treat the car like a seesaw
balanced between the two axles, and use torques to figure out what each axle holds.

\smallskip
\textbf{Step 1 -- Find the car's total weight.}
\begin{equation}
W = mg = (1600)(9.8) = 15{,}680\;\text{N}
\end{equation}

\textbf{Step 2 -- Take torques about the rear axle (eliminates the unknown rear force).}

The center of mass is $b = 1.20$ m in front of the rear axle. The front axle is
$L = 2.00$ m in front of the rear axle.
\begin{equation}
R_f \times L = W \times b \quad\Longrightarrow\quad R_f = W \times \frac{b}{L}
= 15{,}680 \times \frac{1.20}{2.00} = \boxed{9{,}408\;\text{N}}
\end{equation}
\textit{The front axle carries 60\% of the weight because the CM is closer to the
front (0.80 m from front vs.\ 1.20 m from rear).}

\end{solutionbox}

\begin{solutionbox}[Solution 1(b) -- Individual spring forces]

\textbf{Plain-English idea:} Now we do the same trick left-to-right. The CM is
slightly to the right of center, so the right spring holds more weight than the left.
We take torques about the left front wheel to find the right-spring force.

\smallskip
\textbf{Step 1 -- Find the distances.}

The half-track is $W/2 = 0.80$ m from center to each wheel. The CM is $d = 0.10$ m
to the right of center, so:
\begin{itemize}
  \item Distance from \textbf{left} wheel to CM $= 0.80 + 0.10 = 0.90$ m (to the right)
  \item Distance from \textbf{right} wheel to CM $= 0.80 - 0.10 = 0.70$ m (to the left)
\end{itemize}

\textbf{Step 2 -- Torques about the left front wheel (CCW positive):}
\begin{align}
F_R \times 1.60 &= R_f \times 0.90 \\
F_R &= \frac{9{,}408 \times 0.90}{1.60} = \boxed{5{,}292\;\text{N (right spring)}}
\end{align}

\textbf{Step 3 -- Force balance for $F_L$:}
\begin{equation}
F_L = R_f - F_R = 9{,}408 - 5{,}292 = \boxed{4{,}116\;\text{N (left spring)}}
\end{equation}
\textit{The right spring holds about 28\% more force than the left -- because the
heavy engine is offset to the right (e.g., passenger side).}

\end{solutionbox}

\begin{solutionbox}[Solution 1(c) -- Spring compressions]

\textbf{Plain-English idea:} Hooke's Law ($x = F/k$) says: stiffer spring, less
compression. The right spring is stiffer ($k_R > k_L$) but also carries more force.
Let's see which effect wins!

\begin{align}
x_L &= \frac{F_L}{k_L} = \frac{4{,}116}{24{,}000} = \boxed{0.172\;\text{m} = 17.2\;\text{cm}} \\[4pt]
x_R &= \frac{F_R}{k_R} = \frac{5{,}292}{30{,}000} = \boxed{0.176\;\text{m} = 17.6\;\text{cm}}
\end{align}
\textit{They are almost equal (difference of 4 mm) because the right spring is both
stiffer and more loaded -- the two effects nearly cancel. This is no accident: spring
rates are often matched to the load so all four corners sit level.}

\end{solutionbox}

\begin{solutionbox}[Solution 1(d) -- Elastic potential energy]

\textbf{Plain-English idea:} Each compressed spring stores energy $U = \frac{1}{2}kx^2$.
Add both springs together.

\begin{align}
U_L &= \frac{1}{2}(24{,}000)(0.172)^2 = 12{,}000 \times 0.029584 = 355.0\;\text{J} \\
U_R &= \frac{1}{2}(30{,}000)(0.176)^2 = 15{,}000 \times 0.030976 = 464.6\;\text{J}
\end{align}
\begin{equation}
U_{\text{total}} = 355.0 + 464.6 = \boxed{819.6\;\text{J} \approx 820\;\text{J}}
\end{equation}

\end{solutionbox}

\begin{solutionbox}[Solution 1(e) -- Dynamic loading (doubled load)]

\textbf{Plain-English idea:} When the car hits a bump, the road pushes up extra hard --
engineers call this the ``dynamic load factor.'' If the force doubles, Hooke's Law
$x = F/k$ says the compression doubles too (spring force is proportional to
compression).

\smallskip
Under doubled front load: $F_R' = 2 \times 5{,}292 = 10{,}584$ N
\begin{equation}
x_R' = \frac{F_R'}{k_R} = \frac{10{,}584}{30{,}000} = \boxed{0.353\;\text{m} = 35.3\;\text{cm}}
\end{equation}
\textbf{Percentage increase:}
\begin{equation}
\frac{x_R' - x_R}{x_R} \times 100\% = \frac{0.353 - 0.176}{0.176} \times 100\% = \boxed{100\%}
\end{equation}
\textit{Exactly double! Because $F = kx$ is linear, doubling the force always doubles
the compression. This is why engineers design suspension travel (maximum compression)
to be at least twice the static compression.}

\end{solutionbox}

\section*{Solution 2: Force on Elbow Joint (Advanced)}
%==============================================

\begin{solutionbox}[Solution 2(a) -- Free Body Diagram]

\textbf{Plain-English idea:} We draw all forces on the forearm, which acts like a
lever (beam) balanced at the elbow. The tricky part here is that the forearm is
tilted -- this changes the perpendicular distances (moment arms) that weights create.

\smallskip
Forces on the forearm (tilted $30^\circ$ below horizontal):
\begin{itemize}
  \item \textbf{Bicep force} $F_B$ -- at $\beta = 80^\circ$ from the forearm axis,
        at $d = 0.05$ m from elbow. It pulls the forearm \emph{toward} the upper arm.
  \item \textbf{Forearm weight} $W_{\text{arm}} = 1.5 \times 9.8 = 14.7$ N -- straight
        down, at the midpoint ($L/2 = 0.175$ m from elbow along the forearm).
  \item \textbf{Chart weight} $W_c = 2.5 \times 9.8 = 24.5$ N -- straight down,
        at the far end ($L = 0.35$ m from elbow).
  \item \textbf{Joint reaction} $\vec{R}$ -- at the elbow (direction unknown, to be found).
\end{itemize}

\end{solutionbox}

\begin{solutionbox}[Solution 2(b) -- Torque equation and bicep force]

\textbf{Plain-English idea:} Taking torques about the elbow eliminates the unknown
joint force $\vec{R}$ (it has zero moment arm). For a vertical weight $W$ at distance
$r$ along a forearm that is tilted $\phi$ below horizontal, the moment arm (the
perpendicular distance from the elbow to the line of gravity) equals $r\cos\phi$.
This is because the moment arm shrinks as the arm tilts further down.

\smallskip
\textbf{Torques about the elbow (CCW = +):}

For the bicep, the component perpendicular to the forearm creates the torque:
$\tau_B = F_B \sin\beta \times d = F_B \sin 80^\circ \times 0.05$ (CCW, $+$)

For weights, the torque is $\tau = W \times r\cos\phi$ (CW, $-$):
\begin{align}
\tau_{\text{arm}} &= W_{\text{arm}} \times (L/2)\cos\phi = 14.7 \times 0.175 \times \cos 30^\circ
= 14.7 \times 0.175 \times 0.866 = 2.23\;\text{N$\cdot$m}\\
\tau_c &= W_c \times L\cos\phi = 24.5 \times 0.35 \times 0.866 = 7.43\;\text{N$\cdot$m}
\end{align}

\textbf{Set $\sum\tau = 0$:}
\begin{align}
F_B \sin 80^\circ \times 0.05 &= 2.23 + 7.43 = 9.66\;\text{N$\cdot$m} \\
F_B \times 0.9848 \times 0.05 &= 9.66 \\
0.04924\,F_B &= 9.66
\end{align}
\begin{equation}
F_B = \frac{9.66}{0.04924} = \boxed{196\;\text{N}}
\end{equation}
\textit{About 44 lbs of bicep force to hold a chart that weighs only 5.5 lbs -- the
short muscle attachment ($d = 5$ cm) creates a terrible mechanical disadvantage!}

\end{solutionbox}

\begin{solutionbox}[Solution 2(c) -- Elbow joint reaction force]

\textbf{Plain-English idea:} Use $\sum F = 0$ in each direction. First we need to know
the direction of the bicep force. The forearm points at $30^\circ$ below horizontal.
The bicep pulls at $80^\circ$ from the forearm axis (toward the upper arm, upward-inward),
so its direction is $80^\circ - 30^\circ = 50^\circ$ above horizontal.

\smallskip
\textbf{Bicep force components} (taking right = $+x$, up = $+y$):
\begin{align}
F_{B,x} &= -F_B \cos 50^\circ = -196 \times 0.6428 = -125.9\;\text{N (leftward)}\\
F_{B,y} &= +F_B \sin 50^\circ = +196 \times 0.7660 = +150.1\;\text{N (upward)}
\end{align}

\textbf{Force balance $\sum F_x = 0$:}
\begin{equation}
R_x + F_{B,x} = 0 \quad\Longrightarrow\quad R_x = +125.9\;\text{N (rightward)}
\end{equation}

\textbf{Force balance $\sum F_y = 0$:}
\begin{align}
R_y + F_{B,y} - W_{\text{arm}} - W_c &= 0 \\
R_y &= 14.7 + 24.5 - 150.1 = -110.9\;\text{N (downward)}
\end{align}

\textbf{Magnitude of joint reaction:}
\begin{equation}
|\vec{R}| = \sqrt{(125.9)^2 + (110.9)^2} = \sqrt{15{,}851 + 12{,}299}
= \sqrt{28{,}150} = \boxed{167.8\;\text{N}}
\end{equation}
\textit{The joint force ($\approx 38$ lbs) is larger than the chart weight
($\approx 5.5$ lbs) by a factor of 7 -- the skeleton takes a beating so the muscles
can have short, powerful attachments.}

\end{solutionbox}

\begin{solutionbox}[Solution 2(d) -- Comparison with horizontal forearm]

\textbf{Plain-English idea:} When the arm is tilted down, gravity's moment arm shrinks
(by $\cos\phi$). Less torque from the weights means the muscle needs to produce less
torque, so the muscle force is smaller.

\smallskip
For $\phi = 0^\circ$ (horizontal arm), $\cos 0^\circ = 1$:
\begin{align}
\tau_{\text{arm}}' &= 14.7 \times 0.175 \times 1 = 2.57\;\text{N$\cdot$m}\\
\tau_c' &= 24.5 \times 0.35 \times 1 = 8.58\;\text{N$\cdot$m}
\end{align}
\begin{equation}
F_B' = \frac{2.57 + 8.58}{0.04924} = \frac{11.15}{0.04924} = \boxed{226\;\text{N}}
\end{equation}
\textbf{Conclusion:} Horizontal arm requires $F_B' = 226$ N vs.\ $F_B = 196$ N when
tilted at $30^\circ$ below. The tilted position requires \emph{less} force because
the moment arms of the weights ($r\cos\phi$) are smaller. This is why it is easier to
hold something with your arm angled downward slightly.

\end{solutionbox}

\vspace{1.5em}

%==============================================
\section*{Solution 3: Superman Rescue (Advanced -- 2D)}
%==============================================

\begin{solutionbox}[Solution 3(a) -- Person's speed at $h = 80\,\mathrm{m}$]

\textbf{Plain-English idea:} The person fell from $H_p = 200$ m and is now at $h = 80$ m.
She has fallen $\Delta h = 120$ m. Gravity converted all that height into speed.
Use energy conservation (no friction, no air resistance).

\begin{equation}
\frac{1}{2}mv_p^2 = mg\,\Delta h \quad\Longrightarrow\quad
v_p = \sqrt{2g\,\Delta h} = \sqrt{2(9.8)(120)} = \sqrt{2352}
= \boxed{48.5\;\text{m/s (downward)}}
\end{equation}

\end{solutionbox}

\begin{solutionbox}[Solution 3(b) -- Superman's velocity at h = 80 m]

\textbf{Plain-English idea:} After Superman stops his engines, he is like a projectile:
his horizontal speed stays constant (no horizontal force), and gravity accelerates him
downward from rest in the vertical direction. We treat the two directions separately.

\smallskip
Superman starts at $H_S = 180$ m with $v_{Sx,i} = 15.0$ m/s (right) and $v_{Sy,i} = 0$.
He falls $\Delta h_S = 180 - 80 = 100$ m.

\textbf{Horizontal component (unchanged):}
\begin{equation}
v_{Sx} = u = \boxed{15.0\;\text{m/s (rightward)}}
\end{equation}

\textbf{Vertical component (gained from free-fall):}
\begin{equation}
v_{Sy} = \sqrt{2g\,\Delta h_S} = \sqrt{2(9.8)(100)} = \sqrt{1960}
= \boxed{44.3\;\text{m/s (downward)}}
\end{equation}

\textbf{Magnitude of Superman's speed:}
\begin{equation}
|v_S| = \sqrt{(15.0)^2 + (44.3)^2} = \sqrt{225 + 1962.49} = \sqrt{2187.49}
= \boxed{46.8\;\text{m/s}}
\end{equation}
\textit{Superman is moving mostly downward (44 m/s) with a moderate sideways component
(15 m/s). Like a skydiver who was drifting horizontally.}

\end{solutionbox}

\begin{solutionbox}[Solution 3(c) -- Combined velocity after the catch (2D momentum)]

\textbf{Plain-English idea:} The catch is a 2D collision -- they stick together.
Momentum is conserved \emph{independently} in each direction ($x$ and $y$). Think of
it like two trains on perpendicular tracks colliding at a junction: we balance east-west
and north-south momentum separately.

\smallskip
\textbf{Before the catch:}
\begin{center}
\begin{tabular}{lcc}
\toprule
 & $x$ (rightward) & $y$ (downward positive) \\
\midrule
Superman ($M = 90$ kg) & $+15.0$ m/s & $+44.3$ m/s \\
Person ($m = 70$ kg) & $0$ & $+48.5$ m/s \\
\bottomrule
\end{tabular}
\end{center}

\textbf{$x$-momentum conservation:}
\begin{equation}
(M + m)\,v_{fx} = M(15.0) + m(0) = 90 \times 15.0 = 1350\;\text{kg$\cdot$m/s}
\end{equation}
\begin{equation}
v_{fx} = \frac{1350}{90 + 70} = \frac{1350}{160} = \boxed{8.44\;\text{m/s (rightward)}}
\end{equation}

\textbf{$y$-momentum conservation (downward positive):}
\begin{align}
(M + m)\,v_{fy} &= M(44.3) + m(48.5) = 90(44.3) + 70(48.5) \\
&= 3987 + 3395 = 7382\;\text{kg$\cdot$m/s}
\end{align}
\begin{equation}
v_{fy} = \frac{7382}{160} = \boxed{46.1\;\text{m/s (downward)}}
\end{equation}

\textbf{Combined speed:}
\begin{equation}
v_f = \sqrt{(8.44)^2 + (46.1)^2} = \sqrt{71.2 + 2125.2} = \sqrt{2196.4}
= \boxed{46.9\;\text{m/s}}
\end{equation}
\textit{The combined speed is close to the person's original falling speed -- she is
much heavier than Superman's horizontal contribution.}

\end{solutionbox}

\begin{solutionbox}[Solution 3(d) -- Work by Superman's muscles]

\textbf{Plain-English idea:} They are now at $h = 80$ m, moving at 46.9 m/s with a
mix of downward and sideways speed. Superman must bring them to a complete stop at
ground level. We use the work-energy theorem: the muscles, gravity, and the final
zero-KE condition all connect.

\smallskip
\textbf{Step 1 -- KE just after the catch.}
\begin{equation}
KE_i = \frac{1}{2}(M+m)v_f^2 = \frac{1}{2}(160)(46.9)^2
= 80 \times 2199.6 = 175{,}968\;\text{J}
\end{equation}

\textbf{Step 2 -- Work by gravity over the 80 m descent to ground.}
\begin{equation}
W_g = (M+m)\,g\,h = (160)(9.8)(80) = 125{,}440\;\text{J} \quad\text{(positive)}
\end{equation}

\textbf{Step 3 -- Work-energy theorem ($KE_f = 0$).}
\begin{align}
W_{\text{net}} &= KE_f - KE_i = 0 - 175{,}968 = -175{,}968\;\text{J} \\
W_g + W_{\text{muscles}} &= -175{,}968 \\
125{,}440 + W_{\text{muscles}} &= -175{,}968
\end{align}
\begin{equation}
W_{\text{muscles}} = -175{,}968 - 125{,}440 = \boxed{-301{,}408\;\text{J} \approx -301\;\text{kJ}}
\end{equation}
\textit{Superman's muscles must absorb 301 kJ -- about the energy in a stick of
dynamite! The negative sign means his muscles do braking (negative) work, removing
energy from the system.}

\end{solutionbox}

\vspace{1.5em}

%==============================================
\section*{Solution A: Two-Spring Diving Board}
%==============================================

\begin{solutionbox}[Solution A -- Two-Spring System]

\textbf{Plain-English idea:} The board is perfectly still -- all forces and all torques
balance. Two equations (force balance and torque balance) give us our two unknowns.

\smallskip
\textbf{Forces:} $F_1 = 8000\,\delta_1$ (up) at $x=0$;
$F_2 = 12{,}000\,\delta_2$ (up) at $x=2.0$ m;
$W_B = 245$ N (down) at $x=1.5$ m; $W_D = 686$ N (down) at $x=3.0$ m.

\textbf{Force balance:}
\begin{equation}
8000\,\delta_1 + 12{,}000\,\delta_2 = 931\;\text{N}
\label{eq:forceA}
\end{equation}

\textbf{Torque balance about $x = 0$ (pivot at Spring 1 -- makes $F_1$ vanish):}
\begin{align}
12{,}000\,\delta_2 \times 2.0 &= 245 \times 1.5 + 686 \times 3.0 \\
24{,}000\,\delta_2 &= 367.5 + 2058 = 2425.5
\end{align}
\begin{equation}
\delta_2 = \frac{2425.5}{24{,}000} = \boxed{0.101\;\text{m} = 10.1\;\text{cm}}
\end{equation}

\textbf{Substitute into \eqref{eq:forceA}:}
\begin{equation}
\delta_1 = \frac{931 - 12{,}000(0.101)}{8000} = \frac{-281}{8000} = \boxed{-0.035\;\text{m}}
\end{equation}

\textit{Negative $\delta_1$: Spring 1 is in \textbf{tension} (stretched 3.5 cm).
The heavy diver on the right tips the board so much that the left end rises.}

\end{solutionbox}

\vspace{1.5em}

%==============================================
\section*{Solution B: Variable-Force Braking}
%==============================================

\begin{solutionbox}[Solution B -- Variable Force Stopping Distance]

\textbf{Plain-English idea:} The braking force grows with distance. To find work from
a changing force, integrate. Then use the work-energy theorem.

\textbf{Initial KE:}
\begin{equation}
KE_i = \frac{1}{2}(1500)(25)^2 = 468{,}750\;\text{J}
\end{equation}

\textbf{Work by braking force:}
\begin{equation}
W = \int_0^{d}(5000 + 200x)\,dx = 5000d + 100d^2
\end{equation}

\textbf{Set $W = KE_i$, solve quadratic:}
\begin{equation}
100d^2 + 5000d - 468{,}750 = 0 \;\Rightarrow\; d^2 + 50d - 4687.5 = 0
\end{equation}
\begin{equation}
d = \frac{-50 + \sqrt{2500 + 18750}}{2} = \frac{-50 + 145.8}{2} = \boxed{47.9\;\text{m}}
\end{equation}

\textbf{Constant-force comparison:}
$d_{\text{const}} = KE_i / F_0 = 468{,}750 / 5000 = \boxed{93.75\;\text{m}}$

\textit{Variable braking halves the stopping distance compared to constant braking.}

\end{solutionbox}

\section*{Common Exam Mistakes to Avoid}

\begin{warningbox}[Top Mistakes on PHYSICS 1101 Exams]
\begin{enumerate}
  \item \textbf{Torque sign errors:} Always define CCW as $+$ and CW as $-$, and stick
        to it throughout.
  \item \textbf{Moment arms when arm is tilted:} For a vertical weight $W$ at distance
        $r$ along a tilted beam, the moment arm is $r\cos\phi$ (not just $r$)!
  \item \textbf{Dynamic vs.\ static loads:} $F = kx$ is linear, so doubling load doubles
        compression; tripling load triples compression.
  \item \textbf{Unit conversions:} km/h to m/s: divide by 3.6. Always do this before
        plugging into any formula.
  \item \textbf{Integration limits:} Always integrate from 0 to $d$ (stopping distance)
        for braking problems.
  \item \textbf{Quadratic roots:} Physical answers (distance, speed) must be positive --
        discard the negative root.
\end{enumerate}
\end{warningbox}
