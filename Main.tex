\subsection{Learning Objectives}
Upon completion of this section, the student will be able to:
\begin{itemize}
    \item Identify the etiology, pathophysiology, and clinical manifestations of acute and chronic pancreatitis, pancreatic cancer, and cholecystitis.
    \item Interpret diagnostic studies for GI disorders including lab values and imaging.
    \item Apply Ranson's Criteria to assess severity of acute pancreatitis.
    \item Develop nursing management plans for patients with GI disorders and TPN.
    \item Recognize and manage complications of acute pancreatitis.
    \item Understand the indications, components, administration, and monitoring of TPN.
    \item Apply clinical judgment through NGN-style questions.
\end{itemize}

\subsection{Acute Pancreatitis}

\begin{definition}
\textbf{Acute Pancreatitis}: An acute inflammatory process of the pancreas with variable involvement of peripancreatic tissues and remote organ systems. It ranges from mild edematous pancreatitis to severe necrotizing pancreatitis.
\end{definition}

\subsubsection{Etiology - Remember "I GET SMASHED"}
\begin{insightbox}
\textbf{Memory Aid: "I GET SMASHED"} for causes of acute pancreatitis:
\begin{itemize}
    \item \textbf{I} - Idiopathic
    \item \textbf{G} - Gallstones (most common cause, 40\%)
    \item \textbf{E} - Ethanol (alcohol, 35\%)
    \item \textbf{T} - Trauma/Tumors
    \item \textbf{S} - Steroids
    \item \textbf{M} - Mumps/Medications (azathioprine, valproic acid, thiazides)
    \item \textbf{A} - Autoimmune (SLE, polyarteritis nodosa)
    \item \textbf{S} - Scorpion sting
    \item \textbf{H} - Hyperlipidemia/Hypercalcemia/Hypothermia
    \item \textbf{E} - ERCP (endoscopic retrograde cholangiopancreatography)
    \item \textbf{D} - Drugs (didanosine, pentamidine)
\end{itemize}
\end{insightbox}

\subsubsection{Pathophysiology}
\begin{stepbox}
\textbf{Pathophysiological Sequence of Acute Pancreatitis:}
\begin{enumerate}
    \item \textbf{Trigger Event}: Obstruction (gallstone) or direct toxicity (alcohol) damages pancreatic acinar cells.
    \item \textbf{Premature Enzyme Activation}: Trypsinogen is converted to trypsin within the pancreatic duct instead of the duodenum.
    \item \textbf{Autodigestion}: Activated trypsin triggers a cascade of other pancreatic enzymes (elastase, phospholipase A2, lipase).
    \item \textbf{Inflammation}: Enzymes cause tissue damage, inflammation, edema, and vascular damage.
    \item \textbf{Systemic Response}: Release of inflammatory mediators (cytokines) can lead to systemic inflammatory response syndrome (SIRS), acute respiratory distress syndrome (ARDS), and multi-organ dysfunction.
\end{enumerate}
\end{stepbox}

\begin{analogy}
Think of acute pancreatitis like a factory where the workers (pancreatic enzymes) accidentally start their dangerous work inside the factory (pancreas) instead of waiting to go to the construction site (duodenum). This causes the factory to start breaking itself down from the inside.
\end{analogy}

\subsubsection{Clinical Manifestations}
\begin{criticalbox}
\textbf{CARDINAL SYMPTOM: Severe, constant, boring epigastric pain} that often radiates to the back (due to retroperitoneal location of pancreas). Pain is typically:
\begin{itemize}
    \item Worsened by eating
   
\item Relieved by sitting forward or leaning over
    \item Associated with nausea and vomiting
\end{itemize}
\end{criticalbox}

\begin{definition}
\textbf{Cullen's Sign}: Periumbilical ecchymosis (bluish discoloration around the umbilicus) indicating hemorrhagic pancreatitis with retroperitoneal bleeding tracking to the umbilicus.
\end{definition}

\begin{definition}
\textbf{Grey Turner's Sign}: Flank ecchymosis (bluish discoloration on the flanks) indicating hemorrhagic pancreatitis with retroperitoneal bleeding tracking to the flanks.
\end{definition}

\begin{warningbox}
Cullen's and Grey Turner's signs are \textbf{late signs} of severe hemorrhagic pancreatitis and indicate a \textbf{poor prognosis}. Their absence does NOT rule out pancreatitis.
\end{warningbox}

\textbf{Other Clinical Manifestations:}
\begin{itemize}
    \item \textbf{Abdominal distention}: Due to paralytic ileus
    \item \textbf{Tachycardia and hypotension}: From third-spacing of fluid and hypovolemia
    \item \textbf{Low-grade fever}: Due to inflammation
    \item \textbf{Jaundice}: If gallstone obstruction is present
    \item \textbf{Tachypnea}: May indicate pleural effusion or ARDS
    \item \textbf{Ascites}: In severe cases
\end{itemize}

\textbf{High-Yield Tip:} Lipase is \textbf{more specific} than amylase for pancreatitis. If only one enzyme is elevated, lipase is the more reliable marker. Both enzymes should be >3 times the upper limit of normal for diagnosis.
\end{insightbox}

\subsubsection{Severity Assessment - Ranson's Criteria}
\begin{definition}
\textbf{Ranson's Criteria}: A scoring system used within 48 hours of admission to predict the severity and mortality risk of acute pancreatitis. Score is calculated at admission and at 48 hours.
\end{definition}

\begin{longtable}{|p{6cm}|p{6cm}|}
\hline
\textbf{At Admission (0 hours)} & \textbf{During Initial 48 Hours} \\
\hline
\endhead
Age > 55 years & Hematocrit decrease > 10\% \\
\hline
WBC > 16,000/mm$^3$ & BUN increase > 5 mg/dL \\
\hline
Blood glucose > 200 mg/dL & Serum calcium < 8 mg/dL \\
\hline
LDH > 350 IU/L & PaO$_2$ < 60 mmHg \\
\hline
AST > 250 IU/L & Base deficit > 4 mEq/L \\
\hline
& Estimated fluid sequestration > 6 L \\
\hline
\end{longtable}

\begin{theorem}
\textbf{Ranson's Criteria Interpretation:}
\begin{itemize}
    \item \textbf{< 3 criteria}: Mild pancreatitis; mortality < 5\%
    \item \textbf{3-4 criteria}: Moderate pancreatitis; mortality 15-20\%
    \item \textbf{5-6 criteria}: Severe pancreatitis; mortality 40\%
    \item \textbf{> 6 criteria}: Very severe pancreatitis; mortality nearly 100\%
\end{itemize}
\end{theorem}

\begin{criticalbox}
\textbf{Key Ranson's Criteria to Remember for Exams:}
\begin{itemize}
    \item Age > 55 (poor prognosis)
    \item WBC > 16,000 (severe inflammation)
    \item Calcium < 8 (saponification - poor prognosis)
    \item Glucose > 200 (endocrine dysfunction)
    \item Fluid sequestration > 6L (third-spacing)
\end{itemize}
\end{criticalbox}

\subsubsection{Treatment and Nursing Management}
\begin{stepbox}
\textbf{Treatment Priorities for Acute Pancreatitis:}
\begin{enumerate}
    \item \textbf{NPO (Nothing by Mouth)}: Rest the pancreas by eliminating oral intake
    \item \textbf{Aggressive IV Fluid Resuscitation}: Crystalloids (NS or LR) at 250-500 mL/hr to maintain adequate perfusion
    \item \textbf{Pain Management}: IV opioids (morphine was traditionally avoided, but current evidence supports its use; hydromorphone and fentanyl are alternatives)
    \item \textbf{Antiemetics}: Ondansetron (Zofran) for nausea/vomiting
    \item \textbf{Proton Pump Inhibitors}: Reduce gastric acid secretion
    \item \textbf{Nutritional Support}: Enteral nutrition via nasojejunal tube preferred over TPN when tolerated
    \item \textbf{Monitor for Complications}: ARDS, renal failure, DIC
\end{enumerate}
\end{stepbox}

\begin{skillbox}
\textbf{Nursing Management Priorities:}
\begin{itemize}
    \item \textbf{Pain Assessment}: Use 0-10 scale; monitor effectiveness of analgesics
    \item \textbf{Fluid Balance}: Strict I\&O; monitor for signs of hypovolemia (tachycardia, hypotension, decreased urine output)
    \item \textbf{Respiratory Assessment}: Auscultate lung sounds; monitor O$_2$ saturation; watch for ARDS
    \item \textbf{Nutritional Status}: Maintain NPO status; monitor glucose levels; prepare for enteral/parenteral nutrition
    \item \textbf{Positioning}: Place in Fowler's position; encourage leaning forward to relieve pain
    \item \textbf{Lab Monitoring}: Serial amylase/lipase, electrolytes (especially calcium), glucose, renal function
    \item \textbf{Psychosocial Support}: Address anxiety; provide education about disease process
\end{itemize}
\end{skillbox}

\begin{sbarbox}
\textbf{SBAR Example for Acute Pancreatitis:}
\begin{itemize}
    \item \textbf{Situation}: "Dr. Smith, this is Nurse Jones calling about Mr. Johnson in room 402. He is a 58-year-old male admitted yesterday for acute pancreatitis."
    \item \textbf{Background}: "His lipase was 1,200 on admission. He has a history of heavy alcohol use."
    \item \textbf{Assessment}: "Over the past 2 hours, his blood pressure has dropped to 88/52, heart rate is 118, and urine output has decreased to 15 mL/hr. He is diaphoretic and reports increased abdominal pain."
    \item \textbf{Recommendation}: "I believe he may be developing hypovolemic shock from third-spacing. I recommend a fluid bolus and possible transfer to the ICU. Would you like me to obtain any additional labs?"
\end{itemize}
\end{sbarbox}

\subsubsection{Complications of Acute Pancreatitis (Continued)}
\begin{warningbox}
\textbf{Key Complications to Monitor:}
\begin{itemize}
    \item \textbf{Pseudocyst}: A localized collection of pancreatic fluid enclosed by a wall of fibrous tissue. Can cause pain, infection, or rupture.
    \item \textbf{Pancreatic Abscess}: Infected necrotic tissue, typically occurring 4-6 weeks after onset. Presents with fever, leukocytosis, and positive cultures.
    \item \textbf{Pancreatic Necrosis}: Death of pancreatic tissue, diagnosed by contrast-enhanced CT. Associated with high mortality.
    \item \textbf{Acute Respiratory Distress Syndrome (ARDS)}: Due to systemic inflammation and fluid shifts.
    \item \textbf{Multi-Organ Dysfunction Syndrome (MODS)}: Failure of two or more organ systems.
    \item \textbf{Chronic Pancreatitis}: Repeated episodes can lead to permanent damage.
\end{itemize}
\end{warningbox}

\begin{stepbox}
\textbf{Nursing Management of Complications:}
\begin{enumerate}
    \item \textbf{Monitor for Signs of Infection}: Fever, increasing WBC, abdominal rigidity.
    \item \textbf{Assess Respiratory Status}: Auscultate lungs, monitor oxygen saturation, watch for signs of ARDS (tachypnea, dyspnea, crackles).
    \item \textbf{Monitor Fluid Balance}: Strict I\&O, daily weights, assess for edema or dehydration.
    \item \textbf{Assess for Hemorrhage}: Check for Cullen's/Grey Turner's signs, monitor hemoglobin/hematocrit, vital signs for hypovolemia.
    \item \textbf{Prepare for Surgical Intervention}: May be needed for pseudocyst drainage, abscess, or necrotic tissue removal.
\end{enumerate}
\end{stepbox}

\subsection{Summary of Acute Pancreatitis}
\begin{itemize}
    \item Most common causes: Gallstones and alcohol.
    \item Cardinal symptom: Severe epigastric pain radiating to the back, worsened by eating.
    \item Key labs: Elevated serum lipase and amylase.
    \item Primary treatment: NPO, IV fluids, pain management, and addressing the underlying cause.
    \item Monitor for life-threatening complications: SIRS, ARDS, MODS, hemorrhage.
    \item Nursing focus: Pain management, fluid resuscitation, monitoring for complications, and patient education.
\end{itemize}

\section{Chronic Pancreatitis}
\begin{definition}
\textbf{Chronic Pancreatitis}: A progressive inflammatory disease of the pancreas characterized by irreversible fibrosis and permanent structural damage, leading to exocrine and endocrine insufficiency.
\end{definition}

\begin{analogy}
If acute pancreatitis is like a sudden factory fire that can be put out, chronic pancreatitis is like slow, smoldering damage that eventually ruins the factory's machinery permanently.
\end{analogy}

\subsection{Etiology}
\begin{itemize}
    \item \textbf{Chronic alcohol use} (most common cause, 70-80\%)
    \item \textbf{Idiopathic}
    \item \textbf{Genetic mutations} (e.g., PRSS1, SPINK1, CFTR)
    \item \textbf{Autoimmune} (type 1 and type 2)
    \item \textbf{Obstruction} (e.g., pancreatic duct stricture, tumors)
    \item \textbf{Hypercalcemia}, \textbf{Hyperlipidemia}
\end{itemize}

\subsection{Pathophysiology}
\begin{stepbox}
\textbf{Pathophysiological Sequence:}
\begin{enumerate}
    \item Repeated injury (e.g., from alcohol) leads to chronic inflammation.
    \item Inflammation causes fibrosis and scarring of pancreatic tissue.
    \item Fibrosis obstructs pancreatic ducts, leading to ductal hypertension.
    \item Progressive loss of acinar cells (exocrine function) and islet cells (endocrine function).
    \item Results in malabsorption and diabetes mellitus (type 3c).
\end{enumerate}
\end{stepbox}

\subsection{Clinical Manifestations}
\begin{criticalbox}
\textbf{Key Symptoms:}
\begin{itemize}
    \item \textbf{Chronic, dull epigastric pain} that may radiate to the back. Often constant and debilitating.
    \item \textbf{Steatorrhea}: Fatty, foul-smelling, bulky stools due to fat malabsorption (exocrine insufficiency).
    \item \textbf{Weight loss} and \textbf{malnutrition} due to malabsorption and fear of eating (pain).
    \item \textbf{Diabetes mellitus} (endocrine insufficiency) - may develop late.
    \item \textbf{Nausea and vomiting}.
\end{itemize}
\end{criticalbox}

\subsection{Diagnostic Studies}
\begin{itemize}
    \item \textbf{Imaging}: CT scan, MRI/MRCP (shows calcifications, ductal dilation, pseudocysts).
    \item \textbf{Endoscopic Retrograde Cholangiopancreatography (ERCP)}: Gold standard for ductal anatomy.
    \item \textbf{Endoscopic Ultrasound (EUS)}: Highly sensitive for early changes.
    \item \textbf{Stool Tests}: Fecal elastase (low in exocrine insufficiency), qualitative fecal fat.
    \item \textbf{Secretin-Stimulated MRCP}: Assesses pancreatic exocrine function.
    \item \textbf{Lab Tests}: May show elevated amylase/lipase during acute flares, but often normal.
\end{itemize}

\subsection{Treatment}
\begin{stepbox}
\textbf{Treatment Goals: Pain management, correcting malabsorption, managing diabetes.}
\begin{enumerate}
    \item \textbf{Pain Management}: Non-opioid analgesics first, then opioids if needed. Celiac plexus block for refractory pain.
    \item \textbf{Pancreatic Enzyme Replacement Therapy (PERT)}: Take with meals to reduce steatorrhea and pain (feedback inhibition of pancreatic secretion).
    \item \textbf{Dietary Modifications}: Low-fat diet, small frequent meals, fat-soluble vitamin supplementation (A, D, E, K).
    \item \textbf{Diabetes Management}: Insulin therapy (often required).
    \item \textbf{Alcohol Cessation}: Absolute abstinence is critical.
    \item \textbf{Surgical Options}: For ductal drainage, resection of inflamed tissue, or total pancreatectomy with islet autotransplant.
\end{enumerate}
\end{stepbox}

\subsection{Nursing Management}
\begin{skillbox}
\textbf{Key Nursing Interventions:}
\begin{itemize}
    \item \textbf{Pain Assessment and Management}: Use appropriate pain scales, administer analgesics as prescribed, monitor for side effects.
    \item \textbf{Nutritional Support}: Collaborate with dietitian, ensure PERT is taken correctly (with meals/snacks), monitor weight and nutritional status.
    \item \textbf{Diabetes Education}: Teach blood glucose monitoring, insulin administration, signs of hypo/hyperglycemia.
    \item \textbf{Psychosocial Support}: Address depression, anxiety, and coping
\item \textbf{Psychosocial Support}: Address depression, anxiety, and coping strategies related to chronic pain and lifestyle changes.
    \item \textbf{Substance Abuse Referral}: Essential for clients with alcohol-related pancreatitis.
    \item \textbf{Monitoring for Complications}: Pseudocysts, pancreatic cancer (increased risk), malnutrition.
\end{itemize}
\end{skillbox}

\begin{warningbox}
\textbf{Common Pitfall}: Do not confuse acute and chronic pancreatitis management. In chronic pancreatitis, the focus shifts from NPO status to long-term nutritional support with PERT and dietary modifications. Clients with chronic pancreatitis should eat regular meals with enzyme replacement, not be NPO.
\end{warningbox}

\subsection{Pancreatic Cancer}

\begin{definition}
\textbf{Pancreatic Cancer}: A malignancy arising from the exocrine or endocrine cells of the pancreas. The most common type is pancreatic ductal adenocarcinoma (PDAC), which accounts for approximately 90\% of cases. It is often called the "silent killer" due to late presentation and poor prognosis.
\end{definition}

\begin{contextbox}
\textbf{Epidemiology}: Pancreatic cancer is the 4th leading cause of cancer death in the United States, with a 5-year survival rate of only about 10-12\%. It is more common in individuals over age 60, with a slight male predominance. The incidence is higher in African Americans.
\end{contextbox}

\subsubsection{Risk Factors}
\begin{itemize}
    \item \textbf{Modifiable Risk Factors}:
    \begin{itemize}
        \item Cigarette smoking (2-3x increased risk)
        \item Chronic pancreatitis
        \item Obesity
        \item Type 2 diabetes mellitus (new-onset diabetes in elderly may be a presenting sign)
        \item High-fat diet
        \item Heavy alcohol use
    \end{itemize}
    \item \textbf{Non-Modifiable Risk Factors}:
    \begin{itemize}
        \item Age >60 years
        \item Male sex
        \item African American race
        \item Family history of pancreatic cancer
        \item Hereditary syndromes (BRCA1/2, Lynch syndrome, Peutz-Jeghers syndrome, familial atypical multiple mole melanoma [FAMMM])
    \end{itemize}
\end{itemize}

\subsubsection{Clinical Manifestations}
\begin{criticalbox}
\textbf{Key Clinical Features}:
\begin{itemize}
    \item \textbf{Painless jaundice}: Classic presentation for pancreatic head tumors (obstruction of common bile duct)
    \item \textbf{Courvoisier's sign}: Painless, palpable gallbladder in the presence of jaundice
    \item \textbf{Abdominal pain}: Epigastric or back pain (often a late symptom)
    \item \textbf{Weight loss}: Unintentional, significant weight loss (cancer cachexia)
    \item \textbf{New-onset diabetes}: Especially in patients over 60 without typical risk factors
    \item \textbf{Pruritus}: Intense itching due to bile salt deposition in skin from obstructive jaundice
    \item \textbf{Pale stools and dark urine}: Due to biliary obstruction
    \item \textbf{Trousseau's syndrome}: Migratory thrombophlebitis (recurrent blood clots in various locations)
    \item \textbf{Depression}: May precede other symptoms by months
\end{itemize}
\end{criticalbox}

\begin{insightbox}
\textbf{High-Yield Tip}: Courvoisier's sign (painless, palpable gallbladder + jaundice) is highly suggestive of pancreatic head cancer. In contrast, jaundice from gallstones typically causes a non-palpable gallbladder due to chronic inflammation and fibrosis.
\end{insightbox}

\subsubsection{Treatment}
\begin{stepbox}
\textbf{Treatment Approach Based on Staging:}
\begin{enumerate}
    \item \textbf{Resectable Disease} (15-20\% of cases):
    \begin{itemize}

    \item \textbf{Borderline Resectable}:
    \begin{itemize}

    \item \textbf{Locally Advanced Unresectable}:

        \item Palliative biliary stenting for jaundice
    \end{itemize}
    \item \textbf{Metastatic Disease}:
    \begin{itemize}

        \item Supportive care and symptom management
    \end{itemize}

\begin{definition}
\textbf{Cholecystitis}: Inflammation of the gallbladder, most commonly caused by obstruction of the cystic duct by gallstones (calculous cholecystitis). It can also occur without stones (acalculous cholecystitis) in critically ill patients.
\end{definition}

\subsection{Risk Factors for Gallstones}
\begin{insightbox}
\textbf{Remember the "5 F's"} for gallstone risk factors:
\begin{itemize}
    \item \textbf{F}at (obesity)
    \item \textbf{F}emale (estrogen increases cholesterol secretion)
    \item \textbf{F}orty (age >40)
    \item \textbf{F}ertile (multiparity, pregnancy)
    \item \textbf{F}air (Caucasian or Native American ethnicity)
\end{itemize}
Additional risk factors: Family history, rapid weight loss, diabetes mellitus, Crohn's disease, cirrhosis, oral contraceptive use, high-fat/low-fiber diet.
\end{insightbox}

\subsection{Clinical Manifestations}
\begin{criticalbox}
\textbf{Classic Presentation of Acute Cholecystitis:}
\begin{itemize}
    \item \textbf{RUQ pain}: Sudden onset, steady, severe; may radiate to right shoulder or scapula (referred pain via phrenic nerve)
    \item \textbf{Murphy's sign}: Inspiratory arrest during deep palpation of RUQ (pain on palpation of inflamed gallbladder during inspiration)
    \item \textbf{Positive Murphy's sign}: Confirmatory for acute cholecystitis
    \item \textbf{Nausea and vomiting}
    \item \textbf{Fever and chills}
    \item \textbf{RUQ tenderness and guarding}
    \item \textbf{Leukocytosis}: Elevated WBC count
    \item \textbf{Pain often occurs after fatty meals}: Typically 1-2 hours postprandially
\end{itemize}
\end{criticalbox}

\begin{warningbox}
\textbf{Complications of Untreated Cholecystitis:}
\begin{itemize}
    \item \textbf{Gangrenous cholecystitis}: Tissue necrosis due to compromised blood supply
    \item \textbf{Gallbladder perforation}: Life-threatening emergency
    \item \textbf{Cholangitis}: Infection of the biliary tract (Charcot's triad: RUQ pain, jaundice, fever)
    \item \textbf{Pancreatitis}: Gallstone migration blocking pancreatic duct
    \item \textbf{Gallstone ileus}: Large stone erodes into bowel causing obstruction
    \item \textbf{Empyema}: Pus-filled gallbladder
\end{itemize}
\end{warningbox}

\subsection{Diagnostic Studies}
\begin{stepbox}
\textbf{Diagnostic Workup:}
\begin{enumerate}
    \item \textbf{Ultrasound (RUQ)}: First-line imaging; shows gallstones, gallbladder wall thickening (>3mm), pericholecystic fluid
    \item \textbf{Hepatobiliary Iminodiacetic Acid (HIDA) Scan}: Most accurate test for acute cholecystitis; non-visualization of gallbladder indicates cystic duct obstruction
    \item \textbf{Lab Studies}:
    \begin{itemize}
        \item CBC: Leukocytosis (elevated WBC)
        \item Liver function tests: May be mildly elevated (ALT, AST, alkaline phosphatase, bilirubin)
        \item Lipase/amylase: Rule out pancreatitis
    \end{itemize}
    \item \textbf{CT Scan}: If complications suspected (perforation, abscess)
    \item \textbf{MRCP}: For evaluation of common bile duct stones
\end{enumerate}
\end{stepbox}

\subsection{Treatment}
\begin{stepbox}
\textbf{Treatment Approach:}
\begin{enumerate}
    \item \textbf{Acute Cholecystitis}:
    \begin{itemize}
        \item NPO status
        \item IV fluids
        \item Pain management: NSAIDs (ketorolac) first-line; opioids if needed (morphine may increase sphincter of Oddi sp
        \item Pain management: NSAIDs (ketorolac) first-line; opioids if needed (morphine may increase sphincter of Oddi spasm but is still used)
        \item Antibiotics: For infection (ceftriaxone + metronidazole or piperacillin-tazobactam)
        \item \textbf{Cholecystectomy}: Definitive treatment; laparoscopic preferred
    \end{itemize}
    \item \textbf{Surgical Options}:
    \begin{itemize}
        \item \textbf{Laparoscopic cholecystectomy}: Gold standard for uncomplicated cholecystitis; performed within 72 hours of symptom onset
        \item \textbf{Open cholecystectomy}: For complicated cases (gangrene, perforation, severe adhesions)
        \item \textbf{Percutaneous cholecystostomy}: For patients who are poor surgical candidates; drains infected bile
    \end{itemize}
    \item \textbf{Choledocholithiasis (CBD stones)}:
    \begin{itemize}
        \item ERCP with sphincterotomy and stone extraction
        \item May be performed before or after cholecystectomy
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsection{Nursing Management}
\begin{skillbox}
\textbf{Pre-Operative Nursing Care:}
\begin{itemize}
    \item NPO status and IV fluid administration
    \item Pain assessment and management
    \item Administer prescribed antibiotics
    \item Monitor vital signs for signs of worsening infection
    \item Assess for signs of complications (perforation, sepsis)
    \item Pre-operative teaching for laparoscopic cholecystectomy
\end{itemize}

\textbf{Post-Operative Nursing Care (Laparoscopic Cholecystectomy):}
\begin{itemize}
    \item \textbf{Assess surgical sites}: 3-4 small incisions; check for bleeding, drainage, signs of infection
    \item \textbf{Monitor for complications}: Bile leak, retained CBD stones, injury to bile ducts
    \item \textbf{Manage pain}: Shoulder pain is common due to CO\textsubscript{2} insufflation; encourage ambulation
    \item \textbf{Diet advancement}: Typically clear liquids on POD 0, advance as tolerated
    \item \textbf{Dietary teaching}: Low-fat diet for 4-6 weeks post-operatively; fat-soluble vitamin absorption usually normalizes
    \item \textbf{Discharge teaching}: Report fever, jaundice, increasing abdominal pain, or persistent nausea/vomiting
\end{itemize}
\end{skillbox}

\begin{clinicalpearl}
\textbf{Key Point}: After cholecystectomy, the liver continues to produce bile, but it flows directly into the duodenum instead of being stored in the gallbladder. Most patients can return to a normal diet within weeks, though some may experience temporary diarrhea due to continuous bile flow. Advise patients to eat smaller, more frequent meals initially.
\end{clinicalpearl}

\subsection{Total Parenteral Nutrition (TPN)}
\begin{definition}
\textbf{Total Parenteral Nutrition (TPN)}: A method of providing complete nutrition intravenously, bypassing the gastrointestinal tract. TPN contains all essential nutrients including carbohydrates, proteins, fats, electrolytes, vitamins, minerals, and trace elements.
\end{definition}

\begin{analogy}
Think of TPN as an "IV meal" that provides all the nutrition a person would normally get from eating food, but delivered directly into the bloodstream through a large vein. Just as a car needs fuel to run, the body needs nutrients to function, and TPN is the alternative fuel delivery system when the normal "gas tank" (GI tract) is out of service.
\end{analogy}

\subsubsection{Indications for TPN}
\begin{criticalbox}
\textbf{TPN is indicated when the GI tract cannot be used for nutrition for 7-14 days or longer:}
\begin{itemize}
    \item \textbf{GI tract dysfunction}: Severe Crohn's disease, bowel obstruction, short bowel syndrome, severe pancreatitis
    \item \textbf{Post-operative}: Major GI surgery with prolonged ileus, bowel resection
    \item \textbf{Hypermetabolic states}: Severe burns, major trauma, sepsis
    \item \textbf{GI fistulas}: High-output fistulas preventing adequate oral/enteral intake
    \item \textbf{Severe malnutrition}: When enteral nutrition is insufficient or contraindicated
    \item \textbf{Bone marrow transplant}: Severe mucositis preventing oral intake
\end{itemize}
\end{criticalbox}

\begin{warningbox}
\textbf{Rule}: Always try enteral nutrition (tube feeding) first if the GI tract is functional! The GI tract should be used whenever possible because:
\begin{itemize}
    \item Maintains gut mucosal integrity
    \item Prevents bacterial translocation
    \item Lower infection risk
    \item More physiologic
    \item Less expensive
    \item Fewer complications
\end{itemize}
TPN is reserved for situations where enteral nutrition is truly not possible.
\end{warningbox}

\begin{insightbox}
\textbf{Key Point}: TPN solutions are \textbf{hyperosmolar} (typically 1200-1800 mOsm/L) and MUST be administered through a \textbf{central venous catheter} (central line) to prevent severe phlebitis and tissue damage. Never infuse TPN through a peripheral IV line.
\end{insightbox}

\subsubsection{TPN Administration}
\begin{stepbox}
\textbf{Administration Guidelines:}
\begin{enumerate}
    \item \textbf{Central Venous Access}: TPN must be administered through a central line (PICC, subclavian, internal jugular) due to high osmolarity.
    \item \textbf{Infusion Pump}: Use an infusion pump for controlled delivery.
    \item \textbf{Line Dedicated}: TPN line should be dedicated (no other medications or fluids through same line).
    \item \textbf{Filter}: Use a 0.22 micron filter for lipid-free solutions and a 1.2 micron filter for solutions containing lipids.
    \item \textbf{Gradual Initiation}: Start at 50\% of goal rate and advance by 25\% daily until goal rate achieved.
    \item \textbf{Cycling}: Can be administered continuously (24 hours) or cyclically (e.g., 12-18 hours) for patients stable on TPN.
\end{enumerate}
\end{stepbox}

\begin{criticalbox}
\textbf{CRITICAL: Never administer TPN through a peripheral IV!} The high osmolarity will cause phlebitis, vein damage, and extravasation injury. Central access is mandatory.
\end{criticalbox}

\begin{warningbox}
\textbf{Common Error}: Do not add any medications to the TPN bag unless specifically ordered and compatible. Many medications are incompatible with TPN components and can cause precipitation or inactivation.
\end{warningbox}

\subsubsection{Monitoring During TPN Therapy}
\begin{stepbox}
\textbf{Monitoring Parameters:}
\begin{enumerate}
    \item \textbf{Daily}:
    \begin{itemize}
        \item Weight
        \item Strict intake and output
        \item Vital signs (temperature, heart rate, blood pressure)
        \item Blood glucose (every 6 hours initially, then daily once stable)
        \item Electrolytes (Na, K, Cl, CO2, Ca, Mg, PO4) - daily until stable
        \item BUN and creatinine
        \item Triglycerides (if on lipid emulsions)
    \end{itemize}
    \item \textbf{Twice Weekly}:
    \begin{itemize}
        \item Liver function tests (ALT, AST, alkaline phosphatase, bilirubin)
        \item Albumin and prealbumin (nutritional markers)
        \item Complete blood count
        \item Prothrombin time/INR
    \end{itemize}
    \item \textbf{Weekly}:
    \begin{itemize}
        \item Trace element levels (if on long-term TPN)
        \item Zinc, copper, manganese, selenium levels
        \item Vitamin levels (if indicated)
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{Key Monitoring Points:}
\begin{itemize}
    \item \textbf{Blood Glucose}: TPN contains dextrose, which can cause hyperglycemia. Insulin may be added to the TPN bag or given subcutaneously.
    \item \textbf{Electrolytes}: TPN can cause rapid electrolyte shifts. Monitor closely and adjust TPN composition as needed.
    \item \textbf{Triglycerides}: Lipid emulsions can cause hypertriglyceridemia. Hold lipids if triglycerides > 400 mg/dL.
\end{itemize}
\end{insightbox}

\begin{criticalbox}
\textbf{Refeeding Syndrome}: A potentially fatal complication when nutrition is reintroduced to severely malnourished patients. Characterized by:
\begin{itemize}
    \item Hypophosphatemia (most characteristic)
    \item Hypokalemia
    \item Hypomagnesemia
    \item Fluid overload
    \item Cardiac arrhythmias
\end{itemize}
\textbf{Prevention}: Start TPN at 50\% of goal, advance slowly, monitor electrolytes closely, and correct deficiencies before starting.
\end{criticalbox}

\subsubsection{Discontinuing TPN}
\begin{stepbox}
\textbf{Steps for Discontinuing TPN:}
\begin{enumerate}
    \item \textbf{Gradual Weaning}: Do not stop abruptly! Taper over 1-2 hours to prevent hypoglycemia.
    \item \textbf{Monitor Blood Glucose}: Check glucose 1 hour after discontinuation and periodically for 24 hours.
    \item \textbf{Transition to Enteral/Oral Nutrition}: Begin enteral or oral nutrition as soon as possible.
    \item \textbf{Central Line Removal}: Remove central line when no longer needed; send tip for culture if infection suspected.
\end{enumerate}
\end{stepbox}

\begin{warningbox}
\textbf{Warning}: Abrupt discontinuation of TPN can cause rebound hypoglycemia because the body continues to produce insulin in response to the dextrose load. Always taper TPN over 1-2 hours.
\end{warningbox}

\end{warningbox}

\section{Total Parenteral Nutrition (TPN)}
\begin{definition}
\textbf{Total Parenteral Nutrition (TPN)}: Intravenous administration of all necessary nutrients (carbohydrates, proteins, fats, electrolytes, vitamins, trace elements) directly into the bloodstream, bypassing the gastrointestinal tract.
\end{definition}

\begin{analogy}
Think of TPN as an "IV meal" that provides complete nutrition when the gut cannot be used. Just as a car needs fuel to run, the body needs nutrients to function. When the "fuel line" (GI tract) is broken, we must inject fuel directly into the "engine" (bloodstream).
\end{analogy}

\subsection{Indications for TPN}
\begin{criticalbox}
\textbf{TPN is indicated when the GI tract is:}
\begin{itemize}
    \item \textbf{Non-functional}: Severe Crohn's disease, bowel obstruction, short bowel syndrome, severe pancreatitis
    \item \textbf{Post-operative}: Major GI surgery with prolonged ileus, bowel resection
    \item \textbf{Hypermetabolic states}: Severe burns, major trauma, sepsis
    \item \textbf{GI fistulas}: High-output fistulas preventing adequate oral/enteral intake
    \item \textbf{Severe malnutrition}: When enteral nutrition is insufficient or contraindicated
    \item \textbf{Bone marrow transplant}: Severe mucositis preventing oral intake
\end{itemize}
\end{criticalbox}

\begin{warningbox}
\textbf{Rule}: Always try enteral nutrition (tube feeding) first if the GI tract is functional! The GI tract should be used whenever possible because:
\begin{itemize}
    \item Maintains gut mucosal integrity
    \item Prevents bacterial translocation
    \item Lower infection risk
    \item More physiologic
    \item Less expensive
    \item Fewer complications
\end{itemize}
TPN is reserved for situations where enteral nutrition is truly not possible.
\end{warningbox}

\subsection{TPN Administration}
\begin{stepbox}
\textbf{Administration Guidelines:}
\begin{enumerate}
    \item \textbf{Central Venous Access}: TPN must be administered through a central line (PICC, subclavian, internal jugular) due to high osmolarity.
    \item \textbf{Infusion Pump}: Use an infusion pump for controlled delivery.
    \item \textbf{Line Dedicated}: TPN line should be dedicated (no other medications or fluids through same line).
    \item \textbf{Filter}: Use a 0.22 micron filter for lipid-free solutions and a 1.2 micron filter for solutions containing lipids.
    \item \textbf{Gradual Initiation}: Start at 50\% of goal rate and advance by 25\% daily until goal rate achieved.
    \item \textbf{Cycling}: Can be administered continuously over 24 hours or cycled (e.g., 12-16 hours on, 8-12 hours off).
    \item \textbf{Sterile Technique}: Strict aseptic technique during line insertion, dressing changes, and medication administration.
    \item \textbf{Bag Management}: TPN bags should be hung within 24 hours of preparation and changed every 24 hours.
\end{enumerate}
\end{stepbox}

\begin{warningbox}
\textbf{Critical Safety Alert}: Never add anything to a TPN bag unless specifically ordered. TPN is a sterile preparation, and adding medications can cause precipitation, contamination, or incompatibility. Always use a separate line for medications.
\end{warningbox}

\subsection{Monitoring During TPN Therapy}
\begin{stepbox}
\textbf{Essential Monitoring:}
\begin{enumerate}
    \item \textbf{Daily Weights}: At the same time each day, with same scale.
    \item \textbf{Intake and Output}: Strict I\&O monitoring.
    \item \textbf{Vital Signs}: Every 4-8 hours, more frequently initially.
    \item \textbf{Blood Glucose}: Every 6 hours initially, then daily once stable. Hyperglycemia is common due to high dextrose load.
    \item \textbf{Electrolytes}: Daily initially (Na, K, Cl, CO2, Ca, Mg, PO4), then 2-3 times weekly once stable.
    \item \textbf{Liver Function Tests}: Weekly (ALT, AST, alkaline phosphatase, bilirubin).
    \item \textbf{Triglycerides}: Weekly if on lipid emulsions.
    \item \textbf{Nitrogen Balance}: Periodically to assess nutritional adequacy.
    \item \textbf{Central Line Site}: Daily assessment for infection, redness, swelling, tenderness.
    \item \textbf{Temperature}: Every 4-8 hours to monitor for infection.
\end{enumerate}
\end{stepbox}

\begin{clinicalpearl}
\textbf{Refeeding Syndrome}: A potentially fatal complication that can occur when TPN is initiated in severely malnourished patients. Rapid carbohydrate metabolism causes shifts in electrolytes (particularly phosphorus, potassium, and magnesium) leading to:
\begin{itemize}
    \item Severe hypophosphatemia (can cause cardiac arrest, respiratory failure)
    \item Hypokalemia
    \item Hypomagnesemia
    \item Fluid overload
    \item Cardiac arrhythmias
\end{itemize}
\textbf{Prevention}: Start TPN at 50\% of goal rate in
\textbf{Prevention}: Start TPN at 50\% of goal rate in severely malnourished patients, monitor electrolytes closely, and correct deficiencies before starting nutrition.
\end{clinicalpearl}

\subsection{Discontinuing TPN}
\begin{stepbox}
\textbf{Weaning TPN:}
\begin{enumerate}
    \item \textbf{Gradual Weaning}: Never abruptly discontinue TPN. Wean over 1-2 hours to prevent rebound hypoglycemia.
    \item \textbf{Monitor Blood Glucose}: Check blood glucose 1 hour after discontinuation and every 4-6 hours for 24 hours.
    \item \textbf{Transition to Enteral/Oral Nutrition}: Begin when GI function returns. Overlap TPN with enteral nutrition to ensure adequate intake.
    \item \textbf{Line Removal}: Remove central line once TPN is discontinued and no longer needed. Send catheter tip for culture if infection suspected.
\end{enumerate}
\end{stepbox}

\begin{warningbox}
\textbf{Critical Warning:} Abrupt discontinuation of TPN can cause severe hypoglycemia due to rebound insulin secretion. Always wean TPN gradually over 1-2 hours while monitoring blood glucose.
\end{warningbox}

\section{Renal and Urologic Nursing}
\subsection{Acute Kidney Injury (AKI)}
\begin{definition}
\textbf{Acute Kidney Injury (AKI):} A sudden decline in kidney function occurring over hours to days, characterized by an increase in serum creatinine, decrease in urine output, or both. Previously called acute renal failure.
\end{definition}

\begin{analogy}
Think of the kidneys as a water filtration plant. AKI is like a sudden breakdown in the filtration system where waste products (like creatinine) accumulate in the blood (the water supply) and urine production decreases or stops.
\end{analogy}

\subsubsection{Classification of AKI}

\hline

\begin{criticalbox}
\textbf{Prerenal AKI:} The most common type, caused by decreased blood flow to the kidneys. Early recognition and treatment can prevent progression to intrinsic AKI. Common causes: dehydration, hemorrhage, heart failure, sepsis.
\end{criticalbox}

\subsubsection{Phases of AKI}
\begin{stepbox}
\textbf{Four Phases of AKI:}
\begin{enumerate}
    \item \textbf{Initiation Phase}: From initial insult to onset of decreased GFR. May last hours to days. Kidney injury begins but may be reversible with prompt treatment.
    \item \textbf{Oliguric Phase}: Characterized by decreased urine output (<400 mL/day or <0.5 mL/kg/hour for adults). Lasts 1-3 weeks. Fluid/electrolyte imbalances occur. Not all patients develop oliguria (non-oliguric AKI has better prognosis).
    \item \textbf{Diuretic Phase}: Gradual increase in urine output (1-3 L/day) as tubular function begins to recover. Risk of dehydration and electrolyte imbalances (hyponatremia, hypokalemia).
    \item \textbf{Recovery Phase}: Return to normal or near-normal kidney function over 3-12 months. May have residual deficits.
\end{enumerate}
\end{stepbox}

\subsubsection{Clinical Manifestations}
\begin{highlightbox}
\textbf{Key Clinical Manifestations by System:}
\begin{multicols}{2}
\begin{itemize}
    \item \textbf{Urinary Changes}:
    \begin{itemize}
        \item Oliguria (<400 mL/day)
        \item Anuria (<100 mL/day)
        \item Hematuria
        \item Proteinuria
        \item Casts in urine
    \end{itemize}
    \item \textbf{Fluid/Electrolyte Imbalances}:
    \begin{itemize}
        \item Hyperkalemia (most dangerous)
        \item Hyponatremia
        \item Hyperphosphatemia
        \item Hypocalcemia
        \item Metabolic acidosis
        \item Fluid overload (edema, hypertension)
    \end{itemize}
    \columnbreak
    \item \textbf{Uremic Symptoms}:
    \begin{itemize}
        \item Nausea/vomiting
        \item Anorexia
        \item Fatigue
        \item Confusion
        \item Seizures
        \item Pericarditis
        \item Peripheral neuropathy
    \end{itemize}
    \item \textbf{Hematologic}:
    \begin{itemize}
        \item Anemia
        \item Bleeding tendencies
        \item Increased infection risk
    \end{itemize}
\end{itemize}
\end{multicols}
\end{highlightbox}

\subsubsection{Diagnostic Studies}
\begin{stepbox}
\textbf{Diagnostic Workup for AKI:}
\begin{enumerate}
    \item \textbf{Serum Creatinine}: Rising creatinine confirms AKI. Increase of ≥0.3 mg/dL within 48 hours or ≥1.5 times baseline within 7 days.
    \item \textbf{Blood Urea Nitrogen (BUN)}: Elevated.
\textbf{BUN/Creatinine Ratio}: Typically <15:1 in intrinsic renal disease (vs. >20:1 in prerenal).
    \item \textbf{Urinalysis}: Check for casts (muddy brown granular casts in ATN), protein, blood, specific gravity.
    \item \textbf{Urine Sodium}: >40 mEq/L in intrinsic renal damage (vs. <20 mEq/L in prerenal).
    \item \textbf{Fractional Excretion of Sodium (FENa)}: >2\% in intrinsic renal disease, <1\% in prerenal.
    \item \textbf{Renal Ultrasound}: Rule out obstruction, assess kidney size.
    \item \textbf{Serum Electrolytes}: Potassium, sodium, calcium, phosphorus, magnesium.
    \item \textbf{ABG}: Assess for metabolic acidosis.
    \item \textbf{CBC}: Check for anemia, infection.
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{Quick Differentiation of AKI Types:}
\begin{itemize}
    \item \textbf{Prerenal}: BUN/Cr >20:1, Urine Na <20 mEq/L, FENa <1\%, Urine osmolality >500 mOsm/kg
    \item \textbf{Intrinsic}: BUN/Cr <15:1, Urine Na >40 mEq/L, FENa >2\%, Urine osmolality <350 mOsm/kg
    \item \textbf{Postrenal}: Obstruction on ultrasound, may have variable lab values
\end{itemize}
\end{insightbox}

\subsubsection{Treatment and Nursing Management}
\begin{stepbox}
\textbf{Management of AKI:}
\begin{enumerate}
    \item \textbf{Treat Underlying Cause}: Address prerenal factors (fluid resuscitation), remove nephrotoxins, relieve obstruction.
    \item \textbf{Fluid Management}:
    \begin{itemize}
        \item Prerenal: IV fluid resuscitation with isotonic solutions
        \item Intrinsic/Overload: Fluid restriction, diuretics (furosemide) if needed
        \item Strict I\&O monitoring, daily weights
    \end{itemize}
    \item \textbf{Electrolyte Management}:
    \begin{itemize}
        \item \textbf{Hyperkalemia}: Most dangerous! ECG monitoring, calcium gluconate (cardiac protection), insulin+glucose, sodium bicarbonate, kayexalate, dialysis
        \item Hyponatremia: Fluid restriction
        \item Hyperphosphatemia: Phosphate binders (calcium acetate, sevelamer)
        \item Hypocalcemia: Calcium supplementation (cautiously with hyperphosphatemia)
    \end{itemize}
    \item \textbf{Acidosis Management}: Sodium bicarbonate if pH <7.2 or HCO3 <15 mEq/L
    \item \textbf{Nutrition}: Low potassium, low phosphorus, low sodium diet; adequate protein (0.8-1 g/kg/day unless on dialysis)
    \item \textbf{Medication Management}: Adjust doses for renal function, avoid nephrotoxins (NSAIDs, aminoglycosides, contrast dye)
    \item \textbf{Dialysis Indications}: Refractory hyperkalemia, severe acidosis, fluid overload unresponsive to diuretics, uremic symptoms (encephalopathy, pericarditis)
    \item \textbf{Infection Prevention}: Meticulous care of invasive lines, monitor for signs of infection
    \item \textbf{Skin Care}: Meticulous care due to edema and uremia-related pruritus
\end{enumerate}
\end{stepbox}

\begin{criticalbox}
\textbf{Hyperkalemia Management (Priority!):}
\begin{enumerate}
    \item \textbf{ECG Monitoring}: Watch for peaked T waves, widened QRS, sine wave pattern
    \item \textbf{Cardiac Membrane Stabilization}: Calcium gluconate or calcium chloride (does NOT lower K+)
    \item \textbf{Shift K+ into cells}: Insulin + glucose, albuterol, sodium bicarbonate
    \item \textbf{Remove K+ from body}: Kayexalate, loop diuretics, dialysis
    \item \textbf{Dietary Restriction}: Low potassium foods
\end{enumerate}
\end{criticalbox}

\begin{warningbox}
\textbf{Common Pitfall}: In AKI, do not give potassium-sparing diuretics (spironolactone, triamterene) or ACE inhibitors/ARBs as they can worsen hyperkalemia. Also avoid magnesium-containing antacids (can cause hypermagnesemia).
\end{warningbox}

\subsection{Chronic Kidney Disease (CKD)}
\begin{definition}
\textbf{Chronic Kidney Disease (CKD):} Progressive, irreversible loss of kidney function over months to years. Defined as abnormalities in kidney structure or function present for >3 months. Leads to accumulation of waste products, fluid/electrolyte imbalances, and endocrine dysfunction.
\end{definition}

\begin{analogy}
Think of CKD like a slow leak in a tire. At first, the tire (kidney) works fine despite the leak. Over time, the tire becomes increasingly flat (loss of function) until it can no longer support the car (body). Unlike acute flat tire (AKI), the slow leak is permanent and requires ongoing management (patching, air, or replacement).
\end{analogy}

2 & 60-89 & Mild ↓ in GFR & Monitor progression; manage comorbidities \\
\hline
3a & 45-59 & Moderate ↓ in GFR & Complications begin: anemia, bone disease \\
\hline
3b & 30-44 & Moderate ↓ in GFR & More pronounced complications; prepare for renal replacement \\
\hline

\begin{clinicalpearl}
\textbf{GFR Estimation}: GFR is estimated using the CKD-EPI equation based on serum creatinine, age, sex, and race. Normal GFR is approximately 90-120 mL/min/1.73 m². A GFR <60 mL/min for 3 months indicates CKD.
\end{clinicalpearl}

\subsubsection{Causes of CKD}
\begin{highlightbox}
\textbf{Leading Causes of CKD:}
\begin{enumerate}
    \item \textbf{Diabetes Mellitus} (Type 1 and 2): #1 cause in developed countries (diabetic nephropathy)
    \item \textbf{Hypertension}: #2 cause (hypertensive nephrosclerosis)
    \item \textbf{Glomerulonephritis}: Various types (IgA nephropathy, lupus nephritis)
    \item \textbf{Polycystic Kidney Disease}: Autosomal dominant genetic disorder
    \item \textbf{Chronic Obstruction}: BPH, kidney stones, strictures
    \item \textbf{Recurrent Infections}: Chronic pyelonephritis
    \item \textbf{Drug Toxicity}: NSAIDs, lithium, calcineurin inhibitors
    \item \textbf{Vascular Diseases}: Renal artery stenosis
\end{enumerate}
\end{highlightbox}

\subsubsection{Pathophysiology}
\begin{stepbox}
\textbf{Progressive Nature of CKD:}
\begin{enumerate}
    \item \textbf{Initial Injury}: Damages nephrons (functional units)
    \item \textbf{Compensatory Hyperfiltration}: Remaining nephrons work harder to maintain GFR
    \item \textbf{Glomerular Hypertension}: Increased pressure in glomeruli causes further damage
    \item \textbf{Tubulointerstitial Fibrosis}: Scarring of kidney tissue
    \item \textbf{Progressive Nephron Loss}: More nephrons are destroyed
    \item \textbf{Loss of Endocrine Function}: Decreased erythropoietin (anemia), decreased vitamin D activation (bone disease), increased renin (hypertension)
    \item \textbf{Uremia}: Accumulation of waste products (urea, creatinine, uric acid) causing systemic toxicity
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{Key Concept}: CKD progression can be slowed but not reversed. The goal is to slow progression by controlling blood pressure (<130/80 mmHg), blood glucose (HbA1c <7%), and reducing proteinuria (ACE inhibitors/ARBs).
\end{insightbox}

\subsubsection{Clinical Manifestations}
\begin{stepbox}
\textbf{Manifestations by System (develop as GFR declines):}
\begin{enumerate}
    \item \textbf{Cardiovascular}:
    \begin{itemize}
        \item Hypertension (fluid overload, increased renin)
        \item Heart failure (fluid overload)
        \item Pericarditis (uremia)
        \item Accelerated atherosclerosis
    \end{itemize}
    \item \textbf{Hematologic}:
    \begin{itemize}
        \item Anemia (decreased erythropoietin)
        \item Bleeding tendency (platelet dysfunction)
        \item Increased infection risk
    \end{itemize}
    \item \textbf{Musculoskeletal}:
    \begin{itemize}
        \item Renal osteodystrophy (secondary hyperparathyroidism, vitamin D deficiency)
        \item Bone pain, fractures
        \item Muscle cramps
    \end{itemize}
    \item \textbf{Neurologic}:
    \begin{itemize}
        \item Uremic encephalopathy (confusion, seizures, coma)
        \item Peripheral neuropathy (restless legs, paresthesias)
    \end{itemize}
    \item \textbf{GI}:
    \begin{itemize}
        \item Nausea, vomiting, anorexia
        \item Uremic fetor (ammonia breath)
        \item GI bleeding
    \end{itemize}
    \item \textbf{Integumentary}:
    \begin{itemize}
        \item Pruritus (uremia, calcium-phosphorus deposits)
        \item Sallow, yellow-brown skin color
        \item Uremic frost (late finding)
    \end{itemize}
    \item \textbf{Endocrine}:
    \begin{itemize}
        \item Sexual dysfunction
        \item Amenorrhea in women
        \item Glucose intolerance
    \end{itemize}
    \item \textbf{Fluid/Electrolyte}:
    \begin{itemize}
        \item Hyperkalemia
        \item Metabolic acidosis
        \item Hyperphosphatemia
        \item Hypocalcemia
        \item Sodium and water retention
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsubsection{Diagnostic Studies}
\begin{stepbox}
\textbf{Diagnostic Workup for CKD:}
\begin{enumerate}
    \item \textbf{Serum Creatinine and GFR Estimation}: Serial measurements to track progression
    \item \textbf{Urinalysis}: Proteinuria, hematuria, casts
    \item \textbf{Urine Albumin-to-Creatinine Ratio (ACR)}: >30 mg/g indicates albuminuria
    \item \textbf{Serum Electrolytes}: Potassium, bicarbonate, calcium, phosphorus
    \item \textbf{CBC}: Anemia (normocytic, normochromic)
    \item \textbf{Iron Studies}: Ferritin, TIBC, serum iron (to evaluate anemia)
    \item \textbf{Intact PTH}: Elevated in secondary hyperparathyroidism
    \item \textbf{Vitamin D Levels}: 25-hydroxyvitamin D often low
    \item \textbf{Renal Ultrasound}: Small, echogenic kidneys in advanced CKD (vs. large in diabetic nephropathy early)
    \item \textbf{Kidney Biopsy}: May be needed to determine cause if unclear
\end{enumerate}
\end{stepbox}

\subsubsection{Treatment and Nursing Management}
\begin{stepbox}
\textbf{Management of CKD:}
\begin{enumerate}
    \item \textbf{Slow Progression}:
    \begin{itemize}
        \item Blood pressure control: ACE inhibitors/ARBs (first-line for proteinuria)
        \item Glycemic control in diabetics
        \item Low-protein diet (0.6-0.8 g/kg/day in stages 3-5)
        \item Avoid nephrotoxins
        \item Smoking cessation
    \end{itemize}
    \item \textbf{Manage Complications}:
    \begin{itemize}
        \item \textbf{Anemia}: Erythropoiesis-stimulating agents (epoetin alfa, darbepoetin) + iron supplementation
        \item \textbf{Bone Disease}: Phosphate binders, vitamin D analogs (calcitriol), dietary phosphorus restriction
        \item \textbf{Metabolic Acidosis}: Sodium bicarbonate supplementation
        \item \textbf{Fluid Overload}: Sodium/fluid restriction, diuretics
        \item \textbf{Hyperkalemia}: Dietary restriction, potassium binders (patiromer, sodium zirconium cyclosilicate)
        \item \textbf{Dyslipidemia}: Statins
    \end{itemize}
    \
\item \textbf{Prepare for Renal Replacement Therapy}:
    \begin{itemize}
        \item Educate on dialysis options (hemodialysis, peritoneal dialysis) and transplantation
        \item Create vascular access (AV fistula/graft) when GFR <25 mL/min
        \item Refer for transplant evaluation when GFR <20 mL/min
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{criticalbox}
\textbf{ACE Inhibitors/ARBs in CKD}: These are first-line for blood pressure control and proteinuria reduction. However, they can cause hyperkalemia and acute kidney injury (especially in renal artery stenosis). Monitor serum creatinine and potassium within 1-2 weeks of starting or adjusting dose. A small rise in creatinine (up to 30\%) is acceptable if stable.
\end{criticalbox}

\begin{warningbox}
\textbf{NSAIDs in CKD}: Avoid nonsteroidal anti-inflammatory drugs (NSAIDs) in CKD patients. They reduce renal blood flow, worsen kidney function, increase potassium levels, and can precipitate acute kidney injury.
\end{warningbox}

\section{Dialysis}
\subsection{Principles of Dialysis}
\begin{definition}
\textbf{Dialysis}: A treatment that removes waste products and excess fluid from the blood when the kidneys can no longer perform these functions adequately. It works by diffusion (movement of solutes from high to low concentration) and ultrafiltration (removal of fluid by pressure gradient).
\end{definition}

\begin{analogy}
Think of dialysis like a coffee filter. The blood is the coffee grounds mixed with water, and the dialyzer (artificial kidney) is the filter. The filter allows waste products and excess fluid to pass through while keeping important blood cells and proteins in the blood.
\end{analogy}

\subsection{Hemodialysis (HD)}
\subsubsection{Vascular Access}
\begin{stepbox}
\textbf{Types of Vascular Access (in order of preference):}
\begin{enumerate}
    \item \textbf{Arteriovenous (AV) Fistula}: Surgical connection between an artery and vein (usually in forearm). Takes 6-8 weeks to mature. Lowest infection risk, longest lasting. \textbf{Nursing}: Assess for thrill and bruit, no blood pressures or venipuncture on that arm.
    \item \textbf{AV Graft}: Synthetic tube connecting artery and vein. Ready to use in 2-3 weeks. Higher risk of infection and thrombosis than fistula.
    \item \textbf{Central Venous Catheter}: Temporary access (subclavian, internal jugular, or femoral). Used for immediate dialysis or while fistula/graft matures. Highest infection risk. \textbf{Nursing}: Strict aseptic technique, assess for signs of infection.
\end{enumerate}
\end{stepbox}

\subsubsection{Hemodialysis Procedure}
\begin{stepbox}
\textbf{Steps During Hemodialysis:}
\begin{enumerate}
    \item Patient connected to dialysis machine via vascular access.
    \item Blood pumped through dialyzer at 300-500 mL/min.
    \item Dialysate (special fluid) flows countercurrent to blood in dialyzer.
    \item Waste products (urea, creatinine, electrolytes) diffuse across membrane into dialysate.
    \item Excess fluid removed by ultrafiltration (pressure gradient).
    \item Cleaned blood returned to patient.
    \item Treatment lasts 3-5 hours, typically 3 times per week.
\end{enumerate}
\end{stepbox}

\subsection{Peritoneal Dialysis (PD)}
\subsubsection{Types of Peritoneal Dialysis}
\begin{stepbox}
\textbf{Types of PD:}
\begin{enumerate}
    \item \textbf{Continuous Ambulatory Peritoneal Dialysis (CAPD)}:
    \begin{itemize}
        \item Manual exchanges 4-5 times daily
        \item 2 L dialysate infused, dwells 4-6 hours, then drained
        \item No machine required
        \item Patient can perform during daily activities
    \end{itemize}
    \item \textbf{Automated Peritoneal Dialysis (APD)}:
    \begin{itemize}
        \item Cycler machine performs exchanges overnight while patient sleeps
        \item Typically 8-10 hours of treatment
        \item May have one daytime dwell (daytime exchange)
        \item Allows more freedom during day
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsubsection{PD Procedure}
\begin{stepbox}
\textbf{Steps for CAPD Exchange:}
\begin{enumerate}
    \item Prepare clean workspace and gather supplies.
    \item Perform hand hygiene and don mask.
    \item Connect PD bag to catheter using sterile technique.
    \item Drain old dialysate from peritoneal cavity (20-30 minutes).
    \item Infuse new dialysate (10-15 minutes).
    \item Disconnect and cap catheter.
    \item Record intake/output, weight, and vital signs.
\end{enumerate}
\end{stepbox}

\subsection{Continuous Renal Replacement Therapy (CRRT)}
\begin{definition}
\textbf{CRRT}: A slow, continuous form of dialysis used for hemodynamically unstable patients in the ICU (e.g., sepsis, multi-organ failure). It provides gradual fluid and solute removal over 24 hours, better tolerated than intermittent hemodialysis.
\end{definition}

\begin{stepbox}
\textbf{Types of CRRT:}
\begin{enumerate}
    \item \textbf{Continuous Venovenous Hemofiltration (CVVH)}: Uses convective solute removal (solvent drag).
    \item \textbf{Continuous Venovenous Hemodialysis (CVVHD)}: Uses diffusive solute removal (countercurrent dialysate flow).
    \item \textbf{Continuous Venovenous Hemodiafiltration (CVVHDF)}: Combines both diffusion and convection.
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{Key Differences from Hemodialysis}: CRRT is slower (100-200 mL/min blood flow), continuous (24 hours/day), and uses a central venous catheter. It's better for patients with hypotension, cerebral edema, or fluid overload who cannot tolerate rapid fluid shifts.
\end{insightbox}

\subsection{Immunosuppression Therapy}
\begin{stepbox}
\textbf{Standard Immunosuppressive Regimen:}
\begin{enumerate}
    \item \textbf{Induction Therapy}: High-dose immunosuppression at transplant (e.g., basiliximab, thymoglobulin).
    \item \textbf{Maintenance Therapy}: Triple drug regimen:
    \begin{itemize}
        \item \textbf{Calcineurin Inhibitor}: Tacrolimus (preferred) or cyclosporine. Monitor drug levels (trough levels).
        \item \textbf{Antimetabolite}: Mycophenolate mofetil or azathioprine.
        \item \textbf{Corticosteroid}: Prednisone (often tapered over months).
    \end{itemize}
    \item \textbf{Monitoring}: Regular drug level monitoring, complete blood count, renal function, liver function.
    \item \textbf{Side Effects}: Infection risk, malignancy (skin cancer, lymphoma), diabetes (tacrolimus), hypertension, nephrotoxicity, hyperlipidemia.
\end{enumerate}
\end{stepbox}

\subsection{2.7 NGN Practice Questions: GI Disorders (Week 7)}
\begin{ngncase}{Acute Pancreatitis}
\textbf{Scenario}: A 45-year-old male presents to the emergency department with severe epigastric pain radiating to the back, nausea, and vomiting. He reports heavy alcohol use over the past week. Vital signs: T 38.2°C, HR 112, BP 100/60, RR 24, SpO₂ 94% on room air. Abdominal exam reveals tenderness in the epigastric area with guarding. Laboratory results show: WBC 18,000/mm³, serum amylase 450 U/L, serum lipase 680 U/L, glucose 180 mg/dL, calcium 7.8 mg/dL.

\textbf{Question 1}: Based on the clinical presentation and laboratory findings, which condition does the nurse suspect?
\begin{enumerate}[label=\Alph*.]
    \item Cholecystitis
    \item Peptic ulcer disease
    \item Acute pancreatitis
    \item Bowel obstruction
\end{enumerate}

\textbf{Question 2}: The nurse notes the patient's Ranson's criteria at admission. Which of the following findings are part of Ranson's criteria? (Select all that apply)
\begin{enumerate}[label=\Alph*.]
    \item Age > 55 years
    \item WBC > 16,000/mm³
    \item Serum glucose > 200 mg/dL
    \item Serum calcium < 8 mg/dL
    \item Hematocrit decrease > 10\%
\end{enumerate}

\textbf{Question 3}: The physician orders aggressive fluid resuscitation with lactated Ringer's solution. What is the primary rationale for this intervention?
\begin{enumerate}[label=\Alph*.]
    \item To correct hyperglycemia
    \item To prevent third-spacing and hypovolemia
    \item To reduce serum amylase levels
    \item To alleviate pain
\end{enumerate}

\textbf{Question 4}: The patient develops severe respiratory distress. ABG results: pH 7.32, PaCO₂ 30 mmHg, HCO₃⁻ 18 mEq/L. The nurse interprets these results as:
\begin{enumerate}[label=\Alph*.]
    \item Metabolic acidosis with respiratory compensation
    \item Respiratory acidosis with metabolic compensation
    \item Metabolic alkalosis with respiratory compensation
    \item Respiratory alkalosis with metabolic compensation
\end{enumerate}

\textbf{Answers and Rationale:}
\begin{enumerate}
    \item \textbf{C. Acute pancreatitis}. Rationale: Epigastric pain radiating to back, elevated amylase/lipase, history of alcohol use, and leukocytosis are classic for acute pancreatitis.
    \item \textbf{A, B, C, D, E (All)}. Rationale: All five options are part of Ranson's criteria (which has 11 criteria total). Admission criteria include: age >55 (A), WBC >16,000 (B), glucose >200 (C), LDH >350, and AST >250. Forty-eight-hour criteria include calcium <8 mg/dL (D) and hematocrit decrease >10\% (E). The remaining 6 criteria (BUN rise, base deficit, PaO₂, fluid sequestration, LDH, AST) are not listed among the options but are also part of the full scoring tool.
    \item \textbf{B. To prevent third-spacing and hypovolemia}. Rationale: Aggressive fluid resuscitation is critical to maintain intravascular volume and prevent complications like acute kidney injury.
    \item \textbf{A. Metabolic acidosis with respiratory compensation}. Rationale: Low pH with low HCO₃⁻ indicates metabolic acidosis. The low PaCO₂ shows respiratory compensation (hyperventilation).
\end{enumerate}
\end{ngncase}

\textbf{Admission Labs:}
\begin{itemize}
    \item Amylase: 850 U/L (normal: 25-125 U/L)
    \item Lipase: 1,200 U/L (normal: 0-160 U/L)
    \item WBC: 18,000/mm³
    \item Glucose: 245 mg/dL
    \item Calcium: 7.2 mg/dL
    \item BUN: 32 mg/dL
    \item Hematocrit: 52\%
    \item LDH: 400 U/L
\end{itemize}

\textbf{Questions:}

\textbf{1.} Calculate the Ranson's criteria at admission for this patient. List all applicable criteria.

\textbf{2.} The physician orders aggressive fluid resuscitation with lactated Ringer's at 250 mL/hr. Explain why LR is preferred over normal saline in acute pancreatitis.

\textbf{3.} After 5 days of NPO status, the physician orders initiation of TPN. The TPN formula contains 25\% dextrose, 4.25\% amino acids, and 2\% lipids. What is the total caloric content per liter (approximately)?

\textbf{4.} The patient develops respiratory distress on day 3. ABG shows: pH 7.31, PaCO₂ 28 mmHg, HCO₃⁻ 14 mEq/L, PaO₂ 62 mmHG. Interpret these results and explain their significance in pancreatitis.

\textbf{5.} The nurse notes the patient's central line insertion site is red, warm, and tender with purulent drainage. The patient has a fever of 38.8°C. What complication does this suggest, and what are the priority nursing actions?

\textbf{Solutions:}

\textbf{1. Ranson's Criteria at Admission:}
\begin{itemize}
    \item Age > 55 years: NO (patient is 52)
    \item WBC > 16,000/mm³: YES (18,000)
    \item Glucose > 200 mg/dL: YES (245)
    \item LDH > 350 U/L: YES (400)
    \item AST > 250 U/L: Not provided
\end{itemize}
\textbf{Score: At least 3 criteria positive (indicating severe pancreatitis with higher mortality risk)}

\textbf{2.} Lactated Ringer's is preferred because it is more physiologically balanced and has been shown to reduce inflammatory markers (SIRS) compared to normal saline. NS can cause hyperchloremic metabolic acidosis, which can worsen the patient's already compromised acid-base status.

\textbf{3.} Approximate calories per liter:
\begin{itemize}
    \item Dextrose 25\%: 250 g × 3.4 kcal/g = 850 kcal
    \item Amino acids 4.25\%: 42.5 g × 4 kcal/g = 170 kcal
    \item Lipids 2\%: 20 g × 9 kcal/g = 180 kcal
    \item \textbf{Total: approximately 1,200 kcal/L}
\end{itemize}

\textbf{4.} The ABG shows \textbf{compensated metabolic acidosis with hypoxemia}. The low pH with low HCO₃⁻ indicates metabolic acidosis (from tissue hypoperfusion, lactic acidosis). The low PaCO₂ shows respiratory compensation. The low PaO₂ (62 mmHg) indicates hypoxemia, which may suggest ARDS—a serious complication of severe pancreatitis. This requires immediate intervention including oxygen therapy and possible mechanical ventilation.

\textbf{5.} This suggests a \textbf{catheter-related bloodstream infection (CRBSI)}. Priority actions:
\begin{enumerate}
    \item Obtain blood cultures (peripheral and from the catheter) BEFORE starting antibiotics
    \item Notify the physician immediately
    \item Prepare to discontinue and culture the catheter tip
    \item Initiate prescribed antibiotic therapy
    \item Monitor vital signs and hemodynamic status
    \item Document and report findings
\end{enumerate}

\section{Summary: GI Disorders (Week 7)}
\begin{itemize}
    \item \textbf{Acute Pancreatitis}: Remember "I GET SMASHED" for etiology. Key labs are amylase and lipase. Treatment is supportive: NPO, IV fluids, pain management, and monitoring for complications.
    \item \textbf{Ranson's Criteria}: Used to predict severity and mortality in acute pancreatitis. Higher scores = worse prognosis.
    \item \textbf{Chronic Pancreatitis}: Results from repeated episodes of acute pancreatitis. Hallmark is chronic pain and malabsorption (steatorrhea). Patients need pancreatic enzyme replacement.
    \item \textbf{Pancreatic Cancer}: Often diagnosed late due to vague symptoms. Courvoisier's sign (painless palpable gallbladder) is a classic finding. Whipple procedure is the surgical treatment.
    \item \textbf{Cholecystitis}: Inflammation of the gallbladder, usually from gallstones. Murphy's sign is positive. Cholecystectomy is definitive treatment.
    \item \textbf{TPN}: Used when enteral nutrition is not possible. Requires central venous access. Monitor blood glucose, electrolytes, and for complications including infection and refeeding syndrome.
    \item \textbf{Key Nursing Priorities}: Pain management, fluid balance, nutritional support, monitoring for complications, patient education.
\end{itemize}

\section{Acute Kidney Injury (AKI)}

\begin{definition}
\textbf{Acute Kidney Injury (AKI)} is a sudden decline in kidney function occurring over hours to days, resulting in the accumulation of waste products (urea, creatinine) and dysregulation of fluid, electrolyte, and acid-base balance. It is defined as any of the following: increase in serum creatinine ≥0.3 mg/dL within 48 hours; increase in serum creatinine ≥1.5 times baseline within 7 days; or urine volume <0.5 mL/kg/hr for 6 hours.
\end{definition}

\begin{contextbox}
\textbf{Clinical Significance:} AKI affects approximately 13.3 million people worldwide each year and is associated with high mortality rates (up to 50\% in hospitalized patients). Early recognition and treatment can prevent progression to chronic kidney disease and reduce mortality. AKI increases hospital length of stay, healthcare costs, and risk of developing chronic kidney disease.
\end{contextbox}

\subsection{Phases of AKI}
\begin{stepbox}
\textbf{Four Phases of AKI:}
\begin{enumerate}
    \item \textbf{Initiation Phase}: Begins
\item \textbf{Initiation Phase}: Begins with the precipitating event (hypotension, nephrotoxin exposure). Lasts hours to days. GFR begins to decline.
    \item \textbf{Oliguric/Anuric Phase}: Urine output drops below 400 mL/day (oliguria) or 100 mL/day (anuria). Lasts 1-2 weeks. BUN and creatinine rise rapidly. Electrolyte imbalances occur.
    \item \textbf{Diuretic Phase}: Urine output increases (may exceed 3-5 L/day). Gradual recovery of tubular function. Still at risk for dehydration and electrolyte imbalances.
    \item \textbf{Recovery Phase}: GFR gradually returns to baseline over 6-12 months. Some patients may have residual renal impairment.
\end{enumerate}
\end{stepbox}

\begin{criticalbox}
\textbf{High-Risk Periods:} The oliguric phase carries the highest risk for complications including hyperkalemia, fluid overload, metabolic acidosis, and uremia. The diuretic phase requires careful monitoring to prevent dehydration and electrolyte losses (hypokalemia, hyponatremia).
\end{criticalbox}

\subsection{Clinical Manifestations}
\begin{stepbox}
\textbf{Signs and Symptoms of AKI by System:}
\begin{enumerate}
    \item \textbf{Urinary Changes}:
    \begin{itemize}
        \item Oliguria (<400 mL/day) or anuria (<100 mL/day)
        \item Changes in urine color (dark, tea-colored)
        \item Foamy urine (proteinuria)
    \end{itemize}
    \item \textbf{Fluid Volume Overload}:
    \begin{itemize}
        \item Peripheral edema
        \item Pulmonary edema (crackles, dyspnea, orthopnea)
        \item Jugular venous distension (JVD)
        \item Hypertension
        \item Weight gain
    \end{itemize}
    \item \textbf{Electrolyte Imbalances}:
    \begin{itemize}
        \item Hyperkalemia (peaked T waves, widened QRS, cardiac arrhythmias)
        \item Hyperphosphatemia
        \item Hypocalcemia (muscle cramps, tetany, Chvostek's/Trousseau's signs)
        \item Hyponatremia or hypernatremia
        \item Hypomagnesemia
    \end{itemize}
    \item \textbf{Uremic Symptoms}:
    \begin{itemize}
        \item Nausea, vomiting, anorexia
        \item Fatigue, weakness
        \item Confusion, altered mental status
        \item Pruritus
        \item Uremic frost (crystallized urea on skin in severe cases)
        \item Pericarditis (friction rub)
        \item Platelet dysfunction (bleeding)
    \end{itemize}
    \item \textbf{Metabolic Acidosis}:
    \begin{itemize}
        \item Kussmaul respirations (deep, rapid breathing)
        \item Decreased serum bicarbonate
        \item Decreased pH
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{clinicalpearl}
\textbf{The "3 C's" of Hyperkalemia ECG Changes}: As potassium rises, watch for progression: \textbf{C}hronologically - \textbf{C}hanges appear in order: Peaked \textbf{T} waves → Prolonged PR → Widened QRS → Sine wave → Cardiac arrest. Recognize these changes early to prevent fatal arrhythmias.
\end{clinicalpearl}

\subsection{Diagnostic Studies}
\begin{stepbox}
\textbf{Diagnostic Workup for AKI:}
\begin{enumerate}
    \item \textbf{Serum Creatinine and BUN}: Rising creatinine is the hallmark; BUN:Cr ratio helps differentiate type
    \item \textbf{BUN/Creatinine Ratio}:
    \begin{itemize}
        \item >20:1 suggests prerenal cause
        \item <15:1 suggests intrinsic/intrarenal cause
    \end{itemize}
    \item \textbf{Urinalysis}:
    \begin{itemize}
        \item Muddy brown granular casts = ATN (intrinsic)
        \item WBC casts = interstitial nephritis
        \item RBC casts = glomerulonephritis
        \item Specific gravity: High in prerenal, fixed at 1.010 in intrinsic
    \end{itemize}
    \item \textbf{Urine Sodium}:
    \begin{itemize}
        \item <20 mEq/L = prerenal (kidneys retaining sodium)
        \item >40 mEq/L = intrinsic (damaged tubules cannot reabsorb)
    \end{itemize}
    \item \textbf{Fractional Excretion of Sodium (FENa)}:
    \begin{itemize}
        \item <1\% = prerenal
        \item >2\% = intrinsic renal damage
    \end{itemize}
    \item \textbf{Renal Ultrasound}: Rule out obstruction (hydronephrosis), assess kidney size
    \item \textbf{Serum Electrolytes}: Potassium, sodium, calcium, phosphorus, magnesium
    \item \textbf{ABG}: Assess for metabolic acidosis (low pH, low HCO₃⁻)
    \item \textbf{CBC}: Check for anemia, infection
    \item \textbf{Serum Albumin}: Assess nutritional status
\end{enumerate}
\end{stepbox}

\begin{warningbox}
\textbf{FENa Caveat}: FENa may be unreliable in patients taking diuretics (falsely elevated) or with pre-existing CKD, heart failure, or liver disease. In these cases, use FE urea (<35\% suggests prerenal) or clinical judgment.
\end{warningbox}

\subsection{Treatment and Nursing Management}
\begin{stepbox}
\textbf{Management of AKI:}
\begin{enumerate}
    \item \textbf{Treat Underlying Cause}:
    \begin{itemize}
        \item Prerenal: Volume resuscitation with isotonic fluids (NS or LR)
        \item Intrinsic: Remove nephrotoxins, treat infection
        \item Postrenal: Relieve obstruction (catheter, stent, surgery)
    \end{itemize}
    \item \textbf{Fluid Management}:
    \begin{itemize}
        \item Strict I\&O monitoring
        \item Daily weights (same time, same scale, same clothing)
        \item Monitor for fluid overload
        \item Adjust IV fluids based on urine output
    \end{itemize}
    \item \textbf{Electrolyte Management}:
    \begin{itemize}
        \item \textbf{Hyperkalemia}: Cardiac monitoring, calcium gluconate (stabilize membrane), insulin + D50 (shift K⁺ into cells), sodium bicarbonate, kayexalate, dialysis
        \item \textbf{Hyperphosphatemia}: Phosphate binders (calcium acetate, sevelamer)
        \item \textbf{Hypocalcemia}: IV or oral calcium (after phosphate is controlled)
        \item \textbf{Metabolic Acidosis}: Sodium bicarbonate if pH <7.2 or HCO₃⁻ <15
    \end{itemize}
    \item \textbf{Medication Management}:
    \begin{itemize}
        \item Hold or adjust nephrotoxic medications (NSAIDs, aminoglycosides, contrast dye, ACE inhibitors, ARBs)
        \item Adjust renally cleared medications based on GFR
    \end{itemize}
    \item \textbf{Nutritional Support}:
    \begin{itemize}
        \item Protein restriction (0.6-0.8 g/kg/day) if not on dialysis
        \item Adequate protein (1.0-1.5 g/kg/day) if on dialysis
        \item Potassium restriction (<2 g/day)
        \item Phosphorus restriction
        \item Sodium and fluid restriction
    \end{itemize}
    \item \textbf{Dialysis Indications} (A-E-I-O-U):
    \begin{itemize}
        \item \textbf{A}cidosis refractory to treatment
        \item \textbf{E}lectrolyte imbalance (severe hyperkalemia)
        \item \textbf{I}ntoxication (toxic ingestions)
        \item \textbf{O}verload (fluid overload unresponsive to diuretics)
        \item \textbf{U}remic symptoms (pericarditis, encephalopathy, bleeding)
    \end{itemize}
    \item \textbf{Infection Prevention}:
    \begin{itemize}
        \item Hand hygiene
        \item Aseptic technique for invasive procedures
        \item Monitor for signs of infection (immunosuppressed)
    \end{itemize}
    \item \textbf{Skin Care}: Assess for edema, uremic frost; reposition frequently
\end{enumerate}
\end{stepbox}

\section{Chronic Pancreatitis}
\begin{definition}
\textbf{Chronic Pancreatitis} is a progressive inflammatory disease of the pancreas characterized by irreversible morphological changes, leading to permanent loss of endocrine and exocrine function. Unlike acute pancreatitis, it causes permanent damage and fibrosis.
\end{definition}

\subsection{Etiology and Pathophysiology}
\begin{stepbox}
\textbf{Key Etiological Factors:}
\begin{enumerate}
    \item \textbf{Chronic Alcohol Use} (70-80\% of cases): Toxic metabolites cause direct acinar cell injury and protein plug formation.
    \item \textbf{Idiopathic}: No identifiable cause; may be genetic or autoimmune.
    \item \textbf{Genetic Mutations}: PRSS1, SPINK1, CFTR genes.
    \item \textbf{Autoimmune}: Type 1 (IgG4-related) and Type 2.
    \item \textbf{Obstructive}: Due to ductal obstruction (tumor, stricture).
\end{enumerate}
\textbf{Pathophysiology:}
\begin{itemize}
    \item Repeated injury → inflammation → fibrosis → calcification
    \item Destruction of acinar and islet cells → exocrine and endocrine insufficiency
    \item Ductal obstruction → pain, pseudocyst formation
\end{itemize}
\end{stepbox}

\subsection{Clinical Manifestations}
\begin{theorem}
The hallmark symptoms are \textbf{abdominal pain} and \textbf{pancreatic insufficiency} (malabsorption, diabetes).
\end{theorem}

\begin{longtable}{|p{4cm}|p{5cm}|p{5cm}|}
\hline
\textbf{Symptom} & \textbf{Characteristics} & \textbf{Underlying Cause} \\
\hline
\endfirsthead
\hline
\textbf{Symptom} & \textbf{Characteristics} & \textbf{Underlying Cause} \\

\endhead
\hline
\endfoot
Abdominal Pain & Epigastric, constant, radiates to back; worsened by eating, alcohol & Ductal hypertension, inflammation, neural damage \\
\hline

\subsection{Diagnostic Studies}
\begin{stepbox}
\textbf{Diagnostic Approach:}
\begin{enumerate}
    \item \textbf{Imaging}:
    \begin{itemize}
        \item CT abdomen: Calcifications, ductal dilation, pseudocysts
        \item MRCP: Detailed ductal anatomy
        \item EUS: Early changes, small duct disease
    \end{itemize}
    \item \textbf{Functional Tests}:
    \begin{itemize}
        \item Secretin-stimulated MRCP: Assess ductal compliance
        \item Fecal elastase: < 200 µg/g indicates insufficiency
        \item 72-hour fecal fat: Gold standard for steatorrhea
    \end{itemize}
    \item \textbf{Lab Studies}: Amylase/lipase often normal (burned-out pancreas)
\end{enumerate}
\end{stepbox}

\subsection{Treatment and Nursing Management}
\begin{stepbox}
\textbf{Management Goals: Pain control, manage insufficiency, prevent complications.}
\begin{enumerate}
    \item \textbf{Pain Management}:
    \begin{itemize}
        \item Non-opioid analgesics first (acetaminophen, NSAIDs)
        \item Low-potency opioids if needed (avoid meperidine)
        \item Celiac plexus block for refractory pain
        \item Pancreatic enzymes (may reduce pain by negative feedback)
    \end{itemize}
    \item \textbf{Exocrine Insufficiency}:
    \begin{itemize}
        \item Pancreatic enzyme replacement therapy (PERT): Take with meals
        \item Fat-soluble vitamin supplementation (A, D, E, K)
        \item Low-fat diet
    \end{itemize}
    \item \textbf{Endocrine Insufficiency}:
    \begin{itemize}
        \item Insulin therapy (often required)
        \item Monitor for hypoglycemia (brittle diabetes)
    \end{itemize}
    \item \textbf{Surgical Options}: For pain refractory to medical therapy (drainage procedures, resection)
\end{enumerate}
\end{stepbox}

\begin{warningbox}
\textbf{Critical Nursing Consideration:} Patients with chronic pancreatitis are at high risk for \textbf{malnutrition} and \textbf{fat-soluble vitamin deficiencies}. Monitor weight, albumin, and vitamin levels regularly.
\end{warningbox}

\section{Pancreatic Cancer}
\begin{definition}
\textbf{Pancreatic Cancer} (most commonly pancreatic ductal adenocarcinoma) is a malignant neoplasm arising from the exocrine pancreas. It has a very poor prognosis due to late presentation and aggressive biology.
\end{definition}

\subsection{Risk Factors}
\begin{theorem}
The most significant modifiable risk factor is \textbf{cigarette smoking} (2-3x increased risk). Other factors include chronic pancreatitis, diabetes, obesity, and family history.
\end{theorem}

\begin{longtable}{|p{4cm}|p{5cm}|p{5cm}|}
\hline
\textbf{Risk Factor} & \textbf{Mechanism} & \textbf{Relative Risk} \\
\hline
\endfirsthead
\hline
\textbf{Risk Factor} & \textbf{Mechanism} & \textbf{Relative Risk} \\
\hline
\endhead
\hline
\endfoot
Cigarette Smoking & Carcinogen exposure, nitrosamines & 2-3x \\
\hline
Chronic Pancreatitis & Persistent inflammation, fibrosis & 10-20x \\
\hline
Diabetes Mellitus (new-onset) & Insulin resistance, hyperinsulinemia & 2x \\
\hline
Obesity & Adipokines, inflammation & 1.5x \\
\hline
Family History & Genetic predisposition (BRCA2, PALB2) & 2-3x (higher with multiple relatives) \\
\hline
\end{longtable}

\subsection{Clinical Manifestations}
\begin{criticalbox}
\textbf{Red Flag Symptoms:} Painless jaundice (head tumors), new-onset diabetes in elderly, unexplained weight loss, and deep visceral pain radiating to the back.
\end{criticalbox}

\begin{stepbox}
\textbf{Symptoms by Tumor Location:}
\begin{enumerate}
    \item \textbf{Head of Pancreas} (60-70\%):
    \begin{itemize}
        \item Early: Obstructive jaundice (Courvoisier's sign: palpable non-tender gallbladder)
        \item Early: Steatorrhea (bile duct obstruction)
    \end{itemize}
    \item \textbf{Body/Tail} (30-40\%):
    \begin{itemize}
        \item Late presentation: Pain, weight loss, splenic vein thrombosis
        \item Often metastatic at diagnosis
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsection{Diagnostic Studies}
\begin{stepbox}
\textbf{Diagnostic Workup for Pancreatic Cancer:}
\begin{enumerate}
    \item \textbf{Imaging Studies}:
    \begin{itemize}
        \item \textbf{CT Abdomen with contrast} (gold standard): Mass, vascular involvement, metastasis
        \item \textbf{Endoscopic Ultrasound (EUS)}: Most sensitive for small tumors, allows biopsy
        \item \textbf{MRI/MRCP}: Evaluates ductal anatomy, liver metastases
        \item \textbf{ERCP}: Therapeutic (stent placement) and diagnostic (brushings for cytology)
    \end{itemize}
    \item \textbf{Laboratory Studies}:
    \begin{itemize}
        \item \textbf{Tumor Markers}: CA 19-9 (elevated in 80\%, not specific), CEA
        \item \textbf{Liver Function Tests}: Elevated bilirubin, ALP, GGT (obstructive pattern)
        \item \textbf{Pancreatic Enzymes}: May be normal or elevated
    \end{itemize}
    \item \textbf{Biopsy}: EUS-guided fine needle aspiration (FNA) for tissue diagnosis
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{High-Yield Tip:} CA 19-9 is useful for monitoring treatment response and detecting recurrence, but it is NOT recommended for screening (low specificity, can be elevated in biliary obstruction, pancreatitis, and other malignancies).
\end{insightbox}

\subsection{Treatment}
\begin{stepbox}
\textbf{Treatment Approach Based on Resectability:}
\begin{enumerate}
    \item \textbf{Surgical Resection} (only curative option, 15-20\% eligible):
    \begin{itemize}
        \item \textbf{Whipple Procedure} (pancreaticoduodenectomy): For head tumors
        \item \textbf{Distal Pancreatectomy}: For body/tail tumors
        \item \textbf{Total Pancreatectomy}: For diffuse disease
    \end{itemize}
    \item \textbf{Adjuvant Chemotherapy}: Gemcitabine-based regimens (improves survival)
    \item \textbf{Neoadjuvant Therapy}: For borderline resectable tumors
    \item \textbf{Palliative Treatment} (for unresectable disease):
    \begin{itemize}
        \item Chemotherapy (FOLFIRINOX, gemcitabine/nab-paclitaxel)
        \item Biliary stenting for jaundice relief
        \item Pain management (celiac plexus block)
        \item Nutritional support
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsection{Nursing Management}
\begin{stepbox}
\textbf{Nursing Care Priorities for Pancreatic Cancer:}
\begin{enumerate}
    \item \textbf{Pain Management}:
    \begin{itemize}
        \item Assess pain characteristics (worse after eating, lying supine)
        \item Administer analgesics around the clock
        \item Coordinate celiac plexus block for refractory pain
    \end{itemize}
    \item \textbf{Nutritional Support}:
    \begin{itemize}
        \item Pancreatic enzyme replacement with meals
        \item Small, frequent, low-fat meals
        \item Monitor weight and nutritional status
        \item Consider enteral nutrition if oral intake inadequate
    \end{itemize}
    \item \textbf{Manage Complications}:
    \begin{itemize}
        \item Obstructive jaundice: Pruritus management, vitamin K administration
        \item Diabetes: Blood glucose monitoring, insulin administration
        \item Depression/anxiety: Screen and refer for support
    \end{itemize}
    \item \textbf{Psychosocial Support}:
    \begin{itemize}
        \item Address grief and coping
        \item Facilitate advance care planning discussions
        \item Refer to palliative care early
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{clinicalpearl}
\textbf{Clinical Pearl:} Pancreatic cancer has a 5-year survival rate of only 10-15\%. Early palliative care involvement improves quality of life and may even extend survival. Always address goals of care early in the disease trajectory.
\end{clinicalpearl}

\section{Cholecystitis}
\begin{definition}
\textbf{Cholecystitis} is inflammation of the gallbladder, most commonly caused by obstruction of the cystic duct by gallstones (calculous cholecystitis). It can also occur without stones (acalculous cholecystitis).
\end{definition}

\subsection{Types of Cholecystitis}
\begin{longtable}{|p{3.5cm}|p{5cm}|p{5.5cm}|}
\hline
\textbf{Type} & \textbf{Characteristics} & \textbf{Common Settings} \\
\hline
\endfirsthead
\hline
\textbf{Type} & \textbf{Characteristics} & \textbf{Common Settings} \\
\hline
\endhead
\hline
\endfoot
\textbf{Acute Calculous} (90\%) & Gallstone obstructs cystic duct → inflammation & Obese, female, forty, fertile (4 F's) \\
\hline
\textbf{Acalculous} (10\%) & Inflammation without stones, often due to ischemia & Critically ill, major surgery, trauma, burns, TPN \\
\hline
\textbf{Chronic} & Repeated episodes of mild inflammation & Recurrent biliary colic, fibrosis \\
\hline
\end{longtable}

\subsection{Risk Factors for Gallstones}
\begin{theorem}
\textbf{The "5 F's" of Gallstone Risk:}
\begin{itemize}
    \item \textbf{F}at (obesity)
    \item \textbf{F}emale (estrogen effect)
    \item \textbf{F}orty (age > 40)
    \item \textbf{F}ertile (multiparity, pregnancy)
    \item \textbf{F}air (Caucasian, Native American)
\end{itemize}
\end{theorem}

\begin{warningbox}
\textbf{Additional Risk Factors:}
\begin{itemize}
    \item Rapid weight loss or bariatric surgery
    \item Oral contraceptives or hormone replacement therapy
    \item Family history of gallstones
    \item Diabetes mellitus
    \item Cirrhosis
    \item Crohn's disease (ileal disease)
\end{itemize}
\end{warningbox}

\subsection{Clinical Manifestations}
\begin{stepbox}
\textbf{Classic Presentation of Acute Cholecystitis:}
\begin{enumerate}
    \item \textbf{Pain}:
    \begin{itemize}
        \item Location: Right upper quadrant (RUQ) or epigastrium
        \item Radiation: Right shoulder or scapula (referred pain via phrenic nerve)
        \item Character: Steady, severe, lasts > 6 hours
        \item Triggers: Fatty meals, occurs 1-2 hours after eating
    \end{itemize}
    \item \textbf{Associated Symptoms}:
    \begin{itemize}
        \item Nausea and vomiting
        \item Anorexia
        \item Low-grade fever
        \item Malaise
    \end{itemize}
    \item \textbf{Physical Examination Findings}:
    \begin{itemize}
        \item \textbf{Murphy's Sign}: Inspiratory arrest during deep palpation of RUQ (positive in acute cholecystitis)
        \item RUQ tenderness and guarding
        \item Palpable gallbladder (Courvoisier's sign if painless—suggests malignancy)
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{analogy}
Think of the gallbladder as a balloon attached to a pipe (cystic duct). When a stone blocks the pipe, bile can't drain, the balloon stretches, and inflammation develops—similar to a clogged drain causing a sink to back up and overflow.
\end{analogy}

\subsection{Treatment}
\begin{stepbox}
\textbf{Treatment Algorithm for Cholecystitis:}
\begin{enumerate}
    \item \textbf{Initial Management (NPO)}:
    \begin{itemize}
        \item NPO status
        \item IV fluid resuscitation
        \item IV antibiotics (ceftriaxone + metronidazole or piperacillin-tazobactam)
        \item Pain management (NSAIDs, opioids)
        \item Antiemetics
    \end{itemize}
    \item \textbf{Definitive Treatment}:
    \begin{itemize}
        \item \textbf{Laparoscopic Cholecystectomy}: Gold standard, performed within 72 hours of symptom onset
        \item \textbf{Open Cholecystectomy}: If laparoscopic not feasible
        \item \textbf{Percutaneous Cholecystostomy}: For patients too unstable for surgery (acalculous cholecystitis)
    \end{itemize}
    \item \textbf{For Common Bile Duct Stones}:
    \begin{itemize}
        \item ERCP with sphincterotomy and stone extraction
        \item May be performed before or after cholecystectomy
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsection{Nursing Management}
\begin{stepbox}
\textbf{Pre-Operative Nursing Care:}
\begin{enumerate}
    \item NPO status, maintain IV access
    \item Administer prescribed antibiotics and analgesics
    \item Monitor vital signs, especially temperature
    \item Assess for worsening symptoms (perforation, empyema)
    \item Patient education about surgical procedure
\end{enumerate}

\textbf{Post-Operative Nursing Care:}
\begin{enumerate}
    \item \textbf{Monitor for Complications}:
    \begin{itemize}
        \item Bile leak (abdominal pain, distension, fever)
        \item Retained common bile duct stones
        \item Wound infection
        \item Hemorrhage
    \end{itemize}
    \item \textbf{Pain Management}: PCA or scheduled analgesics
    \item \textbf{Dietary Progression}: Clear liquids → low-fat diet → regular diet as tolerated
    \item \textbf{Activity}: Ambulate as soon as possible
    \item \textbf{Patient Education}:
    \begin{itemize}
        \item Temporary dietary fat restriction (4-6 weeks)
        \item Signs of complications to report
        \item Incision care and activity restrictions
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{criticalbox}
\textbf{Critical Alert:} Monitor for signs of \textbf{cholangitis} (Charcot's triad: RUQ pain, fever, jaundice; Reynolds' pentad: adds hypotension and altered mental status). This is a life-threatening emergency requiring immediate intervention!
\end{criticalbox}

\section{Total Parenteral Nutrition (TPN)}
\begin{definition}
\textbf{Total Parenteral Nutrition (TPN)} is a concentrated hypertonic solution containing all essential nutrients (carbohydrates, proteins, fats, vitamins, minerals, and electrolytes) administered intravenously to patients who cannot obtain adequate nutrition through the GI tract.
\end{definition}

\subsection{Indications for TPN}
\begin{theorem}
\textbf{TPN is indicated when the GI tract is nonfunctional, inaccessible, or cannot absorb adequate nutrients for 7-14 days or longer.}
\end{theorem}

\begin{stepbox}
\textbf{Common Indications:}
\begin{enumerate}
    \item \textbf{GI Tract Dysfunction}:
    \begin{itemize}
        \item Inflammatory bowel disease (severe flare)
        \item Short bowel syndrome
        \item Bowel obstruction
        \item Severe pancreatitis (when enteral feeding not tolerated)
        \item High-output enterocutaneous fistula
    \end{itemize}
    \item \textbf{Perioperative}:
    \begin{itemize}
        \item Prolonged post-operative ileus
        \item Complicated GI surgeries
    \end{itemize}
    \item \textbf{Critical Illness}:
    \begin{itemize}
        \item Severe malnutrition with inability to use GI tract
        \item Prolonged mechanical ventilation
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsection{TPN Components}
\begin{longtable}{|p{3cm}|p{4cm}|p{4cm}|p{3cm}|}
\hline
\textbf{Component} & \textbf{Sources} & \textbf{Purpose} & \textbf{Monitoring} \\
\hline
\endfirsthead
\hline
\textbf{Component} & \textbf{Sources} & \textbf{Purpose} & \textbf{Monitoring} \\
\hline
\endhead
\hline
\endfoot
\textbf{Dextrose} (50-70\%) & D50W, D70W & Primary energy source (3.4 kcal/g) & Blood glucose \\
\hline
\textbf{Amino Acids} (5-15\%) & Aminosyn,
\textbf{Travasol, Prosol} & Tissue synthesis, nitrogen balance & BUN, serum protein \\
\hline
\textbf{Lipids} (10-30\%) & Intralipid, Liposyn & Essential fatty acids, energy (9 kcal/g) & Triglycerides, cholesterol \\
\hline
\textbf{Electrolytes} & Na, K, Ca, Mg, PO₄ & Maintain homeostasis & Serum levels \\
\hline
\textbf{Vitamins} & MVI-12, B-complex, Vitamin K & Metabolic functions & Levels (especially thiamine) \\
\hline
\textbf{Trace Elements} & Zinc, copper, manganese, chromium & Enzyme cofactors & Levels (monitor with renal dysfunction) \\
\hline
\end{longtable}

\begin{insightbox}
\textbf{Caloric Content:}
\begin{itemize}
    \item Dextrose: 3.4 kcal/g
    \item Amino acids: 4 kcal/g
    \item Lipids: 9 kcal/g (most concentrated energy source)
\end{itemize}
\end{insightbox}

\subsection{TPN Administration}
\begin{stepbox}
\textbf{TPN Administration Guidelines:}
\begin{enumerate}
    \item \textbf{Access}:
    \begin{itemize}
        \item Central venous catheter (PICC, subclavian, internal jugular)
        \item Dedicated lumen for TPN only
        \item NEVER use peripheral vein (hypertonic solution causes phlebitis)
    \end{itemize}
    \item \textbf{Infusion Rate}:
    \begin{itemize}
        \item Start at 50 mL/hr, increase by 25 mL/hr every 12-24 hours
        \item Goal rate: 80-125 mL/hr (varies by prescription)
        \item Use infusion pump with dedicated channel
    \end{itemize}
    \item \textbf{Timing}:
    \begin{itemize}
        \item Continuous infusion over 24 hours (most common)
        \item Cyclic TPN: 12-18 hours overnight (allows mobility)
        \item Never abruptly stop TPN (risk of hypoglycemia)
    \end{itemize}
    \item \textbf{Compatibility}:
    \begin{itemize}
        \item No other IV medications through same line
        \item Lipids may be infused separately (Y-site) or in 3-in-1 solution
        \item Check compatibility charts for additives
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsection{Monitoring During TPN Therapy}
\begin{stepbox}
\textbf{Essential Monitoring Parameters:}
\begin{enumerate}
    \item \textbf{Daily}:
    \begin{itemize}
        \item Intake and output (strict)
        \item Weight (same time each day)
        \item Vital signs (temperature q4h for infection)
        \item Blood glucose (q6h initially, then daily)
        \item Electrolytes (Na, K, Cl, CO₂, BUN, Cr, Mg, Phos, Ca)
    \end{itemize}
    \item \textbf{Weekly}:
    \begin{itemize}
        \item Complete blood count
        \item Liver function tests (AST, ALT, bilirubin, alkaline phosphatase)
        \item Serum protein (albumin, prealbumin)
        \item Triglycerides (if on lipids)
        \item Coagulation studies (PT/PTT)
    \end{itemize}
    \item \textbf{As Needed}:
    \begin{itemize}
        \item Zinc, copper, manganese levels
        \item Ammonia level (if amino acid intolerance)
        \item Chest X-ray (line placement)
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsection{Complications of TPN}
\begin{criticalbox}
\textbf{Life-Threatening Complications:}
\begin{itemize}
    \item \textbf{Sepsis/Catheter-related bloodstream infection}: Most common serious complication
    \item \textbf{Hyperglycemia}: Osmotic diuresis, hyperosmolar nonketotic coma
    \item \textbf{Refeeding Syndrome}: Hypophosphatemia, hypokalemia, hypomagnesemia
    \item \textbf{Air Embolism}: If catheter disconnected or cracked
\end{itemize}
\end{criticalbox}

\begin{warningbox}
\textbf{Common Complications:}
\begin{itemize}
    \item \textbf{Metabolic}: Hyperglycemia, electrolyte imbalances, liver dysfunction
    \item \textbf{Mechanical}: Pneumothorax, catheter occlusion, thrombosis
    \item \textbf{Infectious}: Exit site infection, tunnel infection, sepsis
    \item \textbf{GI}: Atrophy of gut mucosa, cholestasis, gallbladder sludge
\end{itemize}
\end{warningbox}

\subsection{Discontinuing TPN}
\begin{stepbox}
\textbf{TPN Weaning Process:}
\begin{enumerate}
    \item \textbf{Assess Readiness}:
    \begin{itemize}
        \item GI tract functional (bowel sounds, no nausea/vomiting)
        \item Able to tolerate enteral or oral nutrition
        \item Nutritional goals being met by alternative route
    \end{itemize}
    \item \textbf{Wean Gradually}:
    \begin{itemize}
        \item Decrease rate by 50 mL/hr every 12 hours
        \item Monitor blood glucose during weaning
        \item Never abruptly discontinue (risk of hypoglycemia)
    \end{itemize}
    \item \textbf{Transition to Enteral/Oral}:
    \begin{itemize}
        \item Start enteral feeds while tapering TPN
        \item Overlap for 24-48 hours
        \item Monitor tolerance and adjust accordingly
    \end{itemize}
    \item \textbf{Catheter Management}:
    \begin{itemize}
        \item Flush with heparinized saline after discontinuation
        \item Remove if no longer needed (within 24-48 hours)
        \item Assess for signs of infection before removal
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{ngncase}{Acute Pancreatitis Management}
\textbf{Scenario:} A 45-year-old male presents to the ED with severe epigastric pain radiating to his back, nausea, and vomiting for 2 days. He admits to heavy alcohol use. Vital signs: T 38.2°C, HR 112, BP 100/60, RR 24, SpO₂ 94% on room air. Labs: WBC 18,000, amylase 850 U/L, lipase 1,200 U/L, glucose 220 mg/dL, calcium 7.2 mg/dL.

\textbf{Question 1}: Based on Ranson's criteria at admission, which findings indicate severe pancreatitis?
\begin{enumerate}[label=\Alph*.]
    \item Age > 55, WBC > 16,000, glucose > 200 mg/dL
    \item WBC > 16,000, LDH > 350 IU/L, AST > 250 IU/L
    \item Age > 55, calcium < 8 mg/dL, glucose > 200 mg/dL
    \item BUN increase > 5 mg/dL, hematocrit drop > 10\%, PaO₂ < 60 mmHg
\end{enumerate}

\textbf{Question 2}: The physician orders aggressive IV fluid resuscitation. Which fluid is MOST appropriate?
\begin{enumerate}[label=\Alph*.]
    \item 0.9\% Normal Saline at 250 mL/hr
    \item Lactated Ringer's solution at 500 mL/hr
    \item D5 0.45\% Normal Saline at 200 mL/hr
    \item 3\% Hypertonic Saline at 100 mL/hr
\end{enumerate}

\textbf{Question 3}: The nurse notes the client's pain is unrelieved despite IV morphine. Which intervention should the nurse implement FIRST?
\begin{enumerate}[label=\Alph*.]
    \item Request an order for patient-controlled analgesia (PCA)
    \item Reposition the client in a side-lying position with knees flexed
    \item Apply a heating pad to the abdomen
    \item Administer an additional dose of morphine
\end{enumerate}

\textbf{Question 4}: Which nursing diagnosis is PRIORITY for this client?
\begin{enumerate}[label=\Alph*.]
    \item Imbalanced Nutrition: Less Than Body Requirements
    \item Risk for Deficient Fluid Volume
    \item Acute Pain
    \item Impaired Gas Exchange
\end{enumerate}

\textbf{Answers and Rationale:}
\begin{enumerate}
    \item \textbf{C}. Rationale: Ranson's criteria at admission include: age > 55, WBC > 16,000, glucose > 200 mg/dL, LDH > 350 IU/L, AST > 250 IU/L. Option C includes two admission criteria (age and glucose) plus calcium < 8 mg/dL (another admission criterion).
    \item \textbf{B}. Rationale: Lactated Ringer's solution is preferred for fluid resuscitation in pancreatitis due to its balanced electrolyte composition and lower risk of metabolic acidosis compared to normal saline.
    \item \textbf{B}. Rationale: Non-pharmacological interventions should be tried first. Flexing the knees and side-lying position reduces tension on the abdomen and may provide some relief.
    \item \textbf{D}. Rationale: With SpO₂ 94% and tachypnea, impaired gas exchange is the priority due to risk of respiratory failure from third-spacing, pain, and potential ARDS.
\end{enumerate}
\end{ngncase}

\section{Renal and Urologic Nursing}

\begin{contextbox}
\textbf{Clinical Context:} The kidneys filter approximately 180 liters of blood daily, producing 1-2 liters of urine. They regulate fluid balance, electrolytes, acid-base balance, blood pressure, and produce hormones (erythropoietin, renin, active vitamin D). Kidney dysfunction affects every body system.
\end{contextbox}

\section{Acute Kidney Injury (AKI)}

\begin{definition}
\textbf{Acute Kidney Injury (AKI)} is a sudden decline in kidney function occurring over hours to days, characterized by an increase in serum creatinine and/or decrease in urine output. It is potentially reversible with appropriate treatment.
\end{definition}

\begin{criticalbox}
\textbf{AKI Diagnostic Criteria (KDIGO):}
\begin{itemize}
    \item Increase in serum creatinine ≥0.3 mg/dL within 48 hours
    \item Increase in serum creatinine ≥1.5× baseline within 7 days
    \item Urine output <0.5 mL/kg/hr for ≥6 hours
\end{itemize}
\end{criticalbox}

\subsection{Classification of AKI}

\begin{longtable}{|p{
\hline
\textbf{Type} & \textbf{Cause} & \textbf{Pathophysiology} & \textbf{Example} \\
\hline
\endfirsthead
\hline
\textbf{Type} & \textbf{Cause} & \textbf{Pathophysiology} & \textbf{Example} \\
\hline
\endhead
\hline
\endfoot
\textbf{Prerenal} (55-60\%) & Decreased renal perfusion & Glomerular filtration rate (GFR) decreases due to reduced blood flow; kidneys are structurally normal & Hypovolemia (hemorrhage, dehydration), heart failure, sepsis, hepatorenal syndrome \\
\hline
\textbf{Intrinsic/Renal} (35-40\%) & Direct damage to renal parenchyma & Tubular, glomerular, interstitial, or vascular injury & Acute tubular necrosis (ATN - most common), glomerulonephritis, acute interstitial nephritis (drug-induced), vasculitis \\
\hline

\begin{warningbox}
\textbf{Critical Distinction:} Prerenal AKI is the most common and potentially reversible with volume restoration. Intrinsic AKI (especially ATN) has a higher mortality rate and may require dialysis. Postrenal AKI requires prompt relief of obstruction to prevent permanent damage.
\end{warningbox}

\subsection{Phases of AKI}
\begin{stepbox}
\textbf{Four Phases of Acute Tubular Necrosis (ATN):}
\begin{enumerate}
    \item \textbf{Initiation Phase} (hours to days):
    \begin{itemize}
        \item Begins with the precipitating event (ischemia, nephrotoxin)
        \item GFR begins to decline
        \item Often asymptomatic
    \end{itemize}
    \item \textbf{Maintenance Phase/Oliguric Phase} (1-3 weeks):
    \begin{itemize}
        \item Urine output <400 mL/day (oliguria) or <100 mL/day (anuria)
        \item Rising BUN, creatinine, potassium
        \item Fluid overload, metabolic acidosis
        \item Highest risk for complications (hyperkalemia, pulmonary edema)
    \end{itemize}
    \item \textbf{Diuretic Phase} (1-3 weeks):
    \begin{itemize}
        \item Urine output increases (up to 3-5 L/day)
        \item BUN and creatinine begin to fall
        \item Risk of dehydration, hypokalemia, hyponatremia
    \end{itemize}
    \item \textbf{Recovery Phase} (months to 1 year):
    \begin{itemize}
        \item Gradual return of renal function
        \item May have residual impairment (especially if severe)
        \item BUN and creatinine normalize
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{clinicalpearl}
\textbf{High-Yield Tip:} Not all AKI is oliguric. Non-oliguric AKU (urine output >400 mL/day) has a better prognosis than oliguric AKI but still requires close monitoring of electrolytes and fluid balance.
\end{clinicalpearl}

\subsection{Diagnostic Studies}
\begin{stepbox}
\textbf{Key Laboratory and Diagnostic Tests:}
\begin{enumerate}
    \item \textbf{Serum Chemistry}:
    \begin{itemize}
        \item \textbf{BUN and Creatinine}: Elevated (creatinine more specific)
        \item \textbf{BUN:Creatinine Ratio}: >20:1 suggests prerenal; <20:1 suggests intrinsic
        \item \textbf{Electrolytes}: Hyperkalemia, hyperphosphatemia, hypocalcemia, metabolic acidosis
    \end{itemize}
    \item \textbf{Urinalysis and Indices}:
    \begin{itemize}
        \item \textbf{Urine Specific Gravity}: >1.020 (prerenal), 1.010-1.015 (intrinsic)
        \item \textbf{Urine Osmolality}: >500 mOsm/kg (prerenal), <350 mOsm/kg (intrinsic)
        \item \textbf{Urine Sodium}: <20 mEq/L (prerenal), >40 mEq/L (intrinsic)
        \item \textbf{Fractional Excretion of Sodium (FENa)}: <1\% (prerenal), >2\% (intrinsic)
        \item \textbf{Urine Sediment}: Muddy brown casts (ATN), RBC casts (glomerulonephritis)
    \end{itemize}
    \item \textbf{Imaging}:
    \begin{itemize}
        \item \textbf{Renal Ultrasound}: Assess kidney size, hydronephrosis (obstruction)
        \item \textbf{Doppler}: Evaluate renal blood flow
        \item \textbf{CT/MRI}: If mass or complex obstruction suspected
    \end{itemize}
    \item \textbf{Invasive}:
    \begin{itemize}
        \item \textbf{Renal Biopsy}: For unexplained intrinsic AKI (glomerulonephritis, interstitial nephritis)
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{Quick Calculation:} FENa = (Urine Na × Serum Cr) / (Serum Na × Urine Cr) × 100. <1\% indicates intact tubular function (prerenal), >2\% indicates tubular damage (intrinsic).
\end{insightbox}

\subsection{Treatment and Nursing Management}
\begin{criticalbox}
\textbf{Priority Interventions:}
\begin{enumerate}
    \item \textbf{Identify and treat the underlying cause} (volume depletion, obstruction, nephrotoxin)
    \item \textbf{Maintain fluid and electrolyte balance}
    \item \textbf{Provide adequate nutrition}
    \item \textbf{Prevent complications} (infection, bleeding, pressure injuries)
    \item \textbf{Prepare for dialysis if indicated}
\end{enumerate}
\end{criticalbox}

\begin{stepbox}
\textbf{Nursing Management for AKI:}
\begin{enumerate}
    \item \textbf{Fluid Management}:
    \begin{itemize}
        \item Strict I&O, daily weights (same scale, same time)
        \item Monitor for fluid overload (crackles, edema, JVD)
        \item Administer diuretics as prescribed (furosemide for fluid overload)
    \end{itemize}
    \item \textbf{Electrolyte Management}:
    \begin{itemize}
        \item \textbf{Hyperkalemia}: Cardiac monitoring, restrict K+ intake, administer Kayexalate, insulin/glucose, calcium gluconate for cardiac protection
        \item \textbf{Metabolic Acidosis}: Monitor ABGs, administer sodium bicarbonate if severe
        \item \textbf{Hyperphosphatemia/Hypocalcemia}: Phosphate binders, calcium supplements
    \end{itemize}
    \item \textbf{Medication Management}:
    \begin{itemize}
        \item Adjust all renally excreted medications (antibiotics, opioids, digoxin)
        \item Avoid nephrotoxins (NSAIDs, aminoglycosides, contrast dye)
    \end{itemize}
    \item \textbf{Nutrition}:
    \begin{itemize}
        \item Protein: 0.8-1.0 g/kg/day (not on dialysis); 1.2-1.4 g/kg/day (on dialysis)
        \item Restrict potassium, sodium, phosphorus as indicated
        \item Provide adequate calories (35 kcal/kg/day)
    \end{itemize}
    \item \textbf{Dialysis Indications} (mnemonic AEIOU):
    \begin{itemize}
        \item \textbf{A}cidosis refractory to treatment
        \item \textbf{E}lectrolyte imbalance (severe hyperkalemia)
        \item \textbf{I}ngestion of dialyzable toxin
        \item \textbf{O}verload (fluid) unresponsive to diuretics
        \item \textbf{U}remic symptoms (pericarditis, encephalopathy)
    \end{itemize}
\end{enumerate}
\end{stepbox}

\section{Chronic Kidney Disease (CKD)}
\begin{definition}
\textbf{Chronic Kidney Disease (CKD)} is a progressive, irreversible decline in kidney function over months to years, characterized by structural or functional abnormalities of the kidney for ≥3 months, with implications for health.
\end{definition}

\begin{theorem}
\textbf{CKD is defined as either:}
\begin{itemize}
    \item GFR <60 mL/min/1.73 m² for ≥3 months
    \item Kidney damage markers (albuminuria, urine sediment abnormalities, electrolyte abnormalities, structural abnormalities, history of kidney transplantation) for ≥3 months
\end{itemize}
\end{theorem}

\subsection{Stages of CKD}
\begin{longtable}{|c|p{3cm}|p{3cm}|p{4cm}|}
\hline
\textbf{Stage} & \textbf{GFR (mL/min/1.73 m²)} & \textbf{Description} & \textbf{Clinical Implications} \\
\hline
\endfirsthead
\hline
\textbf{Stage} & \textbf{GFR (mL/min/1.73 m²)} & \textbf{Description} & \textbf{Clinical Implications} \\
\hline
\endhead
\hline
\endfoot

\hline

\hline
3a & 45-59 & Mild to moderate decrease & Complications begin \\
\hline
3b & 30-44 & Moderate to severe decrease & Increased complications \\
\hline
4 & 15-29 & Severely decreased & Preparation for renal replacement \\
\hline

\hline
\end{longtable}

\begin{insightbox}
\textbf{Memory Aid:} "CKD Stages 1-5" - Remember that Stage 5 is when GFR is less than 15, which is about 15\% of normal kidney function. The "rule of halves": each stage represents roughly half the GFR of the previous stage.
\end{insightbox}

\subsection{Causes of CKD}
\begin{theorem}
\textbf{Most Common Causes of CKD:}
\begin{enumerate}
    \item \textbf{Diabetes mellitus} (Type 1 and 2) - 44\% of cases
    \item \textbf{Hypertension} - 29\% of cases
    \item \textbf{Glomerulonephritis}
    \item \textbf{Polycystic kidney disease}
    \item \textbf{Prolonged obstruction} (BPH, stones)
    \item \textbf{Recurrent pyelonephritis}
\end{enumerate}
\end{theorem}

\subsection{Pathophysiology}
\begin{stepbox}
\textbf{Progressive Pathophysiology of CKD:}
\begin{enumerate}
    \item \textbf{Initial Injury}: Diabetes, hypertension, or other causes damage nephrons
    \item \textbf{Hyperfiltration}: Remaining nephrons increase GFR to compensate
    \item \textbf{Glomerular Hypertension}: Increased pressure damages glomeruli
    \item \textbf{Proteinuria}: Damaged glomeruli leak protein, causing tubulointerstitial inflammation
    \item \textbf{Fibrosis}: Progressive scarring replaces functional tissue
    \item \textbf{Uremia}: Accumulation of waste products affects all body systems
\end{enumerate}
\end{stepbox}

        \item \textbf{Urinalysis}: Proteinuria, hematuria, casts
        \item \textbf{Urine Albumin-to-Creatinine Ratio (ACR)}: >30 mg/g indicates albuminuria
        \item \textbf{Electrolytes}: Hyperkalemia, hyperphosphatemia, hypocalcemia
        \item \textbf{CBC}: Anemia (normocytic, normochromic)
        \item \textbf{Intact PTH}: Elevated in secondary hyperparathyroidism
    \end{itemize}
    \item \textbf{Imaging Studies}:
    \begin{itemize}
        \item \textbf{Renal Ultrasound}: Small, echogenic kidneys (chronic), cysts, obstruction
        \item \textbf{Doppler Ultrasound}: Renal artery stenosis
        \item \textbf{CT/MRI}: Structural abnormalities (avoid contrast if GFR <30)
    \end{itemize}
    \item \textbf{Renal Biopsy}: If etiology unclear or to guide treatment
\end{enumerate}
\end{stepbox}

\subsection{Treatment and Nursing Management}
\begin{theorem}
\textbf{Goals of CKD Management:}
\begin{enumerate}
    \item Slow progression of kidney disease
    \item Manage complications
    \item Prepare for renal replacement therapy
    \item Maintain quality of life
\end{enumerate}
\end{theorem}

\begin{stepbox}
\textbf{Comprehensive Management of CKD:}
\begin{enumerate}
    \item \textbf{Blood Pressure Control}:
    \begin{itemize}
        \item Target: <130/80 mmHg (or <140/90 if proteinuria absent)
        \item First-line: ACE inhibitors or ARBs (renoprotective, reduce proteinuria)
        \item Monitor: Serum creatinine and potassium (may increase initially)
    \end{itemize}
    \item \textbf{Glycemic Control} (if diabetic):
    \begin{itemize}
        \item Target HbA1c: <7\% (individualized)
        \item Adjust medications for renal function (avoid metformin if GFR <30)
        \item SGLT2 inhibitors (empagliflozin, dapagliflozin) have renal protective effects
    \end{itemize}
    \item \textbf{Dietary Modifications}:
    \begin{itemize}
        \item Protein: 0.6-0.8 g/kg/day (Stage 3-4), 1.0-1.2 g/kg/day if on dialysis
        \item Sodium: <2,300 mg/day
        \item Potassium: Restrict if hyperkalemic (avoid bananas, oranges, potatoes)
        \item Phosphorus: Restrict to 800-1,000 mg/day (avoid dairy, nuts, colas)
    \end{itemize}
    \item \textbf{Manage Complications}:
    \begin{itemize}
        \item \textbf{Anemia}: Erythropoies
\is-stimulating agents (ESAs: epoetin alfa, darbepoetin alfa) and iron supplementation (IV iron preferred in CKD)
        \item \textbf{Mineral and Bone Disorder}:
        \begin{itemize}
            \item Phosphate binders: Calcium acetate, sevelamer, lanthanum
            \item Vitamin D supplementation: Calcitriol or analogs (paricalcitol, doxercalciferol)
            \item Calcimimetics: Cinacalcet for secondary hyperparathyroidism
        \end{itemize}
        \item \textbf{Metabolic Acidosis}: Sodium bicarbonate supplementation (target serum bicarbonate ≥22 mEq/L)
        \item \textbf{Hyperkalemia}: Dietary restriction, potassium binders (sodium polystyrene sulfonate, patiromer), loop diuretics
    \end{itemize}
    \item \textbf{Medication Adjustments}:
    \begin{itemize}
        \item Avoid nephrotoxins (NSAIDs, aminoglycosides, IV contrast)
        \item Adjust doses for renal function (use Cockcroft-Gault or CKD-EPI equations)
        \item Hold metformin if eGFR <30 mL/min/1.73m²
    \end{itemize}
    \item \textbf{Patient Education}:
    \begin{itemize}
        \item Self-management of diet, medications, and symptoms
        \item Importance of adherence to treatment plan
        \item Signs/symptoms of complications (fluid overload, hyperkalemia, uremia)
        \item Advance care planning and renal replacement therapy options
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{High-Yield Tip:} ACE inhibitors and ARBs are renoprotective in CKD by reducing intraglomerular pressure and proteinuria. However, they can cause hyperkalemia and an initial rise in creatinine (up to 30% is acceptable). Monitor serum creatinine and potassium 1-2 weeks after initiation.
\end{insightbox}

\begin{warningbox}
\textbf{Common Pitfall:} Do not discontinue ACE inhibitors/ARBs solely because of a small rise in creatinine (up to 30% from baseline) unless accompanied by hyperkalemia or symptoms. This rise reflects reduced glomerular pressure and is often beneficial long-term.
\end{warningbox}

\begin{contextbox}
\textbf{Real-World Context:} CKD management requires a multidisciplinary team: nephrologist, primary care provider, renal dietitian, social worker, and nurse educator. Early referral to nephrology (stage 4) improves outcomes and facilitates timely preparation for dialysis or transplant.
\end{contextbox}

\subsection{Dialysis}
\begin{definition}
\textbf{Dialysis} is a treatment that removes waste products and excess fluid from the blood when the kidneys can no longer perform these functions adequately. It works by diffusion, osmosis, and convection across a semipermeable membrane.
\end{definition}

\begin{theorem}
\textbf{Principles of Dialysis:}
\begin{itemize}
    \item \textbf{Diffusion}: Solutes move from high to low concentration across the membrane.
    \item \textbf{Ultrafiltration}: Water removal driven by hydrostatic or osmotic pressure.
    \item \textbf{Convection}: Solutes "dragged" along with water during ultrafiltration (solvent drag).
\end{itemize}
\end{theorem}

\subsubsection{Hemodialysis (HD)}
\begin{stepbox}
\textbf{Hemodialysis Procedure:}
\begin{enumerate}
    \item \textbf{Vascular Access}:
    \begin{itemize}
        \item \textbf{Arteriovenous Fistula (AVF)}: Surgically created anastomosis between artery and vein (preferred; lasts longest, lowest infection risk). Requires 6-8 weeks to mature.
        \item \textbf{Arteriovenous Graft (AVG)}: Synthetic tube connecting artery and vein. Can be used in 2-4 weeks. Higher risk of infection and thrombosis.
        \item \textbf{Central Venous Catheter (CVC)}: Temporary or tunneled catheter (e.g., Perm cath). For immediate use or when AVF/AVG not possible. Highest risk of infection and thrombosis.
    \end{itemize}
    \item \textbf{Dialysis Machine Setup}: Dialyzer (artificial kidney), blood lines, dialysate (electrolyte solution), and monitoring systems.
    \item \textbf{Treatment}: Blood pumped from patient through dialyzer, where waste products and excess electrolytes diffuse into dialysate. Clean blood returned to patient. Typically 3-5 hours, 3 times per week.
    \item \textbf{Adequacy Measurement}: Kt/V (target ≥1.2) or Urea Reduction Ratio (URR target ≥65%).
\end{enumerate}
\end{stepbox}

\begin{criticalbox}
\textbf{Nursing Priorities for Hemodialysis:}
\begin{itemize}
    \item \textbf{Access Care}: NEVER take blood pressure, start IVs, or draw blood from the arm with AVF/AVG. Assess for thrill/bruit every shift.
    \item \textbf{Monitor for Complications}: Hypotension (most common), muscle cramps, air embolism, hemolysis, disequilibrium syndrome.
    \item \textbf{Fluid Management}: Accurate intake/output, daily weights, assess for fluid overload.
    \item \textbf{Medication Timing}: Administer medications (especially antihypertensives) after dialysis to avoid hypotension during treatment.
\end{itemize}
\end{criticalbox}

\subsubsection{Peritoneal Dialysis (PD)}
\begin{stepbox}
\textbf{Peritoneal Dialysis Procedure:}
\begin{enumerate}
    \item \textbf{Access}: Permanent peritoneal catheter surgically placed in abdomen.
    \item \textbf{Principle}: Uses peritoneum as natural semipermeable membrane. Dialysate infused into peritoneal cavity, dwells for prescribed time, then drained.
    \item \textbf{Types}:
    \begin{itemize}
        \item \textbf{Continuous Ambulatory Peritoneal Dialysis (CAPD)}: Manual exchanges 4-5 times daily, 24 hours continuous.
        \item \textbf{Automated Peritoneal Dialysis (APD)}: Cycler machine performs exchanges overnight while patient sleeps.
    \end{itemize}
    \item \textbf{Exchange Steps}: Drain → Fill → Dwell → Drain (repeat).
\end{enumerate}
\end{stepbox}

\begin{skillbox}
\textbf{Nursing Skill: Peritoneal Dialysis Catheter Care}
\begin{enumerate}
    \item Perform hand hygiene and wear mask during exchanges.
    \item Inspect exit site daily for redness, swelling, drainage, or tenderness.
    \item Clean exit site with chlorhexidine or prescribed solution daily.
    \item Secure catheter to prevent tension or pulling.
    \item Teach patient signs of peritonitis: cloudy dialysate, fever, abdominal pain, rebound tenderness.
\end{enumerate}
\end{skillbox}

\begin{insightbox}
\textbf{Key Insight:} Cloudy dialysate effluent in PD is peritonitis until proven otherwise. Obtain sample for cell count, Gram stain, and culture before starting antibiotics.
\end{insightbox}

\subsubsection{Continuous Renal Replacement Therapy (CRRT)}
\begin{definition}
\textbf{CRRT} is a slow, continuous dialysis modality used for hemodynamically unstable patients in the ICU (e.g., sepsis, multi-organ failure). It provides gradual fluid and solute removal over 24 hours, better tolerated than intermittent HD.
\end{definition}

\subsection{Kidney Transplantation}
\begin{definition}
\textbf{Kidney Transplantation} is the surgical placement of a healthy kidney from a donor into a patient with end-stage renal disease (ESRD). It is the preferred renal replacement therapy because it offers better quality of life and survival compared to dialysis.
\end{definition}

\subsubsection{Pre-Transplant Evaluation}
\begin{stepbox}
\textbf{Comprehensive Pre-Transplant Workup:}
\begin{enumerate}
    \item \textbf{Immunological Testing}:
    \begin{itemize}
        \item ABO blood type compatibility
        \item Human Leukocyte Antigen (HLA) typing (A, B, DR)
        \item Panel Reactive Antibody (PRA) - measures sensitization
        \item Crossmatch (T-cell and B-cell)
    \end{itemize}
    \item \textbf{Medical Evaluation}:
    \begin{itemize}
        \item Cardiovascular: Stress test, echocardiogram, carotid studies
        \item Infectious disease: HIV, hepatitis B/C, TB, CMV, EBV
        \item Cancer screening: Age-appropriate (colonoscopy, mammogram, PSA)
        \item Urological: Voiding cystourethrogram, urology consult
    \end{itemize}
    \item \textbf{Psychosocial Evaluation}:
    \begin{itemize}
        \item Mental health assessment
        \item Social support system
        \item Financial counseling
        \item Compliance history
    \end{itemize}
\end{enumerate}
\end{stepbox}

\subsubsection{Post-Transplant Care}
\begin{theorem}
\textbf{Immediate Post-Transplant Goals:}
\begin{enumerate}
    \item Monitor graft function (hourly urine output, daily creatinine)
    \item Prevent rejection with immunosuppression
    \item Prevent infection (immunosuppression increases risk)
    \item Manage surgical complications (bleeding, thrombosis, urine leak)
\end{enumerate}
\end{theorem}

\begin{stepbox}
\textbf{Nursing Management Post-Transplant:}
\begin{enumerate}
    \item \textbf{Vital Signs}: Monitor for hypertension (common) and hypotension (bleeding).
    \item \textbf{Fluid Balance}: Strict I&O, hourly urine output (expect diuresis initially).
    \item \textbf{Graft Assessment}: Palpate graft site (usually iliac fossa) for tenderness, swelling.
    \item \textbf{Laboratory Monitoring}: Daily creatinine, BUN, electrolytes, CBC, immunosuppressant levels.
    \item \textbf{Infection Prevention}:
    \begin{itemize}
        \item Protective isolation as indicated
        \item Hand hygiene, avoid sick contacts
        \item Prophylactic antibiotics (TMP-SMX for PCP, valganciclovir for CMV)
    \end{itemize}
    \item \textbf{Patient Education}:
    \begin{itemize}
        \item Lifelong immunosuppression adherence
        \item Signs of rejection: fever, graft tenderness, decreased urine output, weight gain
        \item Signs of infection: fever, cough, dysuria, wound redness
        \item Avoid live vaccines, sun protection (increased skin cancer risk)
    \end{itemize}
\end{enumerate}
\end{stepbox}

\begin{insightbox}
\textbf{Typical Immunosuppression Regimen:}
\begin{itemize}
    \item \textbf{Induction}: High-intensity therapy at transplant (basiliximab or ATG)
    \item \textbf{Maintenance}: Triple therapy
    \begin{enumerate}
        \item Calcineurin inhibitor (tacrolimus preferred)
        \item Antimetabolite (mycophenolate mofetil)
        \item Corticosteroid (prednisone, often tapered over months)
    \end{enumerate}
    \item Target trough levels: Tacrolimus 8-12 ng/mL (early), 5-8 ng/mL (later)
\end{itemize}
\end{insightbox}

\begin{criticalbox}
\textbf{Critical Nursing Considerations for Immunosuppression:}
\begin{itemize}
    \item \textbf{Adherence}: Lifelong therapy; non-adherence is leading cause of rejection
    \item \textbf{Drug Interactions}: Many medications affect immunosuppressant levels (e.g., azole antifungals increase levels, rifampin decreases levels)
    \item \textbf{Monitoring}: Regular blood levels, renal function, glucose, lipids
    \item \textbf{Infection Risk}: Educate on hand hygiene, avoiding sick contacts, food safety, sun protection
    \item \textbf{Cancer Screening}: Increased risk of skin cancer, lymphoma, and other malignancies
\end{itemize}
\end{criticalbox}

\subsubsection{Transplant Rejection}
\begin{definition}
\textbf{Transplant Rejection} is an immune-mediated response against the transplanted kidney, categorized by timing and mechanism:
\begin{itemize}
    \item \textbf{Hyperacute}: Minutes to hours (preformed antibodies)
    \item \textbf{Acute}: Days to months (cellular or antibody-mediated)
    \item \textbf{Chronic}: Months to years (progressive graft dysfunction)
\end{itemize}
\end{definition}

\textbf{Type} & \textbf{Timing} & \textbf{Mechanism} & \textbf{Clinical Presentation} & \textbf{Treatment} \\

\begin{stepbox}
\textbf{Nursing Management of Transplant Rejection:}
\begin{enumerate}
    \item \textbf{Early Detection}: Monitor for decreased urine output, weight gain, fever, graft tenderness
    \item \textbf{Laboratory Monitoring}: Daily creatinine, BUN; report >20% increase from baseline
    \item \textbf{Medication Administration}: Ensure timely immunosuppressant doses; never hold without MD order
    \item \textbf{Patient Education}: Teach signs of rejection and when to report
    \item \textbf{Psychosocial Support}: Address anxiety, adherence challenges
\end{enumerate}
\end{stepbox}

\begin{clinicalpearl}
\textbf{Banff Classification}: Standardized histological grading system for renal allograft rejection based on biopsy findings. Guides treatment decisions and predicts outcomes.
\end{clinicalpearl}

\begin{warningbox}
\textbf{Common Pitfalls in Transplant Care:}
\begin{itemize}
    \item Assuming increased creatinine always means rejection (consider dehydration, drug toxicity, infection)
    \item Forgetting that immunosuppressants increase risk of infection AND malignancy
    \item Not checking trough levels before administering next dose
    \item Overlooking drug interactions (e.g., grapefruit juice with cyclosporine/tacrolimus)
\end{itemize}
\end{warningbox}

\section{Summary}
\begin{itemize}
    \item \textbf{Acute Pancreatitis}: Remember "I GET SMASHED" for etiology; pain management and fluid resuscitation are priorities; Ranson's criteria assess severity
    \item \textbf{Pancreatic Cancer}: Courvoisier's sign (painless jaundice with palpable gallbladder); Whipple procedure is curative; poor prognosis
    \item \textbf{Cholecystitis}: Charcot's triad (RUQ pain, fever, jaundice) indicates cholangitis; Murphy's sign positive; cholecystectomy is definitive treatment
    \item \textbf{TPN}: Central line required; monitor glucose, electrolytes, triglycerides; strict aseptic technique to prevent infection
    \item \textbf{AKI}: Prerenal (hypoperfusion), intrinsic (parenchymal damage), postrenal (obstruction); oliguria vs. non-oliguria
\item \textbf{CKD}: Stages 1-5 based on GFR; complications include anemia, bone disease, cardiovascular disease
    \item \textbf{Dialysis}: Hemodialysis (intermittent, requires vascular access), peritoneal dialysis (continuous, uses peritoneum), CRRT (continuous, for critically ill)
    \item \textbf{Kidney Transplantation}: Living vs. deceased donors; lifelong immunosuppression required; acute vs. chronic rejection
\end{itemize}

Mrs. Chen, 58 years old, has a 20-year history of type 1 diabetes. She presents with the following:

\textbf{Subjective Data:}

    \item "My ankles are swollen by the end of the day"
    \item "I've been itching all over for the past week"

    \item "I've been short of breath when walking upstairs"
\end{itemize}

\textbf{Objective Data:}
\begin{itemize}
    \item Weight: 72 kg (baseline 68 kg two weeks ago)
    \item BP: 168/94 mmHg
    \item HR: 92 bpm, irregular
    \item RR: 22/min
    \item JVD present, bilateral lung crackles
    \item 3+ pitting edema bilateral lower extremities
    \item Skin: dry, excoriations on arms and legs
\end{itemize}

\textbf{Lab Results:}
\begin{itemize}
    \item GFR: 14 mL/min/1.73m²
    \item BUN: 82 mg/dL
    \item Creatinine: 7.1 mg/dL
    \item Potassium: 6.2 mEq/L
    \item Hemoglobin: 7.8 g/dL
    \item Albumin: 2.8 g/dL
    \item Calcium: 7.2 mg/dL
    \item Phosphorus: 8.1 mg/dL
    \item Intact PTH: 485 pg/mL (normal: 10-65)
\end{itemize}

\textbf{Questions:}
\begin{enumerate}
    \item Identify ALL clinical manifestations of uremia present in this client.
    \item What is the significance of the elevated PTH level?
    \item Calculate the client's fluid overload (weight gain minus expected daily variation).
    \item What cardiac dysrhythmia is the client at risk for based on lab values, and what is the priority nursing intervention?
    \item List five priority nursing interventions for this client.
\end{enumerate}

\textbf{Answers:}
\begin{enumerate}
    \item Clinical manifestations of uremia present in this client include: fatigue, nausea, metallic taste, anorexia, pruritus, edema, shortness of breath, and elevated BUN and creatinine.
    \item Elevated PTH (secondary hyperparathyroidism) indicates renal osteodystrophy due to CKD. The kidneys fail to convert vitamin D to its active form, leading to hypocalcemia, which stimulates PTH release. Hyperphosphatemia also contributes.
    \item Fluid overload: 72 kg - 68 kg = 4 kg (approximately 4 liters) over two weeks, so about 2 liters per week or 286 mL per day.
    \item The client is at risk for cardiac dysrhythmias due to hyperkalemia (K+ 6.2 mEq/L). Specifically, peaked T waves, widened QRS, and potentially ventricular fibrillation.
    \item Priority nursing interventions:
    \begin{itemize}
        \item Monitor cardiac rhythm continuously (ECG)
        \item Administer IV calcium gluconate to stabilize cardiac membrane
        \item Administer insulin and glucose to shift potassium intracellularly
        \item Prepare for emergent dialysis
        \item Restrict dietary potassium and phosphorus
        \item Administer erythropoiesis-stimulating agent and iron for anemia
        \item Provide phosphate binders with meals
        \item Administer antihypertensives and diuretics for fluid overload
        \item Educate on fluid restrictions (typically 1,000-1,500 mL/day)
        \item Refer for vascular access placement for dialysis
    \end{itemize}
\end{enumerate}
\end{exercise}

\section{Summary and Key Takeaways}

\subsection{Critical Endocrine/GI Disorders}

\begin{theorem}
\textbf{High-Yield Concepts for Exam Success:}
\begin{enumerate}
    \item \textbf{Acute Pancreatitis}: "I GET SMASHED" mnemonic; fluid resuscitation is the priority intervention; opioid analgesics (morphine or hydromorphone) acceptable per current evidence — meperidine no longer preferred (toxic metabolite); Ranson's criteria used at admission and 48 hours to predict severity
    \item \textbf{Cholecystitis}: Charcot's triad (RUQ pain, fever, jaundice) = cholangitis; Reynolds' pentad = suppurative cholangitis (emergency); Murphy's sign positive in acute cholecystitis
    \item \textbf{TPN}: Central line required (high osmolarity); monitor glucose q4-6h initially; taper before discontinuing (prevent hypoglycemia); strict aseptic technique
    \item \textbf{Pancreatic Cancer}: Courvoisier's sign (painless jaundice + palpable non-tender gallbladder); Whipple procedure; poor prognosis (5-year survival <5\%)
\end{enumerate}
\end{theorem}

\subsection{Critical Renal Disorders}

\begin{theorem}
\textbf{High-Yield Concepts for Exam Success:}
\begin{enumerate}
    \item \textbf{AKI}: Prerenal (most common, hypoperfusion), intrinsic (ATN most common), postrenal (obstruction); BUN:Cr ratio >20:1 suggests prerenal
    \item \textbf{CKD Stages}: Stage 1 (GFR ≥90 with damage), Stage 5 (GFR <15 = ESRD); complications include anemia (decreased EPO), bone disease (elevated PTH), cardiovascular disease
    \item \textbf{Hemodialysis}: AV fistula preferred access (maturation 6-8 weeks); assess for thrill/bruit; complications include hypotension, disequilibrium syndrome, infection
    \item \textbf{Peritoneal Dialysis}: Cloudy dialysate = peritonitis until proven otherwise; assess for catheter complications; continuous therapy
    \item \textbf{Transplant}: Lifelong immunosuppression; acute rejection (reversible) vs chronic rejection (irreversible); infection is leading cause of death post-transplant
\end{enumerate}
\end{theorem}

\subsection{NCLEX Test-Taking Strategies}

\begin{insightbox}
\textbf{Priority Framework for Exam Questions:}
\begin{enumerate}
    \item \textbf{Airway-Breathing-Circulation}: Always prioritize respiratory and hemodynamic stability
    \item \textbf{Unstable before Stable}: Address acute changes before routine care
    \item \textbf{Assess before Intervene}: Unless the situation is obvious (e.g., cardiac arrest), assess first
    \item \textbf{Actual before Potential}: Address actual problems before potential complications
    \item \textbf{Acute before Chronic}: Acute changes take priority over chronic conditions
\end{enumerate}
\end{insightbox}

\begin{criticalbox}
\textbf{Critical Values to Remember:}
\begin{itemize}
    \item Potassium >6.0 mEq/L or <3.0 mEq/L = cardiac emergency
    \item Creatinine >4.0 mg/dL = likely need for dialysis
    \item GFR <15 mL/min = ESRD, renal replacement therapy needed
    \item Amylase/Lipase >3x normal = pancreatitis
    \item WBC >20,000 with left shift = serious infection
    \item Hemoglobin <7 g/dL = may require transfusion
\end{itemize}
\end{criticalbox}

\begin{warningbox}
\textbf{Common Exam Pitfalls:}
\begin{itemize}
    \item Don't confuse acute vs chronic pancreatitis (different etiologies and presentations)
    \item Opioid analgesics (morphine or hydromorphone) are acceptable for pancreatitis pain per current evidence; meperidine is no longer preferred due to toxic metabolite accumulation risk (normeperidine); some older NCLEX items may still reflect the historical "avoid morphine" teaching
    \item Don't forget to taper TPN before discontinuing (hypoglycemia risk)
    \item Don't confuse hemodialysis and peritoneal dialysis complications
    \item Don't assume all kidney transplant patients need the same immunosuppression regimen
    \item Remember: AV fistula is preferred over AV graft or temporary catheter for hemodialysis
\end{itemize}
\end{warningbox}

\textbf{Question 1:} \textbf{Answer:} B \\
\textbf{Rationale:} Hypotension with tachycardia indicates hypovolemic shock from third-spacing of fluid, which is a life-threatening complication requiring immediate intervention. While elevated amylase, pain, and nausea are expected findings in pancreatitis, hemodynamic instability signals a critical emergency.


\textbf{Question 2:} \textbf{Answer:} A \\
\textbf{Rationale:} Hypocalcemia in pancreatitis occurs due to saponification (calcium binds to fatty acids released by pancreatic enzyme activity). While hypocalcemia is expected in pancreatitis, a level of 7.2 mg/dL is dangerously low and requires treatment with calcium gluconate to prevent cardiac dysrhythmias and tetany.


\textbf{Question 3:} \textbf{Answer:} B \\
\textbf{Rationale:} Serum lipase is more specific for pancreatitis than amylase. Amylase can also be elevated in salivary gland disease, renal failure, and other conditions. Lipase remains elevated longer (8-14 days) and is considered the more reliable marker.


\textbf{Question 4:} \textbf{Answer:} B \\
\textbf{Rationale:} During the acute phase, the client should be NPO (nothing by mouth) to rest the pancreas. Even small sips of water stimulate pancreatic secretions and can worsen the condition. The client should receive IV fluids for hydration until oral intake is resumed as ordered.


\textbf{Question 5:} \textbf{Answer:} A \\
\textbf{Rationale:} Pain management is a priority. Opioid analgesics (such as morphine) are used, though meperidine was traditionally preferred due to less spasm of the sphincter of Oddi. However, current evidence supports using morphine or hydromorphone. Heating pads are avoided as they can increase swelling and bleeding. The client is typically NPO. Sitting upright or leaning forward may help with pain.


\textbf{Question 6:} \textbf{Answer:} B \\
\textbf{Rationale:} The primary purpose of an NG tube in acute pancreatitis is to decompress the stomach by removing gastric secretions, which reduces stimulation of the pancreas (as gastric acid entering the duodenum triggers pancreatic enzyme release). While it can help prevent aspiration, that is not the primary goal in this context.


\textbf{Question 7:} \textbf{Answer:} C \\
\textbf{Rationale:} Fever in a client receiving TPN may indicate a catheter-related bloodstream infection, a serious complication. The other values are within normal limits or expected (weight gain may reflect fluid balance, but a sudden 0.5 kg gain in 24 hours could also be concerning for fluid overload, but fever is more specific for infection).


\textbf{Question 8:} \textbf{Answer:} A \\
\textbf{Rationale:} Cholestyramine is a bile acid sequestrant that binds bile salts in the intestine, reducing pruritus associated with obstructive jaundice. While diphenhydramine may provide some relief, cholestyramine addresses the underlying cause of the itching.


\textbf{Question 9:} \textbf{Answer:} C \\
\textbf{Rationale:} Milky-white drain output with elevated amylase suggests a pancreatic fistula, a serious complication requiring immediate intervention. Greenish-brown output may indicate bile, which is expected. Absence of bowel sounds and moderate pain are expected in the early post-operative period.


\textbf{Question 10:} \textbf{Answer:} B \\
\textbf{Rationale:} During laparoscopic cholecystectomy, CO\textsubscript{2} gas is used to inflate the abdomen for visualization. Residual gas can irritate the diaphragm and cause referred pain to the shoulder. This is a common and expected finding that typically resolves within 24-48 hours. Early ambulation is encouraged, the incisions are small, and long-term dietary restrictions are usually not necessary.


\textbf{Question 11:} \textbf{Answer:} B \\
\textbf{Rationale:} Murphy's sign is elicited by having the patient take a deep breath while the examiner palpates the RUQ. The inflamed gallbladder descends during inspiration and contacts the examining fingers, causing the patient to catch their breath or experience sharp pain. This is a classic finding in acute cholecystitis.


\textbf{Question 12:} \textbf{Answer:} C \\
\textbf{Rationale:} Fever and worsening abdominal pain after ERCP may indicate complications such as perforation, acute pancreatitis, or cholangitis. These require immediate evaluation. Mild throat discomfort is expected from the endoscope, slight distension may occur, and green-tinged emesis could be residual bile but should be monitored.


\textbf{Question 13:} \textbf{Answer:} B \\
\textbf{Rationale:} After cholecystectomy, the body needs time to adjust to the absence of the gallbladder's bile storage function. A gradual return to a normal diet, starting with low-fat foods and smaller meals, helps prevent digestive discomfort. Resuming a normal diet immediately may cause diarrhea and abdominal cramping.


\textbf{Question 14:} \textbf{Answer:} C \\
\textbf{Rationale:} Charcot's triad consists of RUQ pain, jaundice, and fever/chills, and is the classic presentation of cholangitis (infection of the common bile duct). This is a medical emergency requiring antibiotics and biliary decompression (ERCP). Reynolds' pentad adds altered mental status and hypotension, indicating septic shock.


\textbf{Question 15:} \textbf{Answer:} B \\
\textbf{Rationale:} Cholecystitis is an inflammatory condition, so an elevated WBC count (leukocytosis) is expected. Serum bilirubin may be elevated if there is associated CBD obstruction. Amylase would be elevated in pancreatitis, not typically in isolated cholecystitis. Albumin is not typically affected.


\textbf{Question 16:} \textbf{Answer:} B \\
\textbf{Rationale:} Pancrelipase is a pancreatic enzyme replacement that must be taken with food to be effective. It helps digest fats, proteins, and carbohydrates, reducing steatorrhea and improving nutritional status.


\textbf{Question 17:} \textbf{Answer:} C \\
\textbf{Rationale:} Laparoscopic cholecystectomy uses CO₂ gas to inflate the abdomen. Residual gas can irritate the diaphragm and cause referred pain to the shoulder. This is normal and typically resolves in 24-48 hours.


\textbf{Question 18:} \textbf{Answer:} A \\
\textbf{Rationale:} When catheter-related bloodstream infection is suspected, blood cultures should be obtained from the central line AND a peripheral site before initiating antibiotics. This helps identify the causative organism and guide treatment.


\textbf{Question 19:} \textbf{Answer:} C \\
\textbf{Rationale:} A serum phosphorus level of 1.2 mg/dL indicates severe hypophosphatemia, which is a sign of refeeding syndrome. This is a life-threatening complication that requires immediate intervention (IV phosphate replacement).


\textbf{Question 20:} \textbf{Answer:} B \\
\textbf{Rationale:} Murphy's sign is a classic finding in acute cholecystitis. It is elicited by having the client take a deep breath while the examiner palpates the right subcostal area. The inflamed gallbladder descends during inspiration and comes into contact with the examining fingers, causing sudden pain and arrest of inspiration.


\textbf{Question 21:} \textbf{Answer:} B \quad \textbf{Rationale:} Fever with redness at the central line insertion site indicates a potential catheter-related bloodstream infection (CRBSI), a life-threatening complication of TPN. The nurse must notify the provider immediately, obtain blood cultures (peripheral and central), and anticipate catheter removal and antibiotic therapy. Mild hyperglycemia, borderline triglycerides, and slight weight gain are monitored but do not require immediate intervention.


\textbf{Question 22:} \textbf{Answer:} B \\
\textbf{Rationale:} A sudden decrease in urine output with rising creatinine is a classic sign of acute rejection. The transplant surgeon must be notified immediately so that diagnostic testing (including possible biopsy) can be initiated and treatment can begin promptly to save the graft.


\textbf{Question 23:} \textbf{Answer:} C \\
\textbf{Rationale:} Immunosuppressant medications must be taken for life. Stopping prednisone or any immunosuppressant without medical guidance can lead to acute rejection and graft loss. The client must understand that these medications are lifelong.


\textbf{Question 24:} \textbf{Answer:} D \\
\textbf{Rationale:} Hyperkalemia causes characteristic ECG changes that progress as potassium levels increase: peaked T waves (early sign), prolonged PR interval, widened QRS complex, loss of P waves, and eventually sine wave pattern leading to cardiac arrest. All of these changes can be seen with severe hyperkalemia.


\textbf{Question 25:} \textbf{Answer:} B \\
\textbf{Rationale:} Cloudy dialysate effluent is the hallmark sign of peritonitis in peritoneal dialysis. Other signs include abdominal pain, fever, and rebound tenderness. The nurse should obtain a sample of the fluid for culture and sensitivity and notify the provider immediately.


\textbf{Question 26:} \textbf{Answer:} C \\
\textbf{Rationale:} Active malignancy is an absolute contraindication to kidney transplantation because immunosuppressive therapy would worsen the cancer. Age alone is not a contraindication (though outcomes may be less favorable). Non-melanoma skin cancer with adequate treatment and no recurrence is usually acceptable. \subsection*{Case Study 7}
\textbf{Bow-Tie Clinical Judgment Answers:}

\textbf{Recognize Cues:}
\begin{itemize}
    \item Priority assessments: Urine output, vital signs (hypotension, tachycardia), mental status, edema, lung sounds
    \item Lab values: BUN, creatinine, GFR, electrolytes (especially potassium), urinalysis, urine sodium, urine osmolality
\end{itemize}

\textbf{Analyze Cues:}
\begin{itemize}
    \item Most likely cause: Gentamicin nephrotoxicity compounded by sepsis-induced hypoperfusion
    \item Type of AKI: Intrinsic (acute tubular necrosis from aminoglycoside toxicity)
\end{itemize}

\textbf{Prioritize Hypotheses:}
\begin{enumerate}
    \item Acute tubular necrosis (gentamicin + sepsis)
    \item Prerenal AKI (sepsis-induced hypotension)
    \item Postrenal obstruction (less likely given clinical context)
\end{enumerate}
\textbf{Differentiating data:} Urine sodium >40 mEq/L and FENa >2\% suggest intrinsic; urine sodium <20 mEq/L and FENa <1\% suggest prerenal

\textbf{Take Action:}
\begin{itemize}
    \item Immediate interventions: Notify provider, obtain stat labs, discontinue gentamicin, ensure adequate IV fluid resuscitation, strict I&O
    \item Medications: Hold gentamicin, consider alternative antibiotic, avoid nephrotoxins (NSAIDs, contrast)
\end{itemize}

\textbf{Evaluate Outcomes:}
\begin{itemize}
    \item Improvement: Increasing urine output (>0.5 mL/kg/hr), decreasing creatinine, stable electrolytes
    \item Reassess: Hourly urine output, daily creatinine, electrolytes every 6-12 hours until stable
\end{itemize}
