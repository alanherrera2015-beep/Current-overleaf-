\subsection{Formula Reference Sheet}

You may use the following formulas during the exam:

\begin{tcolorbox}[colback=white, colframe=black, title=\textbf{Formula Sheet}]
\textbf{Work:}
\begin{align*}
W &= F d \cos\theta \\
W &= \vec{F} \cdot \vec{d}
\end{align*}

\textbf{Kinetic Energy:}
\[
K = \frac{1}{2}mv^2
\]

\textbf{Work-Energy Theorem:}
\[
W_{\text{net}} = \Delta K = K_f - K_i = \frac{1}{2}mv_f^2 - \frac{1}{2}mv_i^2
\]

\textbf{Gravitational Potential Energy:}
\[
U_g = mgh
\]

\textbf{Elastic Potential Energy:}
\[
U_s = \frac{1}{2}kx^2
\]

\textbf{Thermal Energy from Friction:}
\[
\Delta E_{\text{th}} = f_k \cdot d = \mu_k N d
\]

\textbf{Conservation of Energy:}
\[
K_i + U_i + W_{\text{ext}} = K_f + U_f + \Delta E_{\text{th}}
\]

\textbf{Work by Gravity on Incline:}
\[
W_g = mgh = mgd\sin\theta
\]

\textbf{Power:}
\[
P = \frac{W}{t} = Fv
\]
\end{tcolorbox}

\section{Solutions: Part A --- The Basic Energy Model}

\begin{solution}
\textbf{Solution A1.}

\textbf{(a) System and Environment:}
\begin{itemize}
    \item \textbf{System:} The ball
    \item \textbf{Environment:} Earth (provides gravitational interaction), the person's hand (provides initial push), air (provides resistance)
\end{itemize}

\textbf{(b) Forms of Energy:}
\begin{itemize}
    \item \textbf{Kinetic Energy ($K$):} The ball has kinetic energy due to its motion. This is maximum at the moment of release and at the lowest points, minimum (zero) at the maximum height.
    \item \textbf{Gravitational Potential Energy ($U_g$):} The ball has gravitational potential energy due to its height above the ground. This increases as the ball rises and decreases as it falls.
    \item \textbf{Thermal Energy ($E_{th}$):} Some energy is converted to thermal energy due to air resistance (though this is typically small for a thrown ball).
\end{itemize}

\textbf{(c) Energy Transformations:}
\begin{itemize}

\begin{solution}
\textbf{Solution A3.}

\textbf{(a) Open System:}

An \textbf{open system} is one that can exchange energy with its environment. Energy can flow into or out of the system through work or heat transfer.

\textbf{Example:} A car engine. Fuel (chemical energy) enters the system, and the engine does work on the wheels while also releasing heat to the environment.

\textbf{(b) Closed System:}

A \textbf{closed system} is one that does not exchange energy with its environment. The total energy within the system remains constant.

\textbf{Example:} An ideal pendulum in a vacuum (no air resistance). The total mechanical energy ($K + U_g$) remains constant as energy transforms back and forth between kinetic and potential forms.

\textbf{(c) Isolated System:}

An \textbf{isolated system} is a special case of a closed system where there is no interaction with the environment whatsoever---no energy transfer, no mass transfer, no external forces.

\textbf{Example:} The entire universe (theoretically), or an insulated container with a bouncing ball where we ignore all external influences.

\textbf{Key Distinction:} In practice, truly isolated systems don't exist, but we often model systems as isolated when external influences are negligible. A closed system has constant energy; an isolated system has no external interactions at all.
\end{solution}

\begin{solution}
\textbf{Solution A4.}

The energy principle (also called the first law of thermodynamics for mechanical systems) states:

\begin{center}
\boxed{\Delta E_{sys} = W_{ext}}
\end{center}

Or equivalently:
\begin{center}
\boxed{E_f = E_i + W_{ext}}
\end{center}

Where:
\begin{itemize}
    \item $\Delta E_{sys}$ = change in total system energy = $\Delta K + \Delta U + \Delta E_{th}$
    \item $W_{ext}$ = work done by external forces (forces from the environment)
    \item $E_i$ = initial total energy of the system
    \item $E_f$ = final total energy of the system
\end{itemize}

\textbf{In words:} The change in a system's total energy equals the net work done on the system by external forces.

\textbf{For an isolated system:} $W_{ext} = 0$, so $\Delta E_{sys} = 0$, meaning:
\[E_f = E_i\]
\[K_f + U_f + E_{th,f} = K_i + U_i + E_{th,i}\]

This is the \textbf{law of conservation of energy}: In an isolated system, energy can transform from one form to another, but the total energy remains constant.
\end{solution}

\[W_{net} = \frac{1}{2}mv_f^2 - \frac{1}{2}mv_i^2\]

Since $v_i = 0$:

\begin{solution}
\textbf{Solution C6.}

\textbf{Given:}
\begin{itemize}
    \item Object A: mass $m_A = m$, speed $v_A$
    \item Object B: mass $m_B = 4m$, speed $v_B$
    \item Both objects have the same kinetic energy: $K_A = K_B$
\end{itemize}

\textbf{(a) Ratio of speeds $v_A/v_B$:}

\textbf{Setting kinetic energies equal:}
\[\frac{1}{2}m_A v_A^2 = \frac{1}{2}m_B v_B^2\]

\[\frac{1}{2}(m)v_A^2 = \frac{1}{2}(4m)v_B^2\]

\textbf{Simplifying:}
\[v_A^2 = 4v_B^2\]

\[v_A = 2v_B\]

\[\boxed{\frac{v_A}{v_B} = 2}\]

\textbf{Interpretation:} The lighter object must travel twice as fast to have the same kinetic energy as the heavier object.

\textbf{(b) Ratio of momenta $p_A/p_B$:}

\textbf{Formula for momentum:}
\[p = mv\]

\textbf{For Object A:}
\[p_A = m_A v_A = m \cdot v_A\]

\textbf{For Object B:}
\[p_B = m_B v_B = 4m \cdot v_B\]

\textbf{Using $v_A = 2v_B$:}
\[p_A = m(2v_B) = 2mv_B\]

\[p_B = 4mv_B\]

\textbf{Taking the ratio:}
\[\frac{p_A}{p_B} = \frac{2mv_B}{4mv_B} = \frac{2}{4}\]

\[\boxed{\frac{p_A}{p_B} = \frac{1}{2}}\]

\textbf{Interpretation:} Even though both objects have the same kinetic energy, the heavier object has twice the momentum.

\textbf{(c) Ratio of stopping distances:}

If both objects are brought to rest by the same constant force $F$, we use the work-energy theorem:

\[W = F \cdot d = \Delta K = K_f - K_i = 0 - K_i = -K_i\]

Since the force opposes motion:
\[F \cdot d = K_i\]

\[d = \frac{K_i}{F}\]

Since both objects have the same kinetic energy ($K_A = K_B$) and are stopped by the same force $F$:

\[d_A = \frac{K_A}{F} \quad \text{and} \quad d_B = \frac{K_B}{F}\]

\
\[\frac{d_A}{d_B} = \frac{K_A/F}{K_B/F} = \frac{K_A}{K_B} = 1\]

\[\boxed{\frac{d_A}{d_B} = 1}\]

\textbf{Interpretation:} Objects with the same kinetic energy require the same stopping distance when stopped by the same force, regardless of their mass. This is why kinetic energy (not momentum) determines stopping distance in braking problems.
\end{solution}

\begin{solution}
\textbf{Solution E4.}

    \item Car B: $m_B = 2000$ kg, $v_B = 20$ m/s
    \item Same coefficient of friction: $\mu_k$
\end{itemize}

\textbf{Formula:}
\[K = \frac{1}{2}mv^2\]

\[K_A = \frac{1}{2}(1000)(20)^2 = \frac{1}{2}(1000)(400) = 200{,}000 \text{ J}\]

\[K_B = \frac{1}{2}(2000)(20)^2 = \frac{1}{2}(2000)(400) = 400{,}000 \text{ J}\]

\[\frac{K_A}{K_B} = \frac{200{,}000}{400{,}000} = \frac{1}{2}\]

\[\boxed{\frac{K_A}{K_B} = \frac{1}{2}}\]

\textbf{(b) Ratio of friction forces:}

On level ground: $N = mg$, so $f_k = \mu_k mg$

\[\frac{f_{k,A}}{f_{k,B}} = \frac{\mu_k m_A g}{\mu_k m_B g} = \frac{m_A}{m_B} = \frac{1000}{2000} = \frac{1}{2}\]

\[\boxed{\frac{f_{k,A}}{f_{k,B}} = \frac{1}{2}}\]

\textbf{(c) Ratio of stopping distances:}

Using $\Delta E_{th} = K_i$ and $\Delta E_{th} = f_k \cdot d$:

\[
d = \frac{K_i}{f_k} = \frac{\frac{1}{2}mv^2}{\mu_k mg} = \frac{v^2}{2\mu_k g}\]

Notice that mass cancels! The stopping distance depends only on speed and friction coefficient, not mass.

\[d_A = \frac{v_A^2}{2\mu_k g} = \frac{(20)^2}{2\mu_k g}\]

\[d_B = \frac{v_B^2}{2\mu_k g} = \frac{(20)^2}{2\mu_k g}\]

Since both cars have the same speed and same $\mu_k$:

\[\frac{d_A}{d_B} = 1\]

\[\boxed{\frac{d_A}{d_B} = 1 \text{ (same stopping distance)}}\]

\textbf{(d) Physical explanation:}

This result seems counterintuitive at first—shouldn't the heavier car be harder to stop?

The key insight is that while the heavier car has more kinetic energy to dissipate, it also has a proportionally larger friction force (because $f_k = \mu_k mg$). These two effects exactly cancel:

\begin{itemize}
    \item Car B has \textbf{twice the kinetic energy} as Car A
    \item Car B also has \textbf{twice the friction force} as Car A
    \item Therefore, Car B dissipates energy twice as fast
    \item Result: Both cars stop in the same distance
\end{itemize}

\textbf{Mathematical proof:}
\[d = \frac{K}{f_k} = \frac{\frac{1}{2}mv^2}{\mu_k mg} = \frac{v^2}{2\mu_k g}\]

Mass $m$ appears in both numerator and denominator and cancels completely. This is why stopping distance on a level road depends only on initial speed and road conditions, not vehicle mass.

\textbf{Substitution:}
\[U_{s,i} = \frac{1}{2}(800)(0.10)^2 = \frac{1}{2}(800)(0.01) = 4.0 \text{ J}\]

\[\boxed{U_{s,i} = 4.0 \text{ J}}\]

\textbf{(b) Speed when spring reaches natural length:}

At natural length, $x = 0$, so $U_{s,f} = 0$. All elastic potential energy converts to kinetic energy.

\textbf{Conservation of energy:}
\[K_i + U_{s,i} = K_f + U_{s,f}\]

\[0 + 4.0 = \frac{1}{2}mv_f^2 + 0\]

\textbf{Solving for $v_f$:}
\[v_f = \sqrt{\frac{2(4.0)}{0.50}} = \sqrt{\frac{8.0}{0.50}} = \sqrt{16}\]

\[\boxed{v_f = 4.0 \text{ m/s}}\]

\textbf{(c) Speed when compression is 5.0 cm:}

At $x = 0.05$ m:
\[U_{s} = \frac{1}{2}(800)(0.05)^2 = \frac{1}{2}(800)(0.0025) = 1.0 \text{ J}\]

\textbf{Conservation of energy:}
\[K_i + U_{s,i} = K + U_s\]

\[0 + 4.0 = \frac{1}{2}mv^2 + 1.0\]

\[\frac{1}{2}(0.50)v^2 = 3.0\]

\[v = \sqrt{\frac{2(3.0)}{0.50}} = \sqrt{12}\]

\[\boxed{v = 3.46 \text{ m/s}}\]
\end{solution}

At maximum compression, the block momentarily stops ($v = 0$), so all kinetic energy is converted to elastic potential energy.

\[\boxed{x_{max} = 0.19 \text{ m} = 19 \text{ cm}}\]

\[K_i = K + U_s\]

\[9.0 = \frac{1}{2}(2.0)v^2 + 2.25\]

\[mgL = \frac{1}{2}mv_f^2\]

\[v_f = \sqrt{2gL} = \sqrt{2(9.8)(2.0)} = \sqrt{39.2}\]

\[\boxed{v_f = 6.26 \text{ m/s}}\]

\textbf{(b) Tension in string at lowest point:}

At the lowest point, the ball moves in a circular arc, so it has centripetal acceleration:

\[a_c = \frac{v^2}{L}\]

Applying Newton's second law in the radial direction (positive toward center, i.e., upward):

\[T - mg = m\frac{v^2}{L}\]

\[T = mg + m\frac{v^2}{L} = m\left(g + \frac{v^2}{L}\right)\]

\textbf{Substitution:}
\[T = (5.0)\left(9.8 + \frac{39.2}{2.0}\right) = (5.0)(9.8 + 19.6) = (5.0)(29.4)\]

\[\boxed{T = 147 \text{ N}}\]

\textbf{Note:} The tension is 3 times the weight! This is because the string must support the weight AND provide centripetal force.

\textbf{(c) Speed when string makes 30$^\circ$ with vertical:}

At angle $\theta = 30^\circ$, the height above the lowest point is:
\[h = L - L\cos\theta = L(1 - \cos\theta) = 2.0(1 - \cos 30^\circ) = 2.0(1 - 0.866) = 0.268 \text{ m}\]

\textbf{Conservation of energy:}
\[mgL = \frac{1}{2}mv^2 + mgh\]

\[v = \sqrt{2g(L - h)} = \sqrt{2gL\cos\theta} = \sqrt{2(9.8)(2.0)(0.866)}\]

\[v = \sqrt{33.9}\]

\[\boxed{v = 5.83 \text{ m/s}}\]
\end{solution}

\begin{solution}
\textbf{Solution H1.}

\textbf{Given:}
\begin{itemize}
    \item Mass: $m = 4.0$ kg
    \item Initial speed: $v_i = 8.0$ m/s
    \item Incline angle: $\theta = 25^\circ$
    \item Coefficient of kinetic friction: $\mu_k = 0.30$
\end{itemize}

\textbf{(a) Maximum distance up the incline:}

Let $d$ be the distance traveled up the incline. The height gained is $h = d\sin\theta$.

\textbf{Energy conservation with friction:}
\[K_i = U_{g,f} + \Delta E_{th}\]

\[\frac{1}{2}mv_i^2 = mgh + f_k d\]

The friction force is:
\[f_k = \mu_k N = \mu_k mg\cos\theta\]

Substituting:
\[\frac{1}{2}mv_i^2 = mgd\sin\theta + \mu_k mg\cos\theta \cdot d\]

\[\frac{1}{2}v_i^2 = gd(\sin\theta + \mu_k\cos\theta)\]

\[d = \frac{v_i^2}{2g(\sin\theta + \mu_k\cos\theta)}\]

\textbf{Substitution:}
\[\sin(25^\circ) = 0.423, \quad \cos(25^\circ) = 0.906\]

\[d = \frac{(8.0)^2}{2(9.8)(0.423 + 0.30 \times 0.906)}\]

\[d = \frac{64}{2(9.8)(0.423 + 0.272)} = \frac{64}{2(9.8)(0.695)}\]

\[d = \frac{64}{13.62}\]

\[\boxed{d = 4.70 \text{ m}}\]

\textbf{(b) Speed when returning to starting point:}

On the way down, friction still opposes motion (now pointing up the incline), so more energy is lost.

\textbf{Total energy dissipated (up and down):}
\[\Delta E_{th,total} = f_k \cdot (2d) = \mu_k mg\cos\theta \cdot 2d\]

\textbf{Energy equation:}
\[\frac{1}{2}mv_i^2 = \frac{1}{2}mv_f^2 + \Delta E_{th,total}\]

\[\frac{1}{2}v_i^2 = \frac{1}{2}v_f^2 + \mu_k g\cos\theta \cdot 2d\]

\[v_f^2 = v_i^2 - 4\mu_k gd\cos\theta\]

\textbf{Substitution:}
\[v_f^2 = (8.0)^2 - 4(0.30)(9.8)(4.70)(0.906)\]

\[v_f^2 = 64 - 4(0.30)(9.8)(4.70)(0.906)\]

\[v_f^2 = 64 - 50.0 = 14.0\]

\[v_f = \sqrt{14.0}\]

\[\boxed{v_f = 3.74 \text{ m/s}}\]

\textbf{(c) Total energy lost to friction:}

\[\Delta E_{th,total} = \frac{1}{2}m(v_i^2 - v_f^2) = \frac{1}{2}(4.0)(64 - 14) = \frac{1}{2}(4.0)(50)\]

\[\boxed{\Delta E_{th,total} = 100 \text{ J}}\]
\end{solution}

\begin{solution}
\textbf{Solution H2.}

\textbf{Given:}
\begin{itemize}
    \item Mass: $m = 0.25$ kg
    \item Spring constant: $k = 400$ N/m
    \item Compression: $x = 0.10$ m
    \item Incline angle: $\theta = 35^\circ$
    \item Coefficient of kinetic friction: $\mu_k = 0.20$
\end{itemize}

\textbf{(a) Initial elastic potential energy:}

\[U_s = \frac{1}{2}kx^2 = \frac{1}{2}(400)(0.10)^2 = \frac{1}{2}(400)(0.01)\]

\[\boxed{U_s = 2.0 \text{ J}}\]

\textbf{(b) Maximum distance up the incline:}

Let $d$ be the distance traveled. The spring energy converts to gravitational potential energy and thermal energy:

\[\frac{1}{2}kx^2 = mgd\sin\theta + \mu_k mg\cos\theta \cdot d\]

\[U_s = mgd(\sin\theta + \mu_k\cos\theta)\]

\[d = \frac{U_s}{mg(\sin\theta + \mu_k\cos\theta)}\]

\textbf{Substitution:}
\[\sin(35^\circ) = 0.574, \quad \cos(35^\circ) = 0.819\]

\[d = \frac{2.0}{(0.25)(9.8)(0.574 + 0.20 \times 0.819)}\]

\[d = \frac{2.0}{(2.45)(0.574 + 0.164)} = \frac{2.0}{(2.45)(0.738)}\]

\[d = \frac{2.0}{1.808}\]

\[\boxed{d = 1.11 \text{ m}}\]

\textbf{(c) Does the block return to the spring?}

For the block to slide back down, the gravitational component must exceed the maximum static friction:

\[mg\sin\theta > \mu_s mg\cos\theta\]

\[\tan\theta > \mu_s\]

Assuming $\mu_s \approx \mu_k = 0.20$:
\[\tan(35^\circ) = 0.700 > 0.20 \checkmark\]

\textbf{Yes, the block will slide back down.}

\textbf{(d) Speed when returning to spring:}

On the way down, gravitational PE converts to KE and thermal energy:

\[mgd\sin\theta = \frac{1}{2}mv_f^2 + \mu_k mg\cos\theta \cdot d\]

\[v_f^2 = 2gd(\sin\theta - \mu_k\cos\theta)\]

\textbf{Substitution:}
\[v_f^2 = 2(9.8)(1.11)(0.574 - 0.20 \times 0.819)\]

\[v_f^2 = 2(9.8)(1.11)(0.574 - 0.164) = 2(9.8)(1.11)(0.410)\]

\[v_f^2 = 8.91\]

\[\boxed{v_f = 2.98 \text{ m/s}}\]

\begin{solution}
\textbf{Solution H3.}

\textbf{Given:}
\begin{itemize}
    \item Mass: $m = 1500$ kg
    \item Initial speed: $v_i = 25$ m/s
    \item Distance to obstacle: $D = 50$ m
    \item Reaction time: $t_r = 0.80$ s
    \item Braking deceleration: $a = 8.0$ m/s$^2$ (magnitude)
\end{itemize}

During reaction time, the car travels at constant velocity:

\[d_r = v_i \cdot t_r = (25)(0.80)\]

\[\boxed{d_r = 20 \text{ m}}\]

\textbf{(b) Kinetic energy when braking begins:}

\[K = \frac{1}{2}mv_i^2 = \frac{1}{2}(1500)(25)^2 = \frac{1}{2}(1500)(625)\]

\[\boxed{K = 468{,}750 \text{ J} = 469 \text{ kJ}}\]

\textbf{(c) Stopping distance using energy:}

The work done by braking force equals the kinetic energy:

\[W = F \cdot d_s = ma \cdot d_s\]

\[K = ma \cdot d_s\]

\[d_s = \frac{K}{ma} = \frac{\frac{1}{2}mv^2}{ma} = \frac{v^2}{2a}\]

\textbf{Substitution:}
\[d_s = \frac{(25)^2}{2(8.0)} = \frac{625}{16}\]

\[\boxed{d_s = 39.1 \text{ m}}\]

\textbf{(d) Does the car stop in time?}

Total distance needed:
\[d_{total} = d_r + d_s = 20 + 39.1 = 59.1 \text{ m}\]

Distance to obstacle: 50 m

\[59.1 \text{ m} > 50 \text{ m}\]

\[\boxed{\text{No, the car does NOT stop in time. It overshoots by } 9.1 \text{ m.}}\]

\textbf{(e) At 40 m/s:}

Reaction distance:
\[d_r = (40)(0.80) = 32 \text{ m}\]

Stopping distance:
\[d_s = \frac{(40)^2}{2(8.0)} = \frac{1600}{16} = 100 \text{ m}\]

Total:
\[d_{total} = 32 + 100 = 132 \text{ m}\]

\[\boxed{\text{No, at 40 m/s the car would overshoot by } 82 \text{ m.}}\]

\textbf{Key insight:} Doubling the speed quadruples the stopping distance (since $d_s \propto v^2$), making high-speed driving much more dangerous.
\end{solution}

\section*{Summary of Key Formulas}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black, title=\textbf{Essential Energy Equations}]
\begin{align*}
\textbf{Work:} \quad & W = Fd\cos\theta = \vec{F} \cdot \vec{d} \\[8pt]
\textbf{Kinetic Energy:} \quad & K = \frac{1}{2}mv^2 \\[8pt]
\textbf{Work-Energy Theorem:} \quad & W_{net} = \Delta K = K_f - K_i \\[8pt]
\textbf{Gravitational PE:} \quad & U_g = mgy \\[8pt]
\textbf{Elastic PE:} \quad & U_s = \frac{1}{2}kx^2 \\[8pt]
\textbf{Conservation of Energy:} \quad & K_i + U_i + W_{ext} = K_f + U_f + \Delta E_{th} \\[8pt]
\textbf{Thermal Energy (friction):} \quad & \Delta E_{th} = f_k \cdot d \\[8pt]
\textbf{Power:} \quad & P = \frac{W}{\Delta t} = \frac{dE}{dt} = Fv\cos\theta
\end{align*}
\end{tcolorbox}
