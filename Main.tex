\section{Introduction to Torque (Section 7.3)}

\begin{diagbox}{1. Introduction to Torque}

\textbf{Definition:} \emph{Torque} (symbol $\tau$) is the rotational analog of force.  It measures how effectively a force causes an object to rotate about a pivot point.

\vspace{0.3cm}
\textbf{General formula:}
\[
  \tau = r\,F\sin\theta
\]
where $r$ is the distance from the pivot to the point of force application, $F$ is the force magnitude, and $\theta$ is the angle between $\vec{r}$ and $\vec{F}$.

\vspace{0.3cm}
\textbf{Units:} newton-meters (N$\cdot$m).

\vspace{0.3cm}
\textbf{Sign convention:}
\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Counterclockwise (CCW)} torques are \textbf{positive} ($+$).
  \item \textbf{Clockwise (CW)} torques are \textbf{negative} ($-$).
\end{itemize}

\end{diagbox}

\subsection{Types of Torque Calculations}

\begin{diagbox}{1.1 Types of Torque Calculations}

\textbf{Case 1 --- Perpendicular force} ($\theta = 90^\circ$):
\[
  \tau = rF
\]
\emph{Example:} A 20\,N force perpendicular to a 0.3\,m wrench gives $\tau = (0.3)(20) = 6$\,N$\cdot$m.

\vspace{0.4cm}
\textbf{Case 2 --- Force at an angle $\theta$:}
\[
  \tau = rF\sin\theta
\]
\emph{Example:} A 20\,N force at $30^\circ$ on a 0.3\,m wrench gives $\tau = (0.3)(20)\sin 30^\circ = 3$\,N$\cdot$m.

\vspace{0.4cm}
\textbf{Case 3 --- Using the moment arm:}
\[
  \tau = d_\perp F
\]
where $d_\perp = r\sin\theta$ is the perpendicular distance from the pivot to the line of action.

\vspace{0.4cm}
\textbf{Case 4 --- Net torque:}
\[
  \tau_{\rm net} = \sum_i \tau_i
\]
Each torque is signed before summing.

\end{diagbox}

\subsection{Key Properties of Torque}

\begin{diagbox}{1.2 Key Properties of Torque}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Torque depends on the pivot point.}  The same force produces different torques for different pivot choices.
  \item \textbf{Maximum torque} is achieved when the force is applied perpendicular to the lever arm ($\theta = 90^\circ$).
  \item \textbf{Zero torque} results when $\theta = 0^\circ$ or $180^\circ$.
  \item \textbf{Longer lever arm $\Rightarrow$ larger torque} for the same force.
  \item \textbf{Newton's second law for rotation:}
        \[
          \tau_{\rm net} = I\alpha
        \]
        where $I$ is the moment of inertia and $\alpha$ is the angular acceleration.
\end{enumerate}

\end{diagbox}

\newpage

\section{Gravitational Torque and the Center of Gravity (Section 7.4)}

\begin{diagbox}{2. Gravitational Torque and the Center of Gravity}

The entire gravitational force on an extended object can be treated as acting at a single point: the \textbf{center of gravity} (CG).

\vspace{0.3cm}
\textbf{Gravitational torque} about a pivot at horizontal position $x_0$:
\[
  \tau_{\rm grav} = (x_{\rm CG} - x_0)\,M g
\]
where $M$ is the total mass and $x_{\rm CG}$ is the CG's horizontal coordinate.

\end{diagbox}

\subsection{Finding the Center of Gravity}

\begin{diagbox}{2.1 Finding the Center of Gravity}

For a system of particles with masses $m_1, m_2, \ldots$ at positions $x_1, x_2, \ldots$:
\[
  x_{\rm CG} = \frac{\sum_i m_i\,x_i}{\sum_i m_i}
  \qquad
  y_{\rm CG} = \frac{\sum_i m_i\,y_i}{\sum_i m_i}
\]

\textbf{Worked example:}
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
\textbf{Mass} & \textbf{Value (kg)} & \textbf{Position (m)} \\
\hline
$m_1$ & 2.0 & 0.0 \\
$m_2$ & 3.0 & 1.0 \\
$m_3$ & 1.0 & 2.0 \\
\hline
\end{tabular}
\end{center}
\[
  x_{\rm CG} = \frac{(2.0)(0.0)+(3.0)(1.0)+(1.0)(2.0)}{6.0} = \frac{5}{6} \approx 0.83\text{ m}
\]

\textbf{Tip:} For a uniform symmetric object (rod, disk), the CG is at the geometric center.

\end{diagbox}

\subsection{Key Properties of Center of Gravity}

\begin{diagbox}{2.2 Key Properties of Center of Gravity}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Unique point:} Every object has exactly one center of gravity (in a uniform field).
  \item \textbf{Need not lie within the object:} For a hollow ring or boomerang, the CG is in empty space.
  \item \textbf{Stability criterion:} An object supported at a single point is in equilibrium when the support is directly below (or above) the CG.
  \item \textbf{CG vs.\ CM:} In a uniform gravitational field, the center of gravity and center of mass coincide.
  \item \textbf{Composite objects:} The CG of a composite object can be found by treating each part as a point mass at its own CG.
\end{enumerate}

\end{diagbox}

\section{Torque and Static Equilibrium (Section 8.1)}

\begin{diagbox}{3. Torque and Static Equilibrium}

An object is in \textbf{static equilibrium} when both translational and rotational conditions hold:

\vspace{0.3cm}
\textbf{Condition 1 --- Translational equilibrium:}
\[
  \sum F_x = 0 \quad\text{and}\quad \sum F_y = 0
\]

\textbf{Condition 2 --- Rotational equilibrium:}
\[
  \sum \tau = 0 \quad\text{(about any chosen pivot)}
\]

\vspace{0.4cm}
\textbf{Problem-solving strategy:}
\begin{enumerate}[leftmargin=1.5em]
  \item Draw a free-body diagram with all forces and their points of application.
  \item Choose a pivot strategically (where an unknown force acts, to eliminate it).
  \item Write $\sum\tau = 0$ about the pivot.
  \item Write $\sum F_x = 0$ and $\sum F_y = 0$.
  \item Solve the system of equations.
\end{enumerate}

\vspace{0.4cm}
\textbf{Worked example:} A uniform 4.0-m, 10-kg beam rests on a pivot at its left end.  A 30-kg block sits 3.0\,m from the left end.  Find upward force $F$ at the right end.

Taking the pivot at the left end:
\[
  F(4.0) - (10)(9.8)(2.0) - (30)(9.8)(3.0) = 0
  \;\Rightarrow\;
  F = \frac{196+882}{4} \approx 270\text{ N}
\]

\end{diagbox}

\section{Stability and Balance (Section 8.2)}

\begin{diagbox}{4. Stability and Balance}

\textbf{Stable equilibrium:}  A small displacement raises the CG and produces a restoring torque.

\textbf{Unstable equilibrium:}  A small displacement lowers the CG and causes further departure.

\textbf{Neutral equilibrium:}  A displacement does not change the CG height; no restoring or destabilizing torque.

\vspace{0.4cm}
\textbf{Tipping condition:}  An object tips when its CG moves outside the \emph{base of support}.

\end{diagbox}

\subsection{Factors Affecting Stability}

\begin{diagbox}{4.1 Factors Affecting Stability}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Width of the base of support:} A wider base makes tipping less likely.
  \item \textbf{Height of the center of gravity:} A lower CG increases stability.
  \item \textbf{Position of the CG over the base:} Maximum stability when CG is above the center of the base.
  \item \textbf{Dynamic stability:} Moving objects can maintain balance through gyroscopic effects.
\end{enumerate}

\vspace{0.4cm}
\textbf{Example:} A 1.2-m-tall crate (CG at 0.6\,m, base width 0.4\,m) tips when tilted at angle $\theta$:
\[
  \tan\theta_{\rm tip} = \frac{0.2\text{ m}}{0.6\text{ m}} \approx 0.333
  \;\Rightarrow\;
  \theta_{\rm tip} \approx 18.4^\circ
\]

\end{diagbox}

\section{Springs and Hooke's Law (Section 8.3)}

\begin{diagbox}{5. Springs and Hooke's Law}

\textbf{Hooke's Law:}
\[
  F_{\rm sp} = -k\,x
\]
\begin{itemize}[leftmargin=1.5em]
  \item $k$ = spring constant (N/m)
  \item $x$ = displacement from natural (equilibrium) length
  \item Negative sign: restoring force opposes displacement
\end{itemize}

\textbf{Example:} Spring with $k = 200$\,N/m stretched 0.05\,m: $|F_{\rm sp}| = (200)(0.05) = 10$\,N.

\begin{warnbox}
Hooke's Law holds only within the \textbf{elastic limit}.  Beyond this limit the spring is permanently deformed.
\end{warnbox}

\end{diagbox}

\subsection{Graphical Interpretation of Hooke's Law}

\begin{diagbox}{5.1 Graphical Interpretation of Hooke's Law}

On a graph of applied force $F$ vs.\ displacement $x$:
\begin{itemize}[leftmargin=1.5em]
  \item The relationship is \textbf{linear} for an ideal spring.
  \item The \textbf{slope} equals the spring constant $k$.
  \item The \textbf{area under the curve} equals the elastic potential energy stored: $U_{\rm sp} = \tfrac{1}{2}kx^2$.
  \item Beyond the elastic limit, the graph curves.
\end{itemize}

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|l|l|}
\hline
\textbf{Graph feature} & \textbf{Physical meaning} \\
\hline
Slope $= k$ & Spring stiffness (N/m) \\
Area under curve to $x$ & Work done / elastic PE = $\tfrac{1}{2}kx^2$ \\
Linear region & Valid range of Hooke's Law \\
Deviation from linearity & Beyond elastic limit \\
\hline
\end{tabular}
\end{center}

\end{diagbox}

\section{Forces and Torques in the Body (Section 8.5) --- \textsc{Bio}}

\begin{diagbox}{6. Forces and Torques in the Body (BIO)}

The human musculoskeletal system functions as a collection of \textbf{levers}: rigid bones rotate about pivot joints.

\vspace{0.3cm}
\textbf{Lever elements:}
\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Effort} $F_E$: force applied by muscle.
  \item \textbf{Load} $F_L$: force being overcome.
  \item \textbf{Fulcrum}: the joint (pivot point).
\end{itemize}

\textbf{Mechanical advantage:} $\displaystyle\text{MA} = \frac{d_E}{d_L} = \frac{F_L}{F_E}$

\end{diagbox}

\subsection{Types of Levers in the Body}

\begin{diagbox}{6.1 Types of Levers in the Body}

\begin{center}
\renewcommand{\arraystretch}{1.8}
\begin{tabular}{|c|p{3cm}|p{3.5cm}|p{4cm}|}
\hline
\textbf{Class} & \textbf{Arrangement} & \textbf{Body Example} & \textbf{Key Feature} \\
\hline
\textbf{1st Class} & Fulcrum between effort and load & Head nodding: neck muscles (effort), skull weight (load), atlas vertebra (fulcrum) & Can have MA $>1$ or $<1$; changes force direction \\
\hline
\textbf{2nd Class} & Load between fulcrum and effort & Rising on tiptoe: ball of foot (fulcrum), body weight (load), calf/Achilles tendon (effort) & MA $>1$; effort arm $>$ load arm \\
\hline
\textbf{3rd Class} & Effort between fulcrum and load & Forearm curl: elbow (fulcrum), bicep $\approx$5\,cm from joint (effort), hand/weight $\approx$35\,cm (load) & MA $<1$; trades force for speed and range \\
\hline
\end{tabular}
\end{center}

\vspace{0.4cm}
\textbf{Why muscles exert large forces:}  Most body levers are 3rd-class (MA $\approx 0.14$), so the muscle must exert about 7$\times$ the load force.

\end{diagbox}

\section{The Basic Energy Model (Section 10.1)}

\begin{diagbox}{7. The Basic Energy Model}

\textbf{Energy} is the capacity of a system to do work or cause change.

\vspace{0.3cm}
\textbf{Mechanical energy forms:}
\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Kinetic energy} $K = \tfrac{1}{2}mv^2$
  \item \textbf{Gravitational PE} $U_g = mgy$
  \item \textbf{Elastic PE} $U_{\rm sp} = \tfrac{1}{2}kx^2$
\end{itemize}

\textbf{Non-mechanical:} Thermal energy $E_{\rm th}$ (microscopic KE/PE of atoms).

\textbf{Master energy equation (preview):}
\[
  K_f + U_f + \Delta E_{\rm th} = K_i + U_i + W_{\rm ext}
\]

\end{diagbox}

\subsection{System and Environment}

\begin{diagbox}{7.1 System and Environment}

\begin{itemize}[leftmargin=1.5em]
  \item \textbf{System:} the objects under study.
  \item \textbf{Environment:} everything outside the system.
  \item Energy transfers via \emph{work} (mechanical) or \emph{heat} (thermal).
\end{itemize}

\textbf{Isolated system:} no energy enters or leaves; total energy is conserved.

\textbf{Choosing the system wisely} simplifies problems: including interacting objects avoids internal forces in the energy equation.

\end{diagbox}

\section{Work (Section 10.2)}

\begin{diagbox}{8. Work}

\textbf{Work} is energy transferred by a force over a displacement:
\[
  W = F\,d\cos\theta
\]
where $\theta$ is the angle between $\vec{F}$ and $\vec{d}$.

\textbf{Sign:}
\begin{itemize}[leftmargin=1.5em]
  \item $W > 0$: force component in direction of motion (energy added).
  \item $W < 0$: force component opposite to motion (energy removed).
  \item $W = 0$: force perpendicular to motion.
\end{itemize}

\textbf{Units:} joules (J); $1\,\text{J} = 1\,\text{N}\cdot\text{m}$.

\textbf{Example:} 50-N force at $37^\circ$ over 4.0\,m: $W = (50)(4.0)\cos37^\circ \approx 160$\,J.

\end{diagbox}

\subsection{Work Done by Gravity on an Incline}

\begin{diagbox}{8.1 Work Done by Gravity on an Incline}

\[
  W_{\rm grav} = mg\,d\sin\alpha = mg\,\Delta y
\]
where $\Delta y = d\sin\alpha$ is the vertical drop (positive downward).

\textbf{Key insight:} Work by gravity depends only on vertical displacement, not path.  Gravity is a \emph{conservative} force.

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|l|c|}
\hline
\textbf{Motion} & \textbf{Work by gravity} \\
\hline
Down the incline & $+mgd\sin\alpha > 0$ \\
Up the incline   & $-mgd\sin\alpha < 0$ \\
Horizontal       & $0$ \\
\hline
\end{tabular}
\end{center}

\end{diagbox}

\section{Kinetic Energy (Section 10.3)}

\begin{diagbox}{9. Kinetic Energy}

\textbf{Kinetic energy:}
\[
  K = \tfrac{1}{2}mv^2
\]
\textbf{Units:} joules (J).  Always $K \geq 0$.

\end{diagbox}

\subsection{Understanding Kinetic Energy}

\begin{diagbox}{9.1 Understanding Kinetic Energy}

\begin{itemize}[leftmargin=1.5em]
  \item KE is a \emph{scalar}, always non-negative.
  \item KE depends on the reference frame.
  \item \textbf{Doubling the mass} doubles the KE.
  \item \textbf{Doubling the speed} \emph{quadruples} the KE.
\end{itemize}

\textbf{Example:} 2-kg ball at 3\,m/s vs.\ 6\,m/s:
\[
  K_1 = \tfrac{1}{2}(2)(9) = 9\text{ J}; \quad K_2 = \tfrac{1}{2}(2)(36) = 36\text{ J} = 4K_1
\]

\end{diagbox}

\subsection{The \texorpdfstring{$v^2$}{v\texttwosuperior} Dependence: Why It Matters}

\begin{diagbox}{9.2 The $v^2$ Dependence: Why It Matters}

\begin{itemize}[leftmargin=1.5em]
  \item \textbf{Braking distance} $\propto v^2$: doubling speed quadruples stopping distance.
  \item \textbf{Wind power} scales as $v^3$.
  \item \textbf{Crash energy} at 60\,mph is four times that at 30\,mph.
\end{itemize}

\begin{warnbox}
\textbf{Common mistake:} Writing $K = \tfrac{1}{2}mv$ (forgetting to square $v$).  Check units: $\text{kg}\cdot(\text{m/s})^2 = \text{J}$.
\end{warnbox}

\end{diagbox}

\subsection{The Work--Energy Theorem}

\begin{diagbox}{9.3 The Work--Energy Theorem}

\[
  W_{\rm net} = \Delta K = \tfrac{1}{2}mv_f^2 - \tfrac{1}{2}mv_i^2
\]

\textbf{Derivation (1D, constant force):}  From kinematics $v_f^2 = v_i^2 + 2a\Delta x$, multiply by $\tfrac{1}{2}m$:
\[
  \tfrac{1}{2}mv_f^2 - \tfrac{1}{2}mv_i^2 = ma\,\Delta x = F_{\rm net}\,\Delta x = W_{\rm net}
\]

\textbf{Note:} Only the \emph{net} work from all forces equals $\Delta K$.

\end{diagbox}

\subsection{Kinetic Energy and the Work--Energy Theorem on Inclines}

\begin{diagbox}{9.4 Kinetic Energy and the Work--Energy Theorem on Inclines}

\textbf{Setup:} Mass $m$ slides distance $d$ down frictionless incline at angle $\alpha$.

Work by gravity: $W_g = mgd\sin\alpha$; normal force does zero work.

\[
  W_{\rm net} = mgd\sin\alpha = \tfrac{1}{2}mv_f^2 - \tfrac{1}{2}mv_i^2
\]

Starting from rest ($v_i = 0$): $v_f = \sqrt{2gd\sin\alpha}$.

\textbf{Example:} 5-kg block from rest down 4.0-m, $30^\circ$ frictionless ramp:
\[
  v_f = \sqrt{2(9.8)(4.0)(0.5)} = \sqrt{39.2} \approx 6.3\text{ m/s}
\]

\end{diagbox}

\subsection{Multiple Objects and Kinetic Energy}

\begin{diagbox}{9.5 Multiple Objects and Kinetic Energy}

Total KE of a system:
\[
  K_{\rm total} = \sum_i \tfrac{1}{2}m_i v_i^2
\]

For a connected system: $W_{\rm net,\,ext} = \Delta K_{\rm total}$ (internal forces contribute zero net work).

\textbf{Example (Atwood machine):} $m_1 = 3$\,kg, $m_2 = 5$\,kg, released from rest; each moves 1.0\,m:
\[
  W_{\rm net} = (m_2 - m_1)g\,d = (2)(9.8)(1.0) = 19.6\text{ J}
\]
\[
  \tfrac{1}{2}(8)v^2 = 19.6\text{ J} \;\Rightarrow\; v = \sqrt{4.9} \approx 2.2\text{ m/s}
\]

\end{diagbox}

\subsection{Common Mistakes with Kinetic Energy}

\begin{diagbox}{9.6 Common Mistakes with Kinetic Energy}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Forgetting to square speed:} $K \neq \tfrac{1}{2}mv$; use $K = \tfrac{1}{2}mv^2$.
  \item \textbf{Treating KE as a vector:} KE is a scalar, always $\geq 0$.
  \item \textbf{Applying theorem to one force:} Must sum work from \emph{all} forces.
  \item \textbf{Ignoring friction:} Friction does negative work and reduces $\Delta K$.
  \item \textbf{Confusing KE and momentum:} $K = \tfrac{1}{2}mv^2 \neq p = mv$.
  \item \textbf{Wrong units:} Speed must be in m/s, mass in kg, for KE in joules.
\end{enumerate}

\end{diagbox}

\subsection{Kinetic Energy in Context: Real-World Numbers}

\begin{diagbox}{9.7 Kinetic Energy in Context: Real-World Numbers}

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|l|c|c|c|}
\hline
\textbf{Object} & \textbf{Mass (kg)} & \textbf{Speed (m/s)} & \textbf{KE (J)} \\
\hline
Baseball (pitch) & 0.145 & 40 & 116 \\
Person walking & 70 & 1.4 & 69 \\
Car at 60\,mph & 1200 & 26.8 & $4.3\times10^5$ \\
Bullet & 0.010 & 900 & 4050 \\
Jumbo jet & $3.6\times10^5$ & 250 & $1.1\times10^{10}$ \\
\hline
\end{tabular}
\end{center}

The car's KE at highway speed is $\sim$100$\times$ a baseball pitch, despite being much slower, because mass matters too.

\end{diagbox}

\subsection{Practice Problems: Kinetic Energy (Section 10.3)}

\begin{planbox}

\begin{enumerate}[leftmargin=1.5em, label=\textbf{P\arabic*.}]

  \item A 2.0-kg ball is thrown horizontally at 15\,m/s.  What is its kinetic energy?

  \item A 1500-kg car has $K = 2.7\times10^5$\,J.  What is its speed?

  \item A 0.5-kg hockey puck slides at 8.0\,m/s and is stopped by a 2.0-N friction force.  Find the stopping distance using the Work--Energy Theorem.

  \item Two objects have the same kinetic energy.  Object A has mass $m$ and speed $v$.  Object B has mass $4m$.  What is the speed of object B?

  \item A 10-kg block starts from rest and slides 5.0\,m down a frictionless $25^\circ$ incline.  Find its final speed.

  \item A net force of 40\,N acts on an 8.0-kg object over 3.0\,m.  If the initial speed is 2.0\,m/s, find the final speed.

  \item A bullet of mass 0.012\,kg embeds in a 1.0-kg block (initially at rest).  The block then slides 0.80\,m with $\mu_k = 0.25$.  Find the speed of the bullet--block system just after impact.

  \item Compare the KE of (a) a 1500-kg car at 20\,m/s and (b) a 3000-kg truck at 10\,m/s.  Which is larger?

\end{enumerate}

\end{planbox}

\subsection{Solutions: Kinetic Energy (Section 10.3)}

\begin{keybox}

\begin{enumerate}[leftmargin=1.5em, label=\textbf{P\arabic*.}]

  \item \textbf{$K = 225$\,J.} \quad $K = \tfrac{1}{2}(2.0)(15)^2 = 225$\,J.

  \item \textbf{$v \approx 19.0$\,m/s.} \quad $v = \sqrt{2K/m} = \sqrt{2(2.7\times10^5)/1500} = \sqrt{360} \approx 19.0$\,m/s.

  \item \textbf{$d = 8.0$\,m.} \quad $K_i = f_k\,d \Rightarrow \tfrac{1}{2}(0.5)(8)^2 = (2.0)d \Rightarrow d = 16/2 = 8.0$\,m.

  \item \textbf{$v_B = v/2$.} \quad $\tfrac{1}{2}mv^2 = \tfrac{1}{2}(4m)v_B^2 \Rightarrow v_B^2 = v^2/4 \Rightarrow v_B = v/2$.

  \item \textbf{$v_f \approx 6.4$\,m/s.} \quad $W_g = (10)(9.8)(5.0)\sin25^\circ \approx 207$\,J; $v_f = \sqrt{2(207)/10} \approx 6.4$\,m/s.

  \item \textbf{$v_f \approx 5.8$\,m/s.} \quad $\tfrac{1}{2}mv_f^2 = \tfrac{1}{2}(8)(4) + (40)(3) = 16+120 = 136$\,J; $v_f = \sqrt{2(136)/8} = \sqrt{34} \approx 5.8$\,m/s.

  \item \textbf{$v \approx 1.97$\,m/s.} \quad Friction work: $W_f = -(0.25)(1.0)(9.8)(0.80) = -1.96$\,J. Setting $\tfrac{1}{2}(1.012)v^2 = 1.96$\,J: $v \approx 1.97$\,m/s.

  \item \textbf{Car KE is larger.} \quad Car: $K = \tfrac{1}{2}(1500)(400) = 3.0\times10^5$\,J. Truck: $K = \tfrac{1}{2}(3000)(100) = 1.5\times10^5$\,J.  The car's KE is twice the truck's.

\end{enumerate}

\end{keybox}

\section{Potential Energy (Section 10.4)}

\begin{diagbox}{10. Potential Energy}

\textbf{Potential energy} (PE or $U$) is energy stored in a system due to position or configuration within a \emph{conservative} force field.

\end{diagbox}

\subsection{Gravitational Potential Energy}

\begin{diagbox}{10.1 Gravitational Potential Energy}

Near Earth's surface:
\[
  U_g = mgy
\]
\[
  \Delta U_g = mg\,\Delta y = mg(y_f - y_i)
\]
\begin{itemize}[leftmargin=1.5em]
  \item $\Delta U_g > 0$: object moved upward.
  \item $\Delta U_g < 0$: object moved downward.
\end{itemize}

\end{diagbox}

\subsubsection{Choosing the Reference Level}

\begin{diagbox}{10.1.1 Choosing the Reference Level}

The reference level ($y = 0$) is arbitrary:
\begin{itemize}[leftmargin=1.5em]
  \item Common choices: ground level, lowest point of motion, initial position.
  \item \textbf{Only $\Delta U_g$ matters}; the reference level cancels in the final answer.
\end{itemize}

\textbf{Example:} 3-kg ball 2.0\,m above floor. Reference at floor: $U_g = (3)(9.8)(2.0) = 58.8$\,J. Reference at ball: $U_g = 0$.  Both give $\Delta U_g = -58.8$\,J when ball reaches floor.

\end{diagbox}

\subsubsection{Gravitational PE on Inclines}

\begin{diagbox}{10.1.2 Gravitational PE on Inclines}

Object at distance $s$ along a ramp of angle $\alpha$:
\[
  U_g = mgs\sin\alpha = mgh \quad\text{where } h = s\sin\alpha
\]
PE depends only on vertical height $h$, not on path length along ramp.

\textbf{Example:} 5-kg crate 3.0\,m up a $20^\circ$ ramp:
\[
  h = 3.0\sin20^\circ = 1.03\text{ m}, \quad U_g = (5)(9.8)(1.03) \approx 50.3\text{ J}
\]

\end{diagbox}

\subsection{Elastic (Spring) Potential Energy}

\begin{diagbox}{10.2 Elastic (Spring) Potential Energy}

\[
  U_{\rm sp} = \tfrac{1}{2}kx^2
\]
\begin{itemize}[leftmargin=1.5em]
  \item Always $U_{\rm sp} \geq 0$ (compressed or stretched).
  \item Maximum at maximum compression/extension; zero at natural length.
\end{itemize}

\textbf{Example:} $k = 500$\,N/m, compressed 0.10\,m: $U_{\rm sp} = \tfrac{1}{2}(500)(0.01) = 2.5$\,J.

\end{diagbox}

\subsection{Relationship Between Work and Potential Energy}

\begin{diagbox}{10.3 Relationship Between Work and Potential Energy}

For any \textbf{conservative force}:
\[
  W_c = -\Delta U
\]
Work done by a conservative force equals the \emph{decrease} in PE.

\textbf{Gravity:} $W_{\rm grav} = -\Delta U_g = mg(y_i - y_f)$

\textbf{Spring:} $W_{\rm sp} = -\Delta U_{\rm sp}$

\begin{warnbox}
Only conservative forces have associated PE.  Friction is non-conservative --- it permanently converts mechanical energy to thermal energy.
\end{warnbox}

\end{diagbox}

\newpage

\section{Thermal Energy (Section 10.5)}

\begin{diagbox}{11. Thermal Energy}

\textbf{Thermal energy} $E_{\rm th}$: microscopic KE and PE of atoms/molecules (manifests as temperature).

\textbf{From friction:}
\[
  \Delta E_{\rm th} = f_k\,d \quad\text{where } f_k = \mu_k\,n
\]

This energy is permanently converted from mechanical energy and is not recoverable.

\end{diagbox}

\subsection{Why Thermal Energy Is Different}

\begin{diagbox}{11.1 Why Thermal Energy Is Different}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Irreversible:} Friction converts ordered mechanical energy to disordered thermal energy.
  \item \textbf{Not a PE:} $E_{\rm th}$ cannot be stored and recovered like $U_g$ or $U_{\rm sp}$.
  \item \textbf{Always positive:} $\Delta E_{\rm th} = f_k d \geq 0$; friction never adds mechanical energy.
  \item \textbf{Second Law:} Irreversibility reflects increasing entropy.
\end{enumerate}

\begin{warnbox}
With friction, \emph{mechanical energy is not conserved}:
\[
  K_f + U_f = K_i + U_i - \Delta E_{\rm th}
\]
Total energy (mechanical + thermal) is still conserved.
\end{warnbox}

\end{diagbox}

\subsection{Thermal Energy and Stopping Distance}

\begin{diagbox}{11.2 Thermal Energy and Stopping Distance}

A car of mass $m$ braking from $v_0$ to rest:
\[
  \Delta E_{\rm th} = K_i = \tfrac{1}{2}mv_0^2
\]
Also $\Delta E_{\rm th} = \mu_k\,mg\,d$, so:
\[
  d = \frac{v_0^2}{2\mu_k g}
\]

\textbf{Stopping distance scales as $v_0^2$:} doubling speed quadruples stopping distance.

\textbf{Example:} $\mu_k = 0.70$, $v_0 = 25$\,m/s:
\[
  d = \frac{625}{2(0.70)(9.8)} \approx 45.6\text{ m}
\]

\end{diagbox}

\newpage

\section{Conservation of Energy (Section 10.6)}

\begin{diagbox}{12. Conservation of Energy}

\textbf{Law of Conservation of Energy:} The total energy of an isolated system is constant.

\[
  \Delta K + \Delta U_g + \Delta U_{\rm sp} + \Delta E_{\rm th} = W_{\rm ext}
\]

\end{diagbox}

\subsection{The Master Energy Equation}

\begin{diagbox}{12.1 The Master Energy Equation}

\[
  \boxed{K_f + U_f + \Delta E_{\rm th} = K_i + U_i + W_{\rm ext}}
\]

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|l|l|}
\hline
\textbf{Term} & \textbf{Meaning} \\
\hline
$K_f$ & Final kinetic energy \\
$U_f$ & Final potential energy ($U_g + U_{\rm sp}$) \\
$\Delta E_{\rm th}$ & Thermal energy from friction ($\geq 0$) \\
$K_i$ & Initial kinetic energy \\
$U_i$ & Initial potential energy \\
$W_{\rm ext}$ & Work by external agent \\
\hline
\end{tabular}
\end{center}

\textbf{Special cases:}
\begin{itemize}[leftmargin=1.5em]
  \item No friction, no external work: $K_f + U_f = K_i + U_i$.
  \item With friction, no external work: $K_f + U_f + \Delta E_{\rm th} = K_i + U_i$.
  \item No friction, with external work: $K_f + U_f = K_i + U_i + W_{\rm ext}$.
\end{itemize}

\end{diagbox}

\subsection{Problem-Solving Strategy}

\begin{diagbox}{12.2 Problem-Solving Strategy}

\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Define the system.} Include all objects whose energy changes.
  \item \textbf{Identify initial and final states.} Draw diagrams for each.
  \item \textbf{Choose a reference level} for gravitational PE.
  \item \textbf{Identify energy forms} present at each state: $K$, $U_g$, $U_{\rm sp}$.
  \item \textbf{Check for friction.} Compute $\Delta E_{\rm th} = f_k d$ if present.
  \item \textbf{Check for external work.}  Add $W_{\rm ext}$ if an external agent acts.
  \item \textbf{Apply the master energy equation} and solve.
  \item \textbf{Check units and reasonableness.}
\end{enumerate}

\end{diagbox}

\subsection{Worked Examples: No Friction}

\begin{diagbox}{12.3 Worked Examples: No Friction}

\textbf{Example A --- Ball dropped from rest:}

A 2-kg ball dropped from $h = 5$\,m.  Find speed just before impact.
\[
  mgh = \tfrac{1}{2}mv_f^2 \;\Rightarrow\; v_f = \sqrt{2gh} = \sqrt{2(9.8)(5)} \approx 9.9\text{ m/s}
\]

\vspace{0.4cm}
\textbf{Example B --- Spring launching a block:}

Spring ($k = 800$\,N/m) compressed 0.15\,m releases a 0.5-kg block.  Find the block's speed.
\[
  \tfrac{1}{2}kx^2 = \tfrac{1}{2}mv^2 \;\Rightarrow\; v = x\sqrt{k/m} = 0.15\sqrt{1600} = 6.0\text{ m/s}
\]

\vspace{0.4cm}
\textbf{Example C --- Ball on frictionless ramp:}

3-kg ball from rest at top of 4.0-m-high frictionless ramp.
\[
  v = \sqrt{2gh} = \sqrt{2(9.8)(4.0)} \approx 8.9\text{ m/s}
\]

\end{diagbox}

\subsection{Worked Examples: With Friction}

\begin{diagbox}{12.4 Worked Examples: With Friction}

\textbf{Example D --- Block on rough ramp:}

4-kg block slides 6.0\,m down a $30^\circ$ ramp ($\mu_k = 0.20$) from rest.

$h = 6\sin30^\circ = 3.0$\,m; $n = mg\cos30^\circ = 33.9$\,N; $f_k = (0.20)(33.9) = 6.78$\,N.

$\Delta E_{\rm th} = (6.78)(6.0) = 40.7$\,J.

\[
  \tfrac{1}{2}(4)v^2 = mgh - \Delta E_{\rm th} = 117.6 - 40.7 = 76.9\text{ J}
  \;\Rightarrow\; v = \sqrt{38.45} \approx 6.2\text{ m/s}
\]

\vspace{0.4cm}
\textbf{Example E --- Object stopped by friction:}

1.5-kg box at 5.0\,m/s; $\mu_k = 0.35$.
\[
  \tfrac{1}{2}(1.5)(25) = (0.35)(1.5)(9.8)d \;\Rightarrow\; d \approx 3.6\text{ m}
\]

\end{diagbox}

\subsection{Worked Examples: With External Work}

\begin{diagbox}{12.5 Worked Examples: With External Work}

\textbf{Example F --- Motor lifts a crate:}

Motor does 2400\,J lifting a 50-kg crate 4.0\,m from rest to rest.
\[
  W_{\rm ext} = \Delta U_g + \Delta E_{\rm th} \;\Rightarrow\; 2400 = (50)(9.8)(4.0) + \Delta E_{\rm th}
  \;\Rightarrow\; \Delta E_{\rm th} = 440\text{ J}
\]
Efficiency: $1960/2400 \approx 81.7\%$.

\vspace{0.4cm}
\textbf{Example G --- Cyclist over two hills:}

80-kg cyclist (bike included) coasts over a 5.0-m descent then up a 3.0-m hill.  Friction: $\Delta E_{\rm th} = 120$\,J.  Initial speed: 8.0\,m/s.
\[
  K_f = K_i + Mg\Delta h - \Delta E_{\rm th}
      = \tfrac{1}{2}(80)(64) + (80)(9.8)(2) - 120
      = 2560 + 1568 - 120 = 4008\text{ J}
\]
\[
  v_f = \sqrt{2(4008)/80} \approx 10.0\text{ m/s}
\]

\end{diagbox}

\subsection{Energy Bar Charts}

\begin{diagbox}{12.6 Energy Bar Charts}

\textbf{Energy bar charts} (LOL diagrams) visualize the master energy equation.  Each bar represents an energy form; heights must balance.

\textbf{Drawing procedure:}
\begin{enumerate}[leftmargin=1.5em]
  \item Draw initial and final bar sets separated by a process region.
  \item Label bars: $K$, $U_g$, $U_{\rm sp}$, $E_{\rm th}$.
  \item Include a $W_{\rm ext}$ bar in the process region if needed.
  \item Total initial energy $+$ $W_{\rm ext}$ $=$ total final energy.
\end{enumerate}

\textbf{Example (frictionless ramp):} Initial: tall $U_g$, zero $K$. Final: zero $U_g$, tall $K$ (equal height). \\
\textbf{Example (with friction):} Initial: tall $U_g$. Final: shorter $K$ + positive $E_{\rm th}$. $U_{g,i} = K_f + \Delta E_{\rm th}$.

\end{diagbox}

\subsection{When to Use Energy Methods vs.\ Newton's Laws}

\begin{diagbox}{12.7 When to Use Energy Methods vs.\ Newton's Laws}

\begin{center}
\renewcommand{\arraystretch}{1.8}
\begin{tabular}{|p{4.8cm}|p{4.5cm}|p{4.5cm}|}
\hline
\textbf{Situation} & \textbf{Energy Methods} & \textbf{Newton's Laws} \\
\hline
Known positions; want speed & \textbf{Preferred} -- one equation & Requires kinematics; harder \\
\hline
Non-constant force (spring, curved path) & \textbf{Preferred} -- PE handles it & Requires calculus \\
\hline
Want force or acceleration at a point & Difficult & \textbf{Preferred} \\
\hline
Complex or unknown path & \textbf{Preferred} -- path independence & Very difficult \\
\hline
Need time information & Cannot give time directly & \textbf{Preferred} \\
\hline
Multiple interacting objects & Additive system energy; convenient & Must track all interaction forces \\
\hline
\end{tabular}
\end{center}

\begin{warnbox}
\textbf{Rule of thumb:} Use energy methods when the problem involves \emph{positions and speeds}.  Use Newton's Laws when it involves \emph{forces, accelerations, or time}.
\end{warnbox}

\end{diagbox}

\subsection{Topic 1 --- Torque}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{1\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        A mechanic applies a 60\,N force at the end of a 0.45-m wrench handle,
        directed at $40^\circ$ with respect to the handle.
        Calculate the torque about the bolt.

  \item \textbf{[5 pts --- Difficult]}
        A 2.4-m horizontal beam is pinned at its left end.
        Three forces act on it:
        \begin{itemize}[leftmargin=1.5em]
          \item $F_1 = 80$\,N \emph{downward} at $x = 0.60$\,m,
          \item $F_2 = 120$\,N at $50^\circ$ above horizontal (rightward/upward)
                at $x = 1.50$\,m,
          \item $F_3 = 50$\,N \emph{downward} at $x = 2.40$\,m.
        \end{itemize}
        (a) Find the net torque about the pin (CCW\,$=$\,positive).
        (b) State the resulting rotation direction.

  \item \textbf{[5 pts --- Difficult]}
        A uniform 3.0-m horizontal rod (mass 4.0\,kg) is pivoted at its centre.
        A 12\,N downward force is applied 0.80\,m to the right of the pivot, and
        an unknown perpendicular force $F$ is applied at the right end.
        (a) Find $F$ for rotational equilibrium.
        (b) Find the net torque if $F$ is reduced to 5.0\,N.

  \item \textbf{[5 pts --- Difficult]}
        A disc ($I = 0.60$\,kg$\cdot$m$^2$, radius $r = 0.30$\,m) starts from rest.
        A constant tangential force of 9.0\,N acts at the rim.
        (a) Find the angular acceleration $\alpha$.
        (b) Find $\omega$ after 4.0\,s.
        (c) Find the total angle rotated in 4.0\,s.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{1\Alph*.}]

  \item \textit{Rationale: $\tau = rF\sin\theta$ projects the force onto the
        direction perpendicular to the moment arm.}
        \[
          \tau = rF\sin\theta = (0.45\,\mathrm{m})(60\,\mathrm{N})\sin 40^\circ
               = (27)(0.6428) \approx \mathbf{17.4\text{ N{\cdot}m}}
        \]

  \item \textit{Rationale: Only force components perpendicular to the beam create
        torque about the pin. Resolve $F_2$ vertically. CCW\,$=$\,positive.}
        \begin{align*}
          \tau_1 &= -(80)(0.60) = -48.0\text{ N{\cdot}m}\\
          \tau_2 &= +(120\sin 50^\circ)(1.50)
                  = +(120\times 0.7660)(1.50)
                  = +(91.92)(1.50) = +137.9\text{ N{\cdot}m}\\
          \tau_3 &= -(50)(2.40) = -120.0\text{ N{\cdot}m}\\
          \tau_{\rm net} &= -48.0+137.9-120.0 = \mathbf{-30.1\text{ N{\cdot}m}}
        \end{align*}
        Negative $\Rightarrow$ \textbf{clockwise} rotation.

  \item \textit{Rationale: The rod's weight acts at the pivot, giving zero torque.
        Set $\sum\tau = 0$ about the pivot for part (a); compute the residual for
        part (b).}\\[4pt]
        (a) Moment arms: $r_{12} = 0.80$\,m; $r_F = 1.50$\,m (right end from centre).
        \[
          F(1.50) - (12)(0.80) = 0
          \;\Rightarrow\; F = \frac{9.6}{1.50} = \mathbf{6.4\text{ N}}
        \]
        (b) With $F = 5.0$\,N:
        \[
          \tau_{\rm net} = (5.0)(1.50) - (12)(0.80) = 7.5 - 9.6 = \mathbf{-2.1\text{ N{\cdot}m (CW)}}
        \]

  \item \textit{Rationale: Newton's 2nd law for rotation $\tau = I\alpha$ gives
        $\alpha$; constant-$\alpha$ kinematics give $\omega$ and $\theta$.}\\[4pt]
        (a)
        \[
          \tau = rF = (0.30\,\mathrm{m})(9.0\,\mathrm{N}) = 2.70\text{ N{\cdot}m}
          \;\Rightarrow\; \alpha = \frac{\tau}{I} = \frac{2.70}{0.60} = \mathbf{4.5\text{ rad/s}^2}
        \]
        (b) $\omega = \omega_0 + \alpha t = 0 + (4.5)(4.0) = \mathbf{18.0\text{ rad/s}}$

        (c) $\theta = \tfrac{1}{2}\alpha t^2 = \tfrac{1}{2}(4.5)(4.0)^2
                    = \tfrac{1}{2}(4.5)(16) = \mathbf{36.0\text{ rad}}$

\end{enumerate}
\end{keybox}

\subsection{Topic 2 --- Center of Gravity}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{2\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        Four masses on a rod: 3.0\,kg at $x=0$; 5.0\,kg at $x=1.2$\,m;
        2.0\,kg at $x=2.8$\,m; 4.0\,kg at $x=4.0$\,m.
        Find $x_{\rm CG}$.

  \item \textbf{[5 pts --- Difficult]}
        A uniform 2.0-m, 8.0-kg plank has a 12\,kg block at $x=0$ and a
        5.0\,kg block at $x=1.6$\,m.
        (a) Find $x_{\rm CG}$ of the system.
        (b) Two supports at $x=0.60$\,m and $x=1.80$\,m: find both normal forces.

  \item \textbf{[5 pts --- Difficult]}
        An L-shaped thin plate: Piece~1 (3.0\,kg) spans $0\le x\le2.0$\,m,
        $0\le y\le1.0$\,m; Piece~2 (2.0\,kg) spans $0\le x\le1.0$\,m,
        $1.0\le y\le3.0$\,m.
        Find the CG of the composite plate.

  \item \textbf{[5 pts --- Difficult]}
        A uniform disc (6.0\,kg, radius 0.50\,m) has a circular hole
        (radius 0.10\,m) cut out, centred at $x=0.30$\,m from the disc centre.
        Using the negative-mass method, find the new $x_{\rm CG}$.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{2\Alph*.}]

  \item \textit{Rationale: $x_{\rm CG}=\tfrac{\sum m_i x_i}{\sum m_i}$ is the
        mass-weighted average position.}
        \[
          x_{\rm CG}=\frac{(3.0)(0)+(5.0)(1.2)+(2.0)(2.8)+(4.0)(4.0)}{14.0}
                    =\frac{0+6.0+5.6+16.0}{14.0}
                    =\frac{27.6}{14.0}\approx\mathbf{1.97\text{ m}}
        \]

  \item \textit{Rationale: Treat each piece as a point mass at its own geometric
        centre. Use $\sum\tau=0$ about one support to isolate the other normal force.}\\[4pt]
        (a) Plank CG at $x=1.0$\,m:
        \[
          x_{\rm CG}=\frac{(12)(0)+(8)(1.0)+(5)(1.6)}{25}
                    =\frac{0+8.0+8.0}{25}=\frac{16.0}{25}=\mathbf{0.64\text{ m}}
        \]
        (b) $\sum F_y=0$: $N_1+N_2=(25)(9.8)=245$\,N.\\
        Torques about $x=0.60$\,m (CCW\,$=$\,pos.):
        \begin{align*}
          +(12)(9.8)(0.60)&-(8)(9.8)(0.40)-(5)(9.8)(1.00)+N_2(1.20)=0\\
          70.56&-31.36-49.0+1.20N_2=0\\
          1.20N_2&=9.80\;\Rightarrow\;N_2=\frac{9.80}{1.20}\approx\mathbf{8.2\text{ N}}
        \end{align*}
        $N_1=245-8.2\approx\mathbf{237\text{ N}}$

  \item \textit{Rationale: Locate each piece's CG at its geometric centre; apply
        the weighted-average formula for $x$ and $y$ independently.}\\[4pt]
        Piece-1 CG: $(1.0,\,0.5)$;\quad Piece-2 CG: $(0.5,\,2.0)$.
        \begin{align*}
          x_{\rm CG}&=\frac{(3.0)(1.0)+(2.0)(0.5)}{5.0}
                     =\frac{3.0+1.0}{5.0}=\mathbf{0.80\text{ m}}\\[4pt]
          y_{\rm CG}&=\frac{(3.0)(0.5)+(2.0)(2.0)}{5.0}
                     =\frac{1.5+4.0}{5.0}=\frac{5.5}{5.0}=\mathbf{1.10\text{ m}}
        \end{align*}

  \item \textit{Rationale: Subtract the cut-out disc (treated as negative mass)
        from the intact disc using the CG formula.}\\[4pt]
        $\sigma=M/(\pi R^2)=6.0/(\pi\times(0.50)^2)=6.0/(\pi\times0.25)=6.0/0.7854\approx7.639$\,kg/m$^2$;\quad
        $m_h=\sigma\pi r_h^2=7.639\times\pi\times(0.10)^2=7.639\times0.03142\approx0.2400$\,kg.
        \[
          x_{\rm CG}=\frac{M(0)-m_h(0.30)}{M-m_h}
                    =\frac{(6.0)(0)-(0.2400)(0.30)}{6.0-0.2400}
                    =\frac{-0.0720}{5.760}\approx\mathbf{-0.0125\text{ m}}
        \]
        The CG shifts $12.5$\,mm away from the hole.

\end{enumerate}
\end{keybox}

        A vertical wire supports the right end.
        A 10\,kg box sits 0.80\,m from the wall.
        (a) Find wire tension $T_w$.
        (b) Find the hinge reaction force (magnitude and direction).

  \item \textbf{[5 pts --- Difficult]}
        A 6.0-m, 25-kg uniform boom is hinged at the wall (lower left) and held
        by a \emph{horizontal} cable at the right end.
        The boom makes $55^\circ$ above horizontal.  A 60-kg load hangs from
        the right end.
        (a) Find the cable tension.
        (b) Find the hinge force magnitude.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{3\Alph*.}]

  \item \textit{Rationale: Choose the hinge as pivot to eliminate unknown hinge
        forces from the torque equation; then use $\sum F=0$ for the components.}
        \[
          T\sin35^\circ(5.0)=(20)(9.8)(2.5)+(50)(9.8)(5.0)=490+2450=2940\text{ N{\cdot}m}
        \]
        \[
          T=\frac{2940}{5.0\times\sin35^\circ}=\frac{2940}{5.0\times0.5736}
           =\frac{2940}{2.868}\approx\mathbf{1025\text{ N}}
        \]
        \[
          H_x=T\cos35^\circ=(1025)(0.8192)\approx\mathbf{840\text{ N (toward wall)}}
        \]
        \[
          H_y=(20+50)(9.8)-T\sin35^\circ=686-(1025)(0.5736)=686-588\approx\mathbf{98\text{ N (up)}}
        \]

  \item \textit{Rationale: Pivot at the base of the ladder; the moment arm for each
        load equals (distance along ladder)$\times\cos70^\circ$.}
        \begin{align*}
          N_w\times(4.5\sin70^\circ)
            &=(14)(9.8)(2.0\cos70^\circ)+(80)(9.8)(3.5\cos70^\circ)\\
          N_w\times(4.5\times0.9397)
            &=(14\times9.8\times2.0\times0.3420)+(80\times9.8\times3.5\times0.3420)\\
          N_w\times4.229
            &=93.74+939.1=1032.8\\
          N_w&=\frac{1032.8}{4.229}\approx\mathbf{244\text{ N}}
        \end{align*}
        $\sum F_y$: $N_f=(94)(9.8)=\mathbf{921\text{ N}}$;\quad
        $\sum F_x$: $f=N_w=\mathbf{244\text{ N}}$.\\[3pt]
        Check: $f_{\max}=\mu_s N_f=(0.38)(921)=350\text{ N}>244\text{ N}\;\checkmark$

  \item \textit{Rationale: Pivot at the hinge to find $T_w$ directly; then close
        the force balance to find the hinge reaction.}
        \[
          T_w(1.2)=(4.0)(9.8)(0.60)+(10)(9.8)(0.80)=23.52+78.40=101.9\text{ N{\cdot}m}
        \]
        \[
          T_w=\frac{101.9}{1.2}\approx\mathbf{84.9\text{ N (upward)}}
        \]
        $\sum F_y$: $H_y=(14)(9.8)-84.9=137.2-84.9\approx52.3\text{ N (up)}$;
        $H_x=0$.
        \[
          |\vec{H}|=52.3\text{ N (vertically upward)}
        \]

  \item \textit{Rationale: The horizontal cable's torque arm about the hinge is
        the boom's vertical height. Pivot at the hinge.}
        \begin{align*}
          h_{\rm end}&=6.0\sin55^\circ=6.0\times0.8192=4.915\text{ m}\\
          T_c\times4.915
            &=(25)(9.8)(3.0\cos55^\circ)+(60)(9.8)(6.0\cos55^\circ)\\
            &=(25\times9.8\times3.0\times0.5736)+(60\times9.8\times6.0\times0.5736)\\
            &=420.7+2023.3=2444\text{ N{\cdot}m}
        \end{align*}
        \[
          T_c=\frac{2444}{4.915}\approx\mathbf{497\text{ N}}
        \]
        $H_x=497$\,N;\quad $H_y=(25+60)(9.8)=833$\,N.
        \[
          |\vec{H}|=\sqrt{497^2+833^2}=\sqrt{247\,009+693\,889}
                  =\sqrt{940\,898}\approx\mathbf{970\text{ N}}
        \]

\end{enumerate}
\end{keybox}

  \item \textbf{[5 pts --- Difficult]}
        A 0.50-m-tall, 0.20-m-wide uniform block sits on a turntable accelerating
        at $\alpha_T=2.0$\,rad/s$^2$.  The block's CG is 0.35\,m from the axis

        (a) Find the critical $\omega_c$ at which the block tips about its outer edge.
        (b) Find $\omega_{\rm slide}$.
        (c) Which occurs first?

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{4\Alph*.}]

  \item \textit{Rationale: Tipping occurs when the CG projects directly over the
        lower edge: $\tan\theta_{\rm tip}=(W/2)/(H/2)$.}
        \[
          \tan\theta_{\rm tip}=\frac{0.20}{0.45}=0.4444
          \;\Rightarrow\;\theta_{\rm tip}=\arctan(0.4444)\approx\mathbf{24.0^\circ}
        \]

  \item \textit{Rationale: Tipping angle from geometry of CG vs.\ edge; sliding
        angle from the friction condition $\tan\alpha=\mu_s$.}\\[4pt]
        (a) CG at $H/2=0.30$\,m; $R=0.15$\,m from lower edge:
        \[
          \tan\alpha_{\rm tip}=\frac{0.15}{0.30}=0.500
          \;\Rightarrow\;\alpha_{\rm tip}\approx\mathbf{26.6^\circ}
        \]
        (b) $\tan\alpha_{\rm slide}=0.28\;\Rightarrow\;\alpha_{\rm slide}\approx\mathbf{15.6^\circ}$

        (c) $15.6^\circ<26.6^\circ\;\Rightarrow$ the cylinder \textbf{slides before tipping}.

  \item \textit{Rationale: For tipping, torques about the front lower edge;
        for sliding, $F=\mu_s mg$.}\\[4pt]
        (a)
        \[
          F_{\rm tip}(1.10)=mg(0.30)=(80)(9.8)(0.30)=235.2\text{ N{\cdot}m}
          \;\Rightarrow\;F_{\rm tip}=\frac{235.2}{1.10}\approx\mathbf{214\text{ N}}
        \]
        (b) $F_{\rm slide}=\mu_s mg=(0.45)(80)(9.8)=\mathbf{353\text{ N}}$

        (c) $214\text{ N}<353\text{ N}\;\Rightarrow$ the crate \textbf{tips first}.

  \item \textit{Rationale: On a turntable the centripetal force $m\omega^2 r$ acts
        horizontally; tipping occurs when this ``outward push'' rotates the resultant
        past the outer edge; sliding when $m\omega^2 r=\mu_s mg$.}\\[4pt]
        (a)
        \[
          \frac{m\omega_c^2 r}{mg}=\frac{W/2}{H/2}=\frac{0.10}{0.25}=0.400
          \;\Rightarrow\;\omega_c^2=\frac{(0.400)(9.8)}{0.35}=11.20
          \;\Rightarrow\;\omega_c\approx\mathbf{3.35\text{ rad/s}}
        \]
        (b)
        \[
          \omega_{\rm slide}=\sqrt{\frac{\mu_s g}{r}}=\sqrt{\frac{(0.55)(9.8)}{0.35}}
                            =\sqrt{15.40}\approx\mathbf{3.92\text{ rad/s}}
        \]
        (c) $3.35<3.92\;\Rightarrow$ the block \textbf{tips first}.

\end{enumerate}
\end{keybox}

  \item \textbf{[5 pts --- Difficult]}
        A spring ($k=500$\,N/m) hangs from the ceiling with a 2.0\,kg mass.

        (a) Equilibrium extension; (b) amplitude;
        (c) maximum speed; (d) maximum elastic PE.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{5\Alph*.}]

  \item \textit{Rationale: At maximum height $v=0$; all elastic PE converts to
        gravitational PE: $\tfrac{1}{2}kx^2=mgh$.}
        \[
          h=\frac{kx^2}{2mg}
           =\frac{(360)(0.14)^2}{2(0.25)(9.8)}
           =\frac{360\times0.0196}{4.90}
           =\frac{7.056}{4.90}\approx\mathbf{1.44\text{ m}}
        \]

  \item \textit{Rationale: Series springs share the same tension but different
        extensions; $1/k_{\rm eff}=1/k_1+1/k_2$.}\\[4pt]
        (a)
        \[
          \frac{1}{k_{\rm eff}}=\frac{1}{250}+\frac{1}{400}
                               =0.004000+0.002500=0.006500
          \;\Rightarrow\;k_{\rm eff}=\frac{1}{0.006500}\approx\mathbf{153.8\text{ N/m}}
        \]
        (b)
        \[
          x_{\rm tot}=\frac{mg}{k_{\rm eff}}=\frac{(5.0)(9.8)}{153.8}
                     =\frac{49.0}{153.8}\approx\mathbf{0.319\text{ m}}
        \]
        (c) Both carry $F=mg=(5.0)(9.8)=49.0$\,N:
        \[
          x_1=\frac{49.0}{250}=\mathbf{0.196\text{ m}};\quad
          x_2=\frac{49.0}{400}=\mathbf{0.123\text{ m}}
          \quad(0.196+0.123=0.319\;\checkmark)
        \]
        (d)
        \[
          U_1=\tfrac{1}{2}(250)(0.196)^2=125\times0.03842=\mathbf{4.80\text{ J}};\quad
          U_2=\tfrac{1}{2}(400)(0.123)^2=200\times0.01513=\mathbf{3.03\text{ J}}
        \]

  \item \textit{Rationale: Parallel springs have the same extension; add constants.
        Series combo carries the same force throughout; the parallel springs split
        the force proportionally to their $k$ values.}\\[4pt]
        (a)
        \[
          k_{\rm par}=200+300=500\text{ N/m};\quad
          \frac{1}{k_{\rm eff}}=\frac{1}{500}+\frac{1}{600}
                               =\frac{6+5}{3000}=\frac{11}{3000}
          \;\Rightarrow\;k_{\rm eff}=\frac{3000}{11}\approx\mathbf{272.7\text{ N/m}}
        \]
        (b)
        \[
          x_{\rm tot}=\frac{(4.0)(9.8)}{272.7}=\frac{39.2}{272.7}\approx\mathbf{0.144\text{ m}}
        \]
        (c) $F_C=mg=39.2$\,N.\quad $A$ and $B$ share 39.2\,N:
        \[
          F_A=\frac{200}{500}(39.2)=\mathbf{15.7\text{ N}};\quad
          F_B=\frac{300}{500}(39.2)=\mathbf{23.5\text{ N}}
          \quad(15.7+23.5=39.2\;\checkmark)
        \]

  \item \textit{Rationale: At equilibrium $kx_0=mg$. SHM amplitude equals the
        pull-down distance; max speed occurs at equilibrium (all PE $\to$ KE in SHM).}\\[4pt]
        (a)
        \[
          x_0=\frac{mg}{k}=\frac{(2.0)(9.8)}{500}=\frac{19.6}{500}=\mathbf{0.0392\text{ m}}
        \]
        (b) Amplitude $A=\mathbf{0.20\text{ m}}$

        (c)
        \[
          v_{\max}=A\omega=A\sqrt{\frac{k}{m}}
                 =(0.20)\sqrt{\frac{500}{2.0}}
                 =(0.20)(15.81)\approx\mathbf{3.16\text{ m/s}}
        \]
        (d) Maximum spring PE at lowest point ($x_0+A$ below natural length):
        \[
          x_{\rm bot}=0.0392+0.20=0.2392\text{ m};\quad
          U_{\max}=\tfrac{1}{2}(500)(0.2392)^2=250\times0.05722\approx\mathbf{14.3\text{ J}}
        \]

\end{enumerate}
\end{keybox}

\subsection{Topic 7 --- Work}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{7\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        A 15\,kg box is dragged 6.0\,m along a horizontal floor by a rope at

        Find (a) work by $T$, (b) work by friction, (c) net work.

  \item \textbf{[5 pts --- Difficult]}
        A 3.0\,kg block is pushed up a frictionless $30^\circ$ incline by a
        horizontal force $F$ at constant speed over $d=2.0$\,m.
        (a) Find $F$; (b) find $W_F$, $W_{\rm grav}$, $W_N$;
        (c) verify $W_{\rm net}=0$.

  \item \textbf{[5 pts --- Difficult]}
        A variable force $F(x)=6x^2-4x$\,(N) acts on a 2.0-kg particle moving
        from $x=1.0$\,m to $x=4.0$\,m.
        (a) Find the work done; (b) if $v_0=3.0$\,m/s at $x=1.0$\,m,
        find the speed at $x=4.0$\,m.

  \item \textbf{[5 pts --- Difficult]}
        A 500-kg elevator rises 30\,m at constant speed.
        A 300-kg counterweight descends on the other side of the pulley.
        (a) Net work by gravity on the system;
        (b) work by the motor cable tension;
        (c) minimum motor power if the rise takes 25\,s.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{7\Alph*.}]

  \item \textit{Rationale: $W=Fd\cos\phi$ where $\phi$ is the angle between force
        and displacement. Friction opposes motion ($\phi=180^\circ$ for friction).}\\[4pt]
        (a)
        \[
          W_T=(70)(6.0)\cos28^\circ=(70)(6.0)(0.8829)=\mathbf{370.8\text{ J}}
        \]
        (b) $N=mg-T\sin28^\circ=(15)(9.8)-70(0.4695)=147.0-32.9=114.1$\,N;
            $f_k=(0.20)(114.1)=22.82$\,N.
        \[
          W_f=-(22.82)(6.0)=\mathbf{-136.9\text{ J}}
        \]
        (c) $W_{\rm net}=370.8-136.9=\mathbf{234\text{ J}}$

  \item \textit{Rationale: Constant speed $\Rightarrow\sum F=0$ along incline.
        Each work term is force $\times$ displacement $\times\cos\phi$.}\\[4pt]
        (a)
        \[
          F\cos30^\circ=mg\sin30^\circ
          \;\Rightarrow\;F=(3.0)(9.8)\tan30^\circ=(29.4)(0.5774)=\mathbf{16.97\text{ N}}
        \]
        (b)
        \[
          W_F=(16.97)(2.0)\cos30^\circ=(16.97)(2.0)(0.8660)\approx\mathbf{+29.4\text{ J}}
        \]
        \[
          W_{\rm grav}=-(3.0)(9.8)(2.0)\sin30^\circ=-(29.4)(2.0)(0.500)=\mathbf{-29.4\text{ J}}
        \]
        \[
          W_N=0\text{ J}\quad(\vec{N}\perp\vec{d})
        \]
        (c) $W_{\rm net}=29.4-29.4+0=\mathbf{0\text{ J}}\;\checkmark$

  \item \textit{Rationale: For variable force, $W=\int F(x)\,dx$; then apply
        the Work--Energy Theorem.}\\[4pt]
        (a)
        \begin{align*}
          W&=\int_1^4(6x^2-4x)\,dx=\Bigl[2x^3-2x^2\Bigr]_1^4\\
           &=\bigl(2(64)-2(16)\bigr)-\bigl(2(1)-2(1)\bigr)=(128-32)-(0)=\mathbf{96\text{ J}}
        \end{align*}
        (b)
        \[
          96=\tfrac{1}{2}(2.0)v_f^2-\tfrac{1}{2}(2.0)(3.0)^2=v_f^2-9
          \;\Rightarrow\;v_f=\sqrt{105}\approx\mathbf{10.25\text{ m/s}}
        \]

  \item \textit{Rationale: At constant speed the motor must compensate for the
        net gravitational work done against the system.}\\[4pt]
        (a)
        \begin{align*}
          W_{\rm grav,elev}&=-(500)(9.8)(30)=-147\,000\text{ J}\\
          W_{\rm grav,CW}  &=+(300)(9.8)(30)=+88\,200\text{ J}\\
          W_{\rm grav,net} &=88\,200-147\,000=\mathbf{-58\,800\text{ J}}
        \end{align*}
        (b) $W_{\rm motor}=-W_{\rm grav,net}=\mathbf{+58\,800\text{ J}}$

        (c)
        \[
          P_{\min}=\frac{58\,800}{25}=\mathbf{2352\text{ W}}
        \]

\end{enumerate}
\end{keybox}

  \item \textbf{[5 pts --- Difficult]}

  \item \textbf{[5 pts --- Difficult]}

        How far from the ramp base does it stop?

\end{enumerate}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{8\Alph*.}]

  \item \textit{Rationale: $W_{\rm net}=\Delta K$; the car stops so $K_f=0$.}
        \[
          -F_{\rm brake}\,d=0-\tfrac{1}{2}mv^2
          \;\Rightarrow\;d=\frac{mv^2}{2F_{\rm brake}}
          =\frac{(1200)(25)^2}{2(8000)}
          =\frac{750\,000}{16\,000}\approx\mathbf{46.9\text{ m}}
        \]

  \item \textit{Rationale: Apply $W_{\rm net}=\Delta K$ to the combined two-body
        system; gravity on $B$ does positive work, friction on $A$ does negative work.}
        \begin{align*}
          W_{\rm net}&=(2.5)(9.8)(1.2)-(0.30)(4.0)(9.8)(1.2)\\
                     &=29.4-14.11=15.29\text{ J}
        \end{align*}
        \[
          \tfrac{1}{2}(6.5)v^2=15.29
          \;\Rightarrow\;v^2=\frac{2\times15.29}{6.5}=4.704
          \;\Rightarrow\;v\approx\mathbf{2.17\text{ m/s}}
        \]

  \item \textit{Rationale: Only gravity does work (no air resistance). Height
        fallen is 20\,m regardless of horizontal travel.}
        \[
          W_{\rm net}=mgh=(0.40)(9.8)(20)=78.4\text{ J}
        \]
        \[
          \tfrac{1}{2}mv_f^2=\tfrac{1}{2}(0.40)(8.0)^2+78.4=12.8+78.4=91.2\text{ J}
        \]
        \[
          v_f=\sqrt{\frac{2(91.2)}{0.40}}=\sqrt{456}\approx\mathbf{21.4\text{ m/s}}
        \]
        \emph{Check}: $v_y^2=2(9.8)(20)=392$;\;$v_x^2=(8.0)^2=64$;\;$v_f=\sqrt{64+392}=\sqrt{456}=21.4$\,m/s\;\checkmark

  \item \textit{Rationale: Frictionless ramp converts all gravitational PE to KE at
        the base; friction then dissipates that KE on the flat.}
        \[
          K_{\rm base}=(2.0)(9.8)(3.5)=68.6\text{ J};\quad
          f_k=(0.25)(2.0)(9.8)=4.90\text{ N}
        \]
        \[
          d=\frac{K_{\rm base}}{f_k}=\frac{68.6}{4.90}=\mathbf{14.0\text{ m}}
        \]

\end{enumerate}
\end{keybox}

\subsection{Topic 9 --- Potential Energy}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{9\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        A spring ($k=250$\,N/m) compressed 0.16\,m launches a 0.30\,kg ball
        vertically.  Find the max height.

  \item \textbf{[5 pts --- Difficult]}
        A 2.0-kg block on a frictionless surface is attached to a spring
        ($k=600$\,N/m) at natural length and given $v_0=4.0$\,m/s.
        (a) Find $x_{\rm max}$; (b) find $v$ at $x=0.12$\,m;
        (c) fraction of initial KE stored as spring PE at $x=0.12$\,m.

  \item \textbf{[5 pts --- Difficult]}
        Pendulum: $L=1.4$\,m, $m=0.60$\,kg, released from $\theta=40^\circ$.
        (a) Speed at bottom; (b) tension at bottom;
        (c) angle at which speed equals half the max speed.

  \item \textbf{[5 pts --- Difficult]}
        A 1.5-kg block compressed 0.15\,m against a spring ($k=800$\,N/m) on a
        frictionless floor is released onto a rough $25^\circ$ incline ($\mu_k=0.18$).
        How far up the incline does it travel?

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{9\Alph*.}]

  \item \textit{Rationale: All elastic PE $\to$ gravitational PE at peak; $v=0$
        there: $\tfrac{1}{2}kx^2=mgh$.}
        \[
          h=\frac{kx^2}{2mg}=\frac{(250)(0.16)^2}{2(0.30)(9.8)}
           =\frac{250\times0.0256}{5.88}=\frac{6.40}{5.88}\approx\mathbf{1.09\text{ m}}
        \]

  \item \textit{Rationale: Frictionless surface $\Rightarrow$ energy conserved:
        $K+U_{\rm sp}=\text{const}$.}\\[4pt]
        (a) At max compression $v=0$: all KE $\to$ spring PE.
        \[
          x_{\rm max}=v_0\sqrt{\frac{m}{k}}=4.0\sqrt{\frac{2.0}{600}}
                     =4.0\times0.05774\approx\mathbf{0.231\text{ m}}
        \]
        (b)
        \[
          \tfrac{1}{2}mv_1^2=\tfrac{1}{2}(2.0)(4.0)^2-\tfrac{1}{2}(600)(0.12)^2
                            =16.0-4.32=11.68\text{ J}
          \;\Rightarrow\;v_1=\sqrt{11.68}\approx\mathbf{3.42\text{ m/s}}
        \]
        (c) $4.32/16.0=0.270\;\Rightarrow\;\mathbf{27.0\%}$

  \item \textit{Rationale: Height drop from angle $\theta$ is $L(1-\cos\theta)$;
        at the bottom apply radial Newton's 2nd law for tension.}\\[4pt]
        (a) $\Delta h=(1.4)(1-\cos40^\circ)=(1.4)(0.2340)=0.3276$\,m
        \[
          v=\sqrt{2g\Delta h}=\sqrt{2(9.8)(0.3276)}=\sqrt{6.421}\approx\mathbf{2.53\text{ m/s}}
        \]
        (b)
        \[
          T=m\!\left(g+\frac{v^2}{L}\right)
           =(0.60)\!\left(9.8+\frac{6.421}{1.4}\right)
           =(0.60)(9.8+4.586)
           =(0.60)(14.39)\approx\mathbf{8.63\text{ N}}
        \]
        (c) $v=v_{\max}/2=1.265$\,m/s:
        \[
          \cos\theta=1-\frac{v^2}{2gL}=1-\frac{(1.265)^2}{2(9.8)(1.4)}
                   =1-\frac{1.600}{27.44}=1-0.0583=0.9417
          \;\Rightarrow\;\theta\approx\mathbf{19.7^\circ}
        \]

  \item \textit{Rationale: Spring PE converts to KE on the flat, then is dissipated
        by friction and gravity as the block climbs. Set $K_f=0$.}
        \[
          U_{\rm sp}=\tfrac{1}{2}(800)(0.15)^2=400\times0.0225=9.0\text{ J}
        \]
        \[
          9.0=mgd(\sin25^\circ+\mu_k\cos25^\circ)
             =(1.5)(9.8)d(0.4226+0.18\times0.9063)
             =14.7d(0.5857)=8.609d
        \]
        \[
          d=\frac{9.0}{8.609}\approx\mathbf{1.05\text{ m}}
        \]

\end{enumerate}
\end{keybox}

\subsection{Topic 10 --- Conservation of Energy}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{10\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        A 5.0-kg ball released from rest at $h_i=8.0$\,m on a frictionless ramp.
        Find its speed at $h_f=2.5$\,m.

  \item \textbf{[5 pts --- Difficult]}
        A 2.0-kg block slides 5.0\,m down a $40^\circ$ incline ($\mu_k=0.15$),
        then contacts a spring ($k=1200$\,N/m) on a horizontal surface
        ($\mu_k=0.20$).  Find the maximum spring compression.

  \item \textbf{[5 pts --- Difficult]}
        A 0.50-kg ball is thrown upward at $v_0=18$\,m/s.  Air resistance provides
        a constant opposing force of 1.2\,N.
        (a) Find the maximum height;
        (b) find the return speed;
        (c) compare KE at launch and on return.

  \item \textbf{[5 pts --- Difficult]}
        A 3.0-kg block slides from rest down a frictionless ramp from $h=4.0$\,m,
        compresses a spring ($k=2000$\,N/m) at the base, then bounces back and is
        caught by a sticky pad that absorbs 5.0\,J.
        (a) Find the maximum spring compression;
        (b) find the height $H_f$ at which the block is caught.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{10\Alph*.}]

  \item \textit{Rationale: No friction $\Rightarrow$ mechanical energy is conserved:
        $K_i+U_i=K_f+U_f$. With $v_i=0$, all initial PE converts to KE and
        remaining PE.}
        \[
          v=\sqrt{2g(h_i-h_f)}=\sqrt{2(9.8)(8.0-2.5)}=\sqrt{2\times9.8\times5.5}
           =\sqrt{107.8}\approx\mathbf{10.4\text{ m/s}}
        \]

  \item \textit{Rationale: Gravitational PE minus incline friction loss gives KE at
        the ramp base; that KE minus horizontal friction loss goes into spring PE.}\\[4pt]
        \textbf{Step 1 --- KE at ramp base:}
        \[
          \Delta h=(5.0)\sin40^\circ=(5.0)(0.6428)=3.214\text{ m}
        \]
        \[
          N_1=mg\cos40^\circ=(2.0)(9.8)(0.7660)=15.01\text{ N}
        \]
        \[
          W_{f1}=(0.15)(15.01)(5.0)=11.26\text{ J}
        \]
        \[
          K_1=(2.0)(9.8)(3.214)-11.26=62.99-11.26=51.73\text{ J}
        \]
        \textbf{Step 2 --- spring compression:}
        \[
          51.73=600\,x_c^2+(0.20)(2.0)(9.8)x_c=600\,x_c^2+3.92\,x_c
        \]
        \[
          600\,x_c^2+3.92\,x_c-51.73=0
        \]
        \[
          x_c=\frac{-3.92+\sqrt{3.92^2+4(600)(51.73)}}{1200}
             =\frac{-3.92+\sqrt{124\,167}}{1200}
             =\frac{-3.92+352.4}{1200}\approx\mathbf{0.290\text{ m}}
        \]

  \item \textit{Rationale: Air resistance does negative work both up and down,
        reducing total mechanical energy each way.}\\[4pt]
        (a) Net upward force opposes motion: $(mg+F_{\rm air})H=\tfrac{1}{2}mv_0^2$
        \[
          H=\frac{81.0}{(0.50)(9.8)+1.2}=\frac{81.0}{6.10}\approx\mathbf{13.3\text{ m}}
        \]
        (b) Net downward force drives return:
        \[
          \tfrac{1}{2}mv_{\rm ret}^2=(mg-F_{\rm air})H=\bigl((0.50)(9.8)-1.2\bigr)(13.3)
          =(3.70)(13.3)=49.21\text{ J}
        \]
        \[
          v_{\rm ret}=\sqrt{\frac{2(49.21)}{0.50}}=\sqrt{196.8}\approx\mathbf{14.0\text{ m/s}}
        \]
        (c) $K_{\rm launch}=81.0$\,J;\quad$K_{\rm return}=49.2$\,J;\quad
            return KE is $49.2/81.0\approx60.7\%$ of launch KE.

  \item \textit{Rationale: Track each energy conversion step: gravity $\to$ KE
        $\to$ spring PE $\to$ KE again, then subtract the pad collision loss.}\\[4pt]
        (a) Frictionless ramp; $K_{\rm base}=mgh=(3.0)(9.8)(4.0)=117.6$\,J.
        \[
          x_c=\sqrt{\frac{2K_{\rm base}}{k}}=\sqrt{\frac{2(117.6)}{2000}}
             =\sqrt{0.1176}\approx\mathbf{0.343\text{ m}}
        \]
        (b) Spring returns full $117.6$\,J; pad absorbs 5.0\,J:
        \[
          K_f=117.6-5.0=112.6\text{ J};\quad
          H_f=\frac{K_f}{mg}=\frac{112.6}{29.4}\approx\mathbf{3.83\text{ m}}
        \]

\end{enumerate}
\end{keybox}

\section{Bonus Topic: Ski Slope Problems}

\begin{warnbox}
\textbf{Ski Slope Scenario.}
A skier navigates a mountain course: a steep descent, a flat transition, a
jump ramp, and a snowdrift.  These questions integrate work, energy,
friction, and projectile motion.  Each question is worth \textbf{5 points}.
\end{warnbox}

\subsection{Topic 11 --- Ski Slope: Energy on an Incline}

\begin{planbox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{11\Alph*.}]

  \item \textbf{[5 pts --- Standard]}
        A 70-kg skier starts from rest on a 120-m slope at $30^\circ$
        ($\mu_k=0.08$).
        (a) Find the net force along the slope.
        (b) Find the speed at the bottom.

  \item \textbf{[5 pts --- Difficult]}
        After reaching the bottom from 11A at speed $v_1$, the skier travels
        80\,m on a flat section ($\mu_k=0.08$), then takes a frictionless 15-m
        ramp at $25^\circ$.
        (a) Speed at the end of the flat section.
        (b) Launch speed at the top of the ramp.
        (c) Maximum height above the ramp base after the jump.

  \item \textbf{[5 pts --- Difficult]}
        A 70-kg skier starts from rest on a $35^\circ$ slope ($\mu_k=0.10$),
        travels 90\,m, then hits a snowdrift modelled as a spring ($k=8000$\,N/m).
        (a) Speed just before the snowdrift.
        (b) Maximum compression of the snowdrift.

  \item \textbf{[5 pts --- Difficult]}
        A 65-kg skier descends $h=180$\,m vertically over a path of length
        $L=350$\,m ($\mu_k=0.12$ average).
        (a) Show $\bar\theta=\arcsin(h/L)\approx31.0^\circ$.
        (b) Total friction work.
        (c) Finish-line speed.
        (d) Average power dissipated by friction if the run takes 48\,s.

\end{enumerate}
\end{planbox}

\begin{keybox}
\begin{enumerate}[leftmargin=1.5em, label=\textbf{11\Alph*.}]

  \item \textit{Rationale: Along the slope, the gravity component drives the skier
        while kinetic friction resists; apply Newton's 2nd law, then the
        Work--Energy Theorem.}\\[4pt]
        (a)
        \[
          F_{\rm net}=mg(\sin30^\circ-\mu_k\cos30^\circ)
                     =(70)(9.8)(0.500-0.08\times0.8660)
                     =(686)(0.500-0.0693)
                     =(686)(0.4307)\approx\mathbf{295\text{ N}}
        \]
        (b)
        \[
          W_{\rm net}=F_{\rm net}\times d=(295)(120)=35\,400\text{ J}
        \]
        \[
          v_1=\sqrt{\frac{2\times35\,400}{70}}=\sqrt{1011}\approx\mathbf{31.8\text{ m/s}}
        \]

  \item \textit{Rationale: Track KE through each segment using energy conservation
        with friction losses.}\\[4pt]
        (a) Flat friction loss:
        $W_f=(0.08)(70)(9.8)(80)=(0.08)(686)(80)=4390$\,J.
        \[
          K_{\rm flat}=\tfrac{1}{2}(70)(31.8)^2-4390=35\,345-4390=30\,955\text{ J}
        \]
        \[
          v_{\rm flat}=\sqrt{\frac{2\times30\,955}{70}}=\sqrt{884.4}\approx\mathbf{29.7\text{ m/s}}
        \]
        (b) Ramp height: $h_r=(15)\sin25^\circ=(15)(0.4226)=6.339$\,m.
        \[
          K_{\rm launch}=30\,955-(70)(9.8)(6.339)=30\,955-4349=26\,606\text{ J}
        \]
        \[
          v_{\rm launch}=\sqrt{\frac{2\times26\,606}{70}}=\sqrt{760.2}\approx\mathbf{27.6\text{ m/s}}
        \]
        (c) Vertical launch component: $v_{y0}=(27.6)\sin25^\circ=(27.6)(0.4226)=11.66$\,m/s.
        \[
          H=\frac{v_{y0}^2}{2g}=\frac{(11.66)^2}{19.6}=\frac{135.9}{19.6}\approx\mathbf{6.93\text{ m}}
        \]

  \item \textit{Rationale: Net work along the slope gives KE; that KE fully
        compresses the spring ($v=0$ at maximum compression).}\\[4pt]
        (a)
        \[
          W_{\rm net}=(70)(9.8)(90)(\sin35^\circ-\mu_k\cos35^\circ)
                     =(686)(90)(0.5736-0.10\times0.8192)
                     =(686)(90)(0.4917)=30\,361\text{ J}
        \]
        \[
          v=\sqrt{\frac{2\times30\,361}{70}}=\sqrt{867.5}\approx\mathbf{29.4\text{ m/s}}
        \]
        (b)
        \[
          x_c=v\sqrt{\frac{m}{k}}=29.4\sqrt{\frac{70}{8000}}
             =29.4\times0.09354\approx\mathbf{2.75\text{ m}}
        \]

  \item \textit{Rationale: Global energy balance; the average slope angle links
        the vertical drop to the path length. Friction uses the average normal force.}\\[4pt]
        (a)
        \[
          \sin\bar\theta=\frac{h}{L}=\frac{180}{350}=0.5143
          \;\Rightarrow\;\bar\theta=\arcsin(0.5143)\approx\mathbf{31.0^\circ}
        \]
        $\cos\bar\theta=\sqrt{1-0.5143^2}=\sqrt{0.7355}\approx0.8576$

        (b)
        \[
          W_{\rm friction}=\mu_k mg\cos\bar\theta\cdot L
                          =(0.12)(65)(9.8)(0.8576)(350)
                          =(0.12)(546.7)(350)
                          =0.12\times191\,345\approx\mathbf{22\,961\text{ J}}
        \]
        (c)
        \[
          K_f=mgh-W_{\rm friction}=(65)(9.8)(180)-22\,961
             =114\,660-22\,961=91\,699\text{ J}
        \]
        \[
          v_f=\sqrt{\frac{2\times91\,699}{65}}=\sqrt{2821.5}\approx\mathbf{53.1\text{ m/s}}
        \]
        (d)
        \[
          P_{\rm friction}=\frac{W_{\rm friction}}{t}=\frac{22\,961}{48}\approx\mathbf{478\text{ W}}
        \]

\end{enumerate}
\end{keybox}
